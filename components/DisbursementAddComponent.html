import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormControl, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { DisbursService } from 'src/app/providers/project/disbursement/disburs.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Validator } from 'src/app/_helpers/validators';

@Component({
  selector: 'app-disbursement-add',
  templateUrl: './disbursement-add.component.html',
  styleUrls: ['./disbursement-add.component.scss']
})
export class DisbursementAddComponent implements OnInit {
  projectId: any;
  formCreateDisbursement: any;
  vewId: any;
  installment: any;
  tranche: any;
  budgetId: any;
  budgetType: any;
  perShare: any;
  amount: any;
  submitted = false;
  projectInfo: any;
  parms: any;
  disview: any;
  budgetData: any;
  centre: number = 0;
  state: number = 0;
  dis: any;
  data = {
    distribution: [
      {
        type: '60% Centre - 40% State Others',
        id: 1,
        share: [
          {
            central: 60,
            state: 40,
          },
        ],
      },
      {
        type: '90% Centre -10% State',
        id: 2,
        share: [
          {
            central: 90,
            state: 10,
          },
        ],
      },
      {
        type: '100% Centre',
        id: 3,
        share: [
          {
            central: 100,
            state: 0,
          },
        ],
      },
      {
        type: 'Others',
        id: 4,
        share: [
          {
            central: 0,
            state: 0,
          },
        ],
      },
    ],
  };
  shareofData: any;
  installMent: any;
  trancheData: any;
  percentData: any;

  constructor(
    public common: CommonService,
    public formBuilder: FormBuilder,
    public route: Router,
    public project: ProjectService,
    public disburs: DisbursService,
    public budget: BudgetService,
    public validator: Validator,
    public spinner: NgxSpinnerService,

  ) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId();
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError("Please Select/Create a Project");
    }
    this.formCreateDisbursement = this.formBuilder.group({
      budgetType: [null, Validators.required],
      installment:new FormControl(null, [
        Validators.required,
        Validators.pattern("^[0-9]*$"),
      ]),
      perShare: new FormControl(null, [
        Validators.required,
        Validators.pattern("^[0-9]*$"),
      ]),
      tranche: new FormControl(null, [
        Validators.required,
        Validators.pattern("^[0-9]*$"),
      ]),
    })
  }

  ngOnInit(): void {
    this.parms = history.state.id;
    this.disview = history.state.flag;
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    });
    
    if (this.parms) {
      this.disburs.edit(this.parms).subscribe((resp) => {
        
        if (resp) {
          this.dis = resp.data;
          if(this.dis.budgetType>=3){
            this.formCreateDisbursement.get("budgetType")?.setValue(3);
          }
          else{
            this.formCreateDisbursement.get("budgetType")?.setValue(this.dis.budgetType);
          }
          this.formCreateDisbursement.get("installment")?.setValue(this.dis.installment);
          this.formCreateDisbursement.get("perShare")?.setValue(this.dis.perShare);
          this.formCreateDisbursement.get("tranche")?.setValue(this.dis.tranche);
        }
      })
    }
    this.getBudget();
    this.getShareOf();
    this.getInstallment();
    this.getTranche();
    this.getPercent();
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateDisbursement.controls;
  }
  onSubmit(value: any) {
    this.submitted = true;
    if (this.formCreateDisbursement.invalid) {
      return;
    }
    
    if (this.budgetId) {
      this.formCreateDisbursement.addControl('budgetId', new FormControl('', Validators.required));
      this.formCreateDisbursement.controls['budgetId'].setValue(this.budgetId);
      this.formCreateDisbursement.addControl('shareAmount', new FormControl(''));
      this.formCreateDisbursement.controls['shareAmount'].setValue(null);
    }
    

    let docdata = {
      "budgetType": parseInt(value["budgetType"]),
      "id": parseInt(this.parms),
      "installment": value["installment"],
      "perShare": value["perShare"],
      "tranche": value["tranche"],
      "budgetId": parseInt(this.budgetId),
      "shareAmount": null,
    }

    this.SaveDis(docdata);
    
    

  }
  SaveDis(data: any) {
    this.spinner.show();
    this.disburs.addDis(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        
        this.common.showSuccess('Successfully Created Disbursement');
        this.route.navigate(['/dashboard/project/disbursement']);
      }
      else {
        this.common.showError(resp.message)
      }
    });
  }
  onReset(): void {
    this.submitted = false;
    this.formCreateDisbursement.reset();
  }

  getBudget() {
    this.spinner.show();
    this.budget.getBudget(this.projectId).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        
        this.budgetData = resp;
        this.budgetId = resp.data.id;
        this.amount = resp.data.amount;
        this.state = resp.data?.stateShare || 0;
        this.centre = resp.data?.centralShare || 0;
      }
      else {
        this.common.showSuccess('Please fill in the budget details!');
        this.route.navigate(['/dashboard/project/budget']);
      }
    })
  }
  getShareOf(){
    this.common.getShareOfDate().subscribe((resp) => {
      if (resp) {
         
         this.shareofData = resp.data;
      }
   });
  }

  getInstallment(){
    this.common.getInstallmentData().subscribe((resp) => {
      if (resp) {
         
         this.installMent = resp.data;
      }
   });
  }

  getTranche(){
    this.common.getTrancheData().subscribe((resp) => {
      if (resp) {
         
         this.trancheData = resp.data;
      }
   });
  }
  getPercent(){
    this.common.getPercentData().subscribe((resp) => {
      if (resp) {
         
         this.percentData = resp.data;
      }
   });
  }


}
