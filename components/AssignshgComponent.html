import { Component, OnInit } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-assignshg',
  templateUrl: './assignshg.component.html',
  styleUrls: ['./assignshg.component.scss'],
})
export class AssignshgComponent implements OnInit {
  crpId: any;
  crpInfo: any;
  proName: any;
  datas: any;
  newDatas: any;
  crpinput: any;
  crpidinput: any;
  geoDatas: any;
  role:any;

  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public spinner: NgxSpinnerService,
    public geoData: GeoService
  ) {}
  filterForm = this.formBuilder.group({
    id: [null],
    name: [null],
  });

  ngOnInit(): void {
    this.access();
    this.spinner.hide();
    this.crpId = this.common.getCrpId();
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    });
    this.geoData.getGeoCoverage(0, 1, this.crpId).subscribe((resp) => {
      this.geoDatas = resp.data?.geoCoverageList;
    })
    this.getListShg();
  }
  access() {
    this.role = this.common.screenRole("assignshg");
  }

  getListShg() {
    this.spinner.show();
    this.manager.getAllShg(this.crpId).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.datas = resp.data;
        this.newDatas = resp.data;
        this.common.trigger('updateHeaderDetails',[{
          name : 'Project ID' ,
          value : this.crpInfo.id 
        },{
          name : "Project Name" ,
          value : this.crpInfo.name
        },{
          name : 'State' ,
          value : this.geoDatas[0]?.stateName 
        }])
      }
    });
  }
  getShgNameList(name: any) {
    let shgModelList = name;
    if (shgModelList != null) {
      shgModelList = shgModelList.filter(
        (item: { selected: any }) => item.selected == true
      );
      return shgModelList.length;
    }
  }
  clearFilter() {
    this.filterForm.get('name')?.setValue(null);
    this.filterForm.get('id')?.setValue(null);
    this.datas = this.newDatas;
    this.crpinput = undefined;
    this.crpidinput = undefined;
    this.common.closeActiveFilter();
  }
  edit(i: any, group:any) {
    this.route.navigateByUrl('/dashboard/crp/add-assignshg', {
      state: { flag: 'edit', index: i, shgModelList: group },
    });
  }
  view(i: any, group:any) {
    this.route.navigateByUrl('/dashboard/crp/add-assignshg', {
      state: { flag: 'view', index: i, shgModelList: group },
    });
  }
  add(i: any, group: any) {
    if (group.shgResponseModel == null) {
      this.common.showError('The selected Block ID does not have any SHGs.');
    } else {
      this.route.navigateByUrl('/dashboard/crp/add-assignshg', {
        state: { flag: 'add', index: i, shgModelList: group },
      });
    }
  }
  delete(pid: any) {
    if (confirm('Are you sure you want to delete this record?')) {
      this.manager.deletShg(pid, this.crpId).subscribe((resp) => {
        if (resp) {
          this.common.showSuccess('Successfully Deleted');
          this.getListShg();
        }
      });
    }
  }
  onSubmit(value: any): void {
    let crpdata = [];
    let crpidData = [];

    if (value['name'] != undefined) {
      let crpinputLow = value['name'];
      crpdata = this.newDatas.filter((crps: { crpName: any | any[] }) =>
        crps.crpName.includes(crpinputLow)
      );
      if (value['id'] != undefined) {
        crpidData = crpdata.filter(
          (item: { personId: any }) => item.personId == value['id']
        );
        this.datas = crpidData;
      } else {
        this.datas = crpdata;
      }
    }
    this.common.closeActiveFilter();
  }
}
