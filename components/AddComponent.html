import { Component, ElementRef, OnInit, ViewChild } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { CommonService } from 'src/app/providers/common.service';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { NgxSpinnerService } from "ngx-spinner";


@Component({
  selector: 'app-add',
  templateUrl: './add.component.html',
  styleUrls: []
})
export class AddComponent implements OnInit {
  formCreateGeoCoverage: FormGroup;
  noOfGramPanchayath: any;
  geocoverage: any;
  submitted = false;
  projectId: any;
  stateName: any;
  blockName: any;
  districtName: any;
  noOfVillage: any;
  projectInfo: any;
  parms: any;
  geoview: any;
  stateList: any;
  districtList: any;
  blockList: any;
  changeId: number = 0;
  keyword: any;
  showState: boolean = false;
  blockCode: any;
  pageNo: number = 1;
  pageSize: number = 100;
  geolist: any = [];
  lockState: boolean = false;

  @ViewChild('userNameId') userNameId!: ElementRef;


  constructor(
    public common: CommonService,
    public formBuilder: FormBuilder,
    public route: Router,
    public project: ProjectService,
    public spinner: NgxSpinnerService,
    public geo: GeoService) {
    this.projectId = this.common.getProjectId();
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    });
    this.formCreateGeoCoverage = this.formBuilder.group({
      stateName: ['', Validators.required],
      blockName: ['', Validators.required],
      districtName: ['', Validators.required],
      noOfVillage: ['', Validators.required],
      noOfGramPanchayath: ['', Validators.required],
    });
  }

  ngOnInit(): void {
    this.parms = history.state.id;
    this.geoview = history.state.flag;
    this.common.getState().subscribe((resp) => {
      if (resp) {
        this.stateList = resp.data;
        if (this.parms) {
          this.geo.getGeoById(this.parms).subscribe((resp) => {

            if (resp) {
              this.geocoverage = resp.data?.geoCoverageModel;
              this.formCreateGeoCoverage.get("stateName")?.setValue(this.geocoverage.stateName);
              this.formCreateGeoCoverage.get("districtName")?.setValue(this.geocoverage.districtName);
              this.formCreateGeoCoverage.get("blockName")?.setValue(this.geocoverage.blockName);
              this.formCreateGeoCoverage.get("noOfVillage")?.setValue(this.geocoverage.noOfVillage);
              this.formCreateGeoCoverage.get("noOfGramPanchayath")?.setValue(this.geocoverage.noOfGramPanchayath);
              this.getStateInfo(this.geocoverage.stateName);
            }
          });
        }
        this.getGeoCov();
      }
    })


  }
  getGeoCov() {
    this.geo.getGeoCoverage(this.pageNo, this.pageSize, this.projectId).subscribe(resp => {

      if (resp) {
        this.geolist = resp.data?.geoCoverageList;

        if (this.geolist.length > 0) {
          this.lockState = true;
          if (!this.parms) {
            this.formCreateGeoCoverage.get("stateName")?.setValue(this.geolist[0]?.stateName);
            this.getStateInfo(this.geolist[0]?.stateName);
          }
          else {
            if (this.geolist.length == 1) {
              this.lockState = false;
            }
          }
        }
      }
    })
  }
  getStateInfo(statename: any) {
    let state = this.stateList.find((st: any) => st.locationName == statename);

    this.getDistrictList(state.locationCode);
  }
  getDistInfo(distname: any) {
    let dist = this.districtList.find((dt: any) => dt.locationName == distname);

    this.getBlockList(dist.locationCode, dist.parentLocationCode);
  }

  get f(): { [key: string]: AbstractControl } {
    return this.formCreateGeoCoverage.controls;
  }

  onSubmit(value: any) {
    this.submitted = true;
    this.spinner.show();
    if (this.formCreateGeoCoverage.invalid) {
      this.spinner.hide();
      return;
    }
    this.formCreateGeoCoverage.addControl('projectId', new FormControl(null, Validators.required));
    this.formCreateGeoCoverage.controls['projectId'].setValue(this.projectId);

    this.formCreateGeoCoverage.addControl('locationId', new FormControl(null));
    this.formCreateGeoCoverage.controls['locationId'].setValue(this.blockCode);

    this._setFields()

    if (value['districtName']) {
      this._submit(value);
    }
  }
  _setFields() {
    if (this.parms) {
      this.formCreateGeoCoverage.addControl('id', new FormControl(null, Validators.required));
      this.formCreateGeoCoverage.controls['id'].setValue(this.geocoverage.id);
      this.formCreateGeoCoverage.addControl('locationId', new FormControl(null, Validators.required));
      this.formCreateGeoCoverage.controls['locationId'].setValue(this.geocoverage.locationId);
    }
  }
  _submit(value: any) {
    let submit: boolean = true;
    for (let i = 0; i < this.geolist.length; i++) {
      if (value['districtName'] == this.geolist[i].districtName) {
        if (value['blockName'] == this.geolist[i].blockName) {
          submit = false;
        }
      }
    }
    if ((this.parms) && (value["districtName"] == this.geocoverage.districtName) && (value["blockName"] == this.geocoverage.blockName)) {
      submit = true;
    }
    if (submit == true) {
      this.SaveGeo(
        JSON.stringify(this.formCreateGeoCoverage.value, null, 2)
      );
    }
    else {
      this.spinner.hide();
      this.common.showError("Same district and same block already exists.")
    }
  }


  SaveGeo(data: any) {
    this.spinner.show();
    this.geo.saveGeo(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        if (this.parms) {
          this.common.showSuccess('Successfully Updated Geo Coverage');
          this.route.navigate(['/dashboard/project/geo-coverage']);
        }
        else {
          this.common.showSuccess('Successfully Created Geo Coverage');
          this.route.navigate(['/dashboard/project/geo-coverage']);
        }
      }
    });
  }
  onReset(): void {
    this.submitted = false;
    this.formCreateGeoCoverage.reset();
  }

  stateChange(event: any) {
    this.blockList = [];
    this.districtList = [];
    this.showState = true;
    this.formCreateGeoCoverage.get("districtName")?.setValue("");
    this.formCreateGeoCoverage.get("blockName")?.setValue("");
    this.formCreateGeoCoverage.get("noOfGramPanchayath")?.setValue("");
    this.formCreateGeoCoverage.get("noOfVillage")?.setValue("");

    const SelectVal = event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.getDistrictList(SelectVal);
  }

  getDistrictList(SelectVal: any) {
    this.common.getDistrict(SelectVal).subscribe((resp) => {
      this.districtList = resp.data;
      this.changeId = SelectVal; //state id
      if (this.parms) { this.getDistInfo(this.geocoverage.districtName) }
    });
  }

  distChange(event: any) {
    this.blockList = [];
    const SelectVal = event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.getBlockList(SelectVal, this.changeId);
    this.formCreateGeoCoverage.get("noOfGramPanchayath")?.setValue("");
    this.formCreateGeoCoverage.get("noOfVillage")?.setValue("");
  }
  getBlockList(SelectVal: any, changeId: any) {
    this.common.getBlock(SelectVal, changeId).subscribe((resp) => {
      if (resp) {
        this.blockList = resp.data;
      }
    })
  }
  blockChange(event: any) {
    const SelectVal = event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.blockCode = SelectVal;

    this.formCreateGeoCoverage.get("noOfGramPanchayath")?.setValue("");
    this.formCreateGeoCoverage.get("noOfVillage")?.setValue("");
  }

  editState() {
    return;
  }

}
