import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { EventService } from 'src/app/providers/admin/kpiconfig/event/event.service';
import { OutcomeService } from 'src/app/providers/admin/kpiconfig/outcome/outcome.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { BsModalService, BsModalRef } from 'ngx-bootstrap/modal';

@Component({
  selector: 'app-event-add',
  templateUrl: './event-add.component.html',
  styleUrls: ['./event-add.component.scss'],
})
export class EventAddComponent implements OnInit {
  formCreateActivity: FormGroup;
  submitted: boolean = false;
  masterData: any;
  view: any;
  activityDocument: any;
  image: any;
  outcomeId: any;
  parms: any;
  outcome: any;
  eveView: any;
  event: any;
  shg: boolean = true;
  cato: any;
  subcato: any;
  subCategory: any;
  activitycato: any;
  newSector: any;
  indicatorData: any;
  sectorlist: any;
  OutcomeCato: any;
  outComeSubcato: any;
  flag: string = 'add';
  code='en';
  unitMes: any;

  constructor(
    public eve: EventService,
    public out: OutcomeService,
    public common: CommonService,
    public formBuilder: FormBuilder,
    public route: Router,
    public project: ProjectService,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService,
    public modalService: BsModalService,
    public bsModalRef: BsModalRef
  ) {
    this.formCreateActivity = this.formBuilder.group({
      category: ['', Validators.required],
      subCategory: ['', Validators.required],
      keyMeasure: [null],
      type: [''],
      name: ['', Validators.required],
      description: [''],
      kpi: ['', Validators.required],
      units: ['', Validators.required],
      mandatoryProof: [null],
      shgMonitoring: [null],
      reportingOwner: ['', Validators.required],
      reportingFreq: ['', Validators.required],
      imageName: ['', Validators.required],
      activityDocument: [null, Validators.required],
      image: [null, Validators.required],
      docName: ['', Validators.required],
    });
  }

  ngOnInit(): void {
    this.outcomeId = parseInt(this.common.getOutcomeId() || '');
    this.parms = history.state.id;
    if (this.parms) this.flag = 'edit';
    this.indicatorData = this.modalService.config.initialState;
    if (this.indicatorData.flag == 'project') {
      this.outcomeId = this.indicatorData.outcome;
      this.parms = null;
    } else {
      this.eveView = history.state.flag;
      if (this.eveView) this.flag = 'view';
    }
    this.common.getAllSectors().subscribe((resp) => {
      if (resp) {
        this.sectorlist = resp.data;
      }
    });
    this.eve.getUnits(this.code).subscribe((resp) => {
      if(resp){
        this.unitMes = resp.data;
      }
    })
    this.getOutcome();
    if (this.parms) {
      this.spinner.show();
      this.eve.getEventById(this.parms).subscribe((resp) => {
        if (resp) {
          this.spinner.hide();
          this.event = resp.data;
          console.log(this.event);
          this.formCreateActivity
            .get('category')
            ?.setValue(this.event.category);
          this.formCreateActivity
            .get('subCategory')
            ?.setValue(this.event.subCategory);
          this.formCreateActivity
            .get('keyMeasure')
            ?.setValue(this.event.keyMeasure);
          this.formCreateActivity.get('type')?.setValue(this.event.type);
          this.formCreateActivity
            .get('description')
            ?.setValue(this.event.description);
          this.formCreateActivity.get('name')?.setValue(this.event.name);
          this.formCreateActivity.get('kpi')?.setValue(this.event.kpi);
          this.formCreateActivity.get('units')?.setValue(this.event.units);
          this.formCreateActivity
            .get('mandatoryProof')
            ?.setValue(this.event.mandatoryProof);
          this.formCreateActivity
            .get('shgMonitoring')
            ?.setValue(this.event.shgMonitoring);
          this.formCreateActivity
            .get('reportingOwner')
            ?.setValue(this.event.reportingOwner);
          this.formCreateActivity
            .get('reportingFreq')
            ?.setValue(this.event.reportingFreq);
          this.formCreateActivity
            .get('imageName')
            ?.setValue(this.event.imageName);
          this.formCreateActivity
            .get('activityDocument')
            ?.setValue(this.event.activityDocument);
          this.formCreateActivity.get('image')?.setValue(this.event.image);
          this.formCreateActivity.get('docName')?.setValue(this.event.docName);
        }
      });
    }

  }

  get f(): { [key: string]: AbstractControl } {
    return this.formCreateActivity.controls;
  }

  onSubmit(value: any) {
    this.spinner.show();
    this.submitted = true;
    if (this.parms) {
      this.formCreateActivity.controls['activityDocument']?.clearValidators();
      this.formCreateActivity.controls[
        'activityDocument'
      ]?.updateValueAndValidity();
      this.formCreateActivity.controls['image']?.clearValidators();
      this.formCreateActivity.controls['image']?.updateValueAndValidity();
    }
    if (this.formCreateActivity.invalid) {
      return;
    }
    if (this.parms == undefined) {
      this.parms = '';
    }
    let shgvalue = value['shgMonitoring'];
    if (value['shgMonitoring'] == undefined) {
      shgvalue = false;
    }

    let keyMeasure = value['keyMeasure'];
    if (keyMeasure == null) {
      keyMeasure = false;
    }
    let mandatoryProof = value['mandatoryProof'];
    if (mandatoryProof == null) {
      mandatoryProof = false;
    }

    let activity: any = {
      id: this.parms,
      outcomeId: this.outcomeId,
      category: value['category'],
      subCategory: value['subCategory'],
      keyMeasure: keyMeasure,
      type: value['type'],
      name: value['name'],
      description: value['description'],
      kpi: value['kpi'],
      units: value['units'],
      mandatoryProof: mandatoryProof,
      shgMonitoring: shgvalue,
      reportingOwner: value['reportingOwner'],
      reportingFreq: value['reportingFreq'],
      lastUpdatedBy: 'user',
      docName: value['docName'],
      imageName: value['imageName'],
    };

    this.saveEvent(activity);
  }
  saveEvent(activity: any) {
    let formData: any = new FormData();
    if (this.activityDocument) {
      formData.append(
        'activityDocument',
        this.activityDocument,
        this.activityDocument.name
      );
    } else {
      formData.append('activityDocument', null);
    }
    if (this.image) {
      formData.append('image', this.image, this.image.name);
    } else {
      formData.append('image', null);
    }

    formData.append(
      'activity',
      new Blob([JSON.stringify(activity, null, 2)], {
        type: 'application/json',
      })
    );
    this.spinner.show();
    this.out.saveEvent(formData).subscribe((resp) => {
        this.spinner.hide();
        if (resp) {
          if (this.parms) {
            this.common.showSuccess('Successfully Updated Activity');
            this.route.navigate(['/dashboard/admin/event']);
          } else {
            this.common.showSuccess('Successfully Created Activity');
            if (this.indicatorData.flag == 'project') {
              this.decline();
            } else {
              this.route.navigate(['/dashboard/admin/event']);
            }
          }
        }
      },
      (error) => {
        this.common.showError(
          'Something bad happened; please try again later.'
        );
        this.spinner.hide();
      }
    );
  }

  onReset(): void {
    this.submitted = false;
    this.formCreateActivity.reset();
  }

  onChange(event: any, type: string) {
    if (type == 'doc') {
      if (event.target.files[0].size >= this.common.maxFileSize) {
        this.common.showError(
          'File size too large. Please upload a file less than 50mb'
        );
        this.formCreateActivity.get('image')?.setValue(null);
        return;
      } else if (
        event.target.files[0].name.length >= this.common.maxFileNameLength
      ) {
        this.common.showError(
          'File name length restricted to maximum ' +
          this.common.maxFileNameLength +
          ' characters'
        );
        this.formCreateActivity.get('image')?.setValue(null);
        return;
      } else {
        this.activityDocument = event.target.files[0];
      }
    }
    if (type == 'image') {
      if (event.target.files[0].size >= this.common.maxFileSize) {
        this.common.showError(
          'File size too large. Please upload a file less than 50mb'
        );
        this.formCreateActivity.get('activityDocument')?.setValue(null);
        return;
      } else if (
        event.target.files[0].name.length >= this.common.maxFileNameLength
      ) {
        this.common.showError(
          'File name length restricted to maximum ' +
          this.common.maxFileNameLength +
          ' characters'
        );
        this.formCreateActivity.get('activityDocument')?.setValue(null);
        return;
      } else {
        this.image = event.target.files[0];
      }
    }
  }

  getOutcome() {
    this.spinner.show();
    this.out.getOutById(this.outcomeId).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.outcome = resp.data;
        console.log(this.outcome);
        if (this.outcome.outComeLevel == 'Project level') {
          this.formCreateActivity.get('shgMonitoring')?.disable();
          this.formCreateActivity.get('mandatoryProof')?.disable();
        } else {
          this.formCreateActivity.get('shgMonitoring')?.enable();
          this.formCreateActivity.get('mandatoryProof')?.enable();
        }
        setTimeout(() => {
          this.fillSectorDdl(
            this.outcome.sector,
            this.outcome.interventionCategory,
            this.outcome.subCategory
          );
        }, 500);
      }
    });
  }
  _getActivityCato = (sector: any, cato: any, subcato: any) => {
    for (let i: any = 0; i < this.OutcomeCato.length; i++) {
      if (cato === this.OutcomeCato[i].masterValue) {
        if (this.outcome.outComeLevel == 'Project level') {
          this.getActivityCato(this.OutcomeCato[i]?.id, 'pl');
        } else if (this.outcome.outComeLevel == 'Planning Adoption') {
          this.getActivityCato(this.OutcomeCato[i]?.id, 'pad');
        }
      }
    }
  };
  fillSectorDdl(sector: any, cato: any, subcato: any) {
    for (let i: any = 0; i < this.sectorlist.length; i++) {
      if (sector == this.sectorlist[i]?.masterValue) {
        console.log(sector, '=', this.sectorlist[i]?.id);
        this.common.getCato(this.sectorlist[i]?.id).subscribe((resp) => {
          if (resp) {
            this.OutcomeCato = resp.data;
            this._getActivityCato(sector, cato, subcato);
          }
        });
      }
    }
  }
  _getActivities = (i: any) => {
    if (this.activitycato[i].masterValue == this.event.category) {
      if (this.outcome.outComeLevel == 'Project level') {
        this.getActivitySubCato(this.activitycato[i].id, 'pl');
      } else if (this.outcome.outComeLevel == 'Planning Adoption') {
        this.getActivitySubCato(this.activitycato[i].id, 'pad');
      }
    }
  };
  getActivityCato(id: number, pl: string) {
    this.common.getActivityCato(id, pl).subscribe((resp) => {
      if (resp) {
        this.activitycato = resp.data;
        if (this.parms) {
          for (let i = 0; i < this.activitycato.length; i++) {
            this._getActivities(i);
          }
        }
      }
    });
  }

  getActivitySubCato(id: number, pl: string) {
    this.common.getActivitySubCato(id, pl).subscribe((resp) => {
      if (resp) {
        this.subcato = resp.data;
      }
    });
  }

  catoChange(event: any) {
    this.subCategory = [];
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    if (this.outcome.outComeLevel == 'Project level') {
      this.getActivitySubCato(SelectVal, 'pl');
    } else if (this.outcome.outComeLevel == 'Planning Adoption') {
      this.getActivitySubCato(SelectVal, 'pad');
    }
  }
  decline(): void {
    this.bsModalRef.hide();
    this.modalService.hide(0);
  }
  download(url: string, docname:string) {
    this.common.dwnldFile(url, docname);
  }
}
