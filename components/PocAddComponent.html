import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { PocService } from 'src/app/providers/project/poc/poc.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Validator } from 'src/app/_helpers/validators';

@Component({
  selector: 'app-poc-add',
  templateUrl: './poc-add.component.html',
  styleUrls: [ ],
})
export class PocAddComponent implements OnInit {
  formCreatePoc: FormGroup;
  submitted = false;
  name: any;
  alternateContactNumber: any;
  contactNumber: any;
  department: any;
  description: any;
  title: any;
  email: any;
  id: any;
  projectId: any;
  parms: any;
  pocview: any;
  poC: any;
  projectInfo: any;

  constructor(
    public poc: PocService,
    public common: CommonService,
    public formBuilder: FormBuilder,
    public validator: Validator,
    public route: Router,
    public project: ProjectService,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService
  ) {
    this.projectId = this.common.getProjectId();
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    });

    this.formCreatePoc = this.formBuilder.group({
      name: [null, Validators.required],
      alternateContactNumber: new FormControl(null, [  Validators.required,
        Validators.pattern('[0-9]{10}'),
      ]),
      contactNumber: new FormControl(null, [
        Validators.required,
        Validators.pattern('[0-9]{10}'),
      ]),
      department:  [null, Validators.required],
      description: [],
      title: [null, Validators.required],
      email: new FormControl(null, [
        Validators.required,
        Validators.pattern('^[a-z0-9A-Z._%+-]+@[a-z0-9A-Z.-]+\\.[a-zA-Z]{2,4}$'),
      ]),
      personId: [1],
    });
  }

  ngOnInit(): void {
    this.parms = history.state.id;
    this.pocview = history.state.flag;

    if (this.parms) {
      this.poc.getPocById(this.parms).subscribe((resp) => {
        
        if (resp) {
          this.poC = resp.data;
          this.formCreatePoc.get('name')?.setValue(this.poC.name);
          this.formCreatePoc
            .get('alternateContactNumber')
            ?.setValue(this.poC.alternateContactNumber);
          this.formCreatePoc
            .get('contactNumber')
            ?.setValue(this.poC.contactNumber);
          this.formCreatePoc.get('department')?.setValue(this.poC.department);
          this.formCreatePoc.get('description')?.setValue(this.poC.description);
          this.formCreatePoc.get('title')?.setValue(this.poC.title);
          this.formCreatePoc.get('email')?.setValue(this.poC.email);
        }
      });
    }
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreatePoc.controls;
  }

  onSubmit() {
    this.submitted = true;
    if (this.formCreatePoc.invalid) {
      return;
    }

    this.formCreatePoc.addControl(
      'projectId',
      new FormControl('', Validators.required)
    );
    this.formCreatePoc.controls['projectId'].setValue(this.projectId);
    if (this.parms) {
      this.formCreatePoc.addControl(
        'id',
        new FormControl('', Validators.required)
      );
      this.formCreatePoc.controls['id'].setValue(this.poC.id);
    }
    
    this.SavePoc(JSON.stringify(this.formCreatePoc.value, null, 2));
  }

  SavePoc(data: any) {
    this.spinner.show();
    this.poc.addPoc(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        

        if (this.parms) {
          this.common.showSuccess('Successfully Updated Poc');
          this.route.navigate(['/dashboard/project/poc']);
        } else {
          this.common.showSuccess('Successfully Created Poc');
          this.route.navigate(['/dashboard/project/poc']);
        }
      }
    });
  }
  onReset(): void {
    this.submitted = false;
    this.formCreatePoc.reset();
  }
}
