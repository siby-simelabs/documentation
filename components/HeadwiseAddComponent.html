import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { HeadwiseService } from 'src/app/providers/project/headwise/headwise.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-headwise-add',
  templateUrl: './headwise-add.component.html',
  styleUrls: [ ],
})
export class HeadwiseAddComponent implements OnInit {
  projectId: any;
  vewId: any;
  projectInfo: any;
  headwiseInfo: any = [];
  headwiseId: any;
  headwiseview: any;
  formHeadWise: FormGroup;
  submitted = false;
  parms: any;
  headWiseComp: any;
  ExpenseComp: any;
  components: any;
  projectComponent: any;

  constructor(
    public common: CommonService,
    public formBuilder: FormBuilder,
    public route: Router,
    public project: ProjectService,
    public spinner: NgxSpinnerService,
    public headwise: HeadwiseService
  ) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId();
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError('Please Select/Create a Project');
    }
    this.formHeadWise = this.formBuilder.group({
      projectComponent: [null, Validators.required],
      expenseHead: [null, Validators.required],
      amount: [null, Validators.required],
      perDevAllowed: [5, Validators.required],
    });
  }

  ngOnInit(): void {
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    });
    
    this.parms = history.state.id;
    this.headwiseview = history.state.flag;

    if (this.parms) {
      this.spinner.show();
      this.headwise.getHeadwiseById(this.parms).subscribe((resp) => {
        this.spinner.hide();
        if (resp) {
          this.headwiseInfo = resp.data;
          this.headwiseId = this.headwiseInfo.id;
          this.formHeadWise
            .get('projectComponent')
            ?.setValue(this.headwiseInfo.projectComponent);
          this.formHeadWise
            .get('expenseHead')
            ?.setValue(this.headwiseInfo.expenseHead);
          this.formHeadWise.get('amount')?.setValue(this.headwiseInfo.amount);
          this.formHeadWise
            .get('perDevAllowed')
            ?.setValue(this.headwiseInfo.perDevAllowed);

          this.getPrCompoenets();
        }
      });
      
    }
    else{
      this.getPrCompoenets();
    }
    
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formHeadWise.controls;
  }
  onSubmit() {
    this.formHeadWise.addControl(
      'projectId',
      new FormControl('', Validators.required)
    );
    this.formHeadWise.controls['projectId'].setValue(this.projectId);
    this.submitted = true;
    if (this.formHeadWise.invalid) {
      return;
    }
    if (this.parms) {
      this.formHeadWise.addControl(
        'id',
        new FormControl('', Validators.required)
      );
      this.formHeadWise.controls['id'].setValue(this.headwiseId);
    }
    
    this.saveHeadWise(JSON.stringify(this.formHeadWise.value, null, 2));
  }
  saveHeadWise(data: any) {
    this.spinner.show();
    this.headwise.addHeadwise(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        if (this.parms) {
          this.common.showSuccess('Successfully Updated Head wise Provsn.');
          this.route.navigate(['/dashboard/project/headwise']);
        } else {
          this.common.showSuccess('Successfully Added Head wise Provsn.');
          this.route.navigate(['/dashboard/project/headwise']);
        }
      }
      else {
        this.common.showError(resp.message)
      }
    },
    (error: any) => {
      console.error(error);
      this.spinner.hide();
      this.common.showError("The same expense head has already been added");
    });
  }
  onReset(): void {
    this.submitted = false;
    this.formHeadWise.reset();
  }
  componentChane(event: any) {
    const SelectVal = event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.getExpenseHead(SelectVal);
  }
  optionChange(event: any) {
    
    if (event.target.value == "Admin/MNE Cost") {
      this.formHeadWise.get('perDevAllowed')?.setValue('6');
    }
    else {
      this.formHeadWise.get('perDevAllowed')?.setValue('5');
    }
  }
  getPrCompoenets() {
    this.common.getPrComponents().subscribe((resp) => {
      if (resp) {
        
        this.components = resp.data;
        if (this.parms) {
          for (let i: any = 0; i < this.components.length; i++) {
            if (this.headwiseInfo.projectComponent == this.components[i]?.masterValue) {
              this.getExpenseHead(this.components[i]?.id)
            }
          }
        }
      }
    });
  }
  getExpenseHead(id: number) {
    this.common.getExpence(id).subscribe((resp) => {
      if (resp) {
        
        this.ExpenseComp = resp.data;
      }
    });
  }
}
