import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-assigninter',
  templateUrl: './assigninter.component.html',
  styleUrls: ['./assigninter.component.scss']
})
export class AssigninterComponent implements OnInit {
  data: any = [];
  submitted: boolean = false;
  filterform: FormGroup;
  intervention: any;
  interventioList: any;
  datas: any;
  prCount: any;
  name: any;
  projectInfo: any;
  finalProId: any;
  search: any = null;
  constructor(
    public assign: AssigntargetService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public project: ProjectService
  ) {
    this.filterform = this.formBuilder.group({
      intervention: ['', Validators.required]
    })
  }

  ngOnInit(): void {
    this.finalProId = this.common.getFundId();
    this.getAssignTarget();
    this.project.getProjectByID(this.finalProId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger( this.common.headerEvent , [{
          name : "Project ID" ,
          value : resp.data.id 
        },{
          name : "Project Name" ,
          value : resp.data.name 
        }])
      }
    })
  }
  get f(): { [key: string]: AbstractControl } {
    return this.filterform.controls;
  }
  onSubmit() {
    
    let interventionData: any = [];
    this.assign.getInterventionsByName(this.name, this.finalProId).subscribe((resp) => {
      if (resp) {
        this.datas = resp.data?.intervention_list;
        this.interventioList = resp.data?.intervention_list;
        
        let temp: any = [];
        for (let i = 0; i < this.interventioList?.length; i++) {
          if (this.interventioList[i]?.outcome != null) {
            temp.push(this.interventioList[i])
          }
        }
        this.interventioList = temp;
        

        if (this.datas == null) this.datas = [];
      }
      this.common.closeActiveFilter();
    })


  }
  getAssignTarget() {
    this.spinner.show();
    this.assign.getAllIntervention(this.finalProId, this.search).subscribe(resp => {
      if (resp.message == "SUCCESS") {
        
        this.spinner.hide()
        this.interventioList = resp.data?.intervention_list;
        
        let temp: any = [];
        for (let i = 0; i < this.interventioList.length; i++) {
          if (this.interventioList[i].outcome != null) {
            temp.push(this.interventioList[i])
          }
        }
        this.interventioList = temp;
      }
      else{
        this.common.showError(resp.message);
      }
    })
  }
  clearFilter() {
    this.datas = [];
    this.getAssignTarget();
    this.filterform.reset();
    this.common.closeActiveFilter();
  }
  view(grp: any) {
    this.common.saveAssingTarget(grp);
    this.route.navigateByUrl("/dashboard/assigntarget/activities");
  }

}
