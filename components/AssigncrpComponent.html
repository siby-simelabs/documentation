import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder } from '@angular/forms';
import { Router, ActivatedRoute } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { BsModalService, BsModalRef, ModalOptions } from 'ngx-bootstrap/modal';
import { HistoryComponent } from '../distlevel/history/history.component';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { filter } from 'rxjs';

@Component({
  selector: 'app-assigncrp',
  templateUrl: './assigncrp.component.html',
  styleUrls: ['./assigncrp.component.scss'],
})
export class AssigncrpComponent implements OnInit {
  crpId: any;
  crpInfo: any;
  proName: any;
  pageNo: number = 1;
  pageSize: number = 100;
  datas: any;
  searchData: any = [];
  currentPage: number | undefined;
  state = 16;
  blockList: any;
  districtList: any;
  changeId: any;
  blockCode: any;
  filterName: any;
  filterId: any;
  filterGram: any;
  filterBlock: any;
  newCrplist: any;
  grmList: any;
  name: any;
  geoDatas: any = [];
  role: any;

  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public budget: BudgetService,
    public modalService: BsModalService,
    public spinner: NgxSpinnerService,
    public bsModalRef: BsModalRef,
    public geoData: GeoService
  ) {}
  filterForm = this.formBuilder.group({
    districtName: [null],
    blockName: [null],
    gramPanchayath: [null],
    id: [null],
    name: [null],
  });
  ngOnInit(): void {
    this.access();
    this.spinner.hide();
    this.crpId = this.common.getCrpId();
    this.getDist();
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    });
    this.getListCrp();
    this.geoData.getGeoCoverage(0, 1, this.crpId).subscribe((resp) => {
      this.geoDatas = resp.data?.geoCoverageList;
    });
  }
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }
  access() {
    this.role = this.common.screenRole('assigncrp');
  }
  getListCrp() {
    this.spinner.show();
    this.manager.getCrpList(this.crpId).subscribe((resp) => {
      this.spinner.hide();

      if (resp) {
        this.datas = resp.data;
        this.newCrplist = resp.data;
        if (this.datas.length == 0) {
          this.datas = null;
        }
        this.common.trigger('updateHeaderDetails', [
          {
            name: 'Project ID',
            value: this.crpInfo.id,
          },
          {
            name: 'Project Name',
            value: this.crpInfo.name,
          },
          {
            name: 'State',
            value: this.geoDatas[0]?.stateName,
          },
        ]);
      }
    });
  }

  delete(personId: any) {
    if (
      confirm(
        'This will remove the CRP from the list for future assignment and change the CRP status to inactive.'
      )
    ) {
      this.manager
        .deleteAssignVillage(personId, this.crpId)
        .subscribe((resp) => {
          if (resp) {
            this.common.showSuccess(
              'Successfully deleted the record and CRP status changed to inactive.'
            );
            this.getListCrp();
          }
        });
    }
  }
  edit(personId: any) {
    this.route.navigateByUrl('/dashboard/crp/add-assigncrp', {
      state: { personId: personId, flag: 'edit' },
    });
  }
  view(personId: any) {
    this.route.navigateByUrl('/dashboard/crp/add-assigncrp', {
      state: { personId: personId, flag: 'view' },
    });
  }
  getDist() {
    this.common.getDistrict(this.state).subscribe((resp) => {
      this.districtList = resp.data;
    });
    this.getAllBlockList();
  }
  getAllBlockList() {
    this.spinner.show();
    this.manager.getBlkList(this.crpId).subscribe((resp) => {
      this.spinner.hide();

      if (resp) {
        this.blockList = resp.data;
      }
    });
  }
  distChange(event: any) {
    this.blockList = [];
    this.filterForm.get('blockName')?.setValue(null);
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.common.getBlock(SelectVal, this.changeId).subscribe((resp) => {
      this.blockList = resp.data;
    });
  }
  blockChange(event: any) {
    let gramArray: any = [];
    const SelectVal = event.item.blockName;
    for (let i = 0; i < this.newCrplist.length; i++) {
      for (let j = 0; j < this.newCrplist[i].villageList.length; j++) {
        if (this.newCrplist[i].villageList[j].blockName == SelectVal) {
          gramArray = gramArray.concat(this.newCrplist[i].villageList[j]);
        }
      }
    }
    const gids = gramArray.map(
      (o: { gramapanchayat: any }) => o.gramapanchayat
    );
    const filteredGram = gramArray.filter(
      ({ gramapanchayat }: any, index: number) =>
        !gids.includes(gramapanchayat, index + 1)
    );
    this.grmList = filteredGram;
    this.blockCode = SelectVal;
  }
  _getBlockArr(value: any) {
    let blockArr: any = [];
    for (let i = 0; i < this.newCrplist.length; i++) {
      for (let j = 0; j < this.newCrplist[i].villageList.length; j++) {
        if (this.newCrplist[i].villageList[j].blockName == value['blockName']) {
          blockArr = blockArr.concat(this.newCrplist[i]);
        }
      }
    }
    return blockArr;
  }
  _getGramPanArr(value: any, filtered: any) {
    let gramArray: any = [];
    let grampanja = value['gramPanchayath'];
    grampanja = grampanja.toUpperCase();
    for (let i = 0; i < filtered.length; i++) {
      for (let j = 0; j < filtered[i].villageList.length; j++) {
        if (filtered[i].villageList[j].gramapanchayat == grampanja) {
          gramArray = gramArray.concat(filtered[i]);
        }
      }
    }
    return gramArray;
  }
  onSubmit(value: any): void {
    this.datas = this.newCrplist;
    let blockArr: any = [];
    if (value['blockName'] != undefined) {
      blockArr = this._getBlockArr(value);
      const ids = blockArr.map((o: { crpName: any }) => o.crpName);
      const filtered = blockArr.filter(
        ({ crpName }: any, index: number) => !ids.includes(crpName, index + 1)
      );
      this.datas = filtered;
      let gramArray: any = [];
      if (value['gramPanchayath'] != undefined) {
        gramArray = this._getGramPanArr(value, filtered);
        const gids = gramArray.map((o: { crpName: any }) => o.crpName);
        const filteredGram = gramArray.filter(
          ({ crpName }: any, index: number) =>
            !gids.includes(crpName, index + 1)
        );
        this.datas = filteredGram;
        let idNameFIlter: any = [];
        idNameFIlter = filteredGram;
        let valueID: any = value['id'];
        if (value['id'] == '') {
          valueID = undefined;
        }
        if (valueID != undefined) {
          idNameFIlter = idNameFIlter.filter(
            (item: { personId: any }) => item.personId == valueID
          );
        }
        let valueName: any = value['name']?.toLowerCase();
        if (value['name'] == '') {
          valueName = undefined;
        }
        if (valueName != undefined) {
          idNameFIlter = this.find(idNameFIlter, valueName);
        }
        this.datas = idNameFIlter;
      }
    }
    this.common.closeActiveFilter();
  }
  find(items: any, text: any) {
    text = text.split(' ');
    return items.filter((item: { crpName: any | any[] }) => {
      return text.every((el: any) => {
        let result = item.crpName;
        let caps = result.toLowerCase();
        return caps?.includes(el);
      });
    });
  }
  clearFilter() {
    this.getListCrp();
    this.filterForm.reset();
    this.searchData = [];
    this.currentPage = 1;
    this.datas = this.newCrplist;
    this.common.closeActiveFilter();
  }
  openModalWithComponent(id: any, villages: any) {
    let data = {
      projectId: this.crpId,
      villageList: villages,
    };
    const initialState: ModalOptions = {
      initialState: {
        locId: id,
        endpoint: 'getCrpLevelHistory',
        data: data,
      },
      class: 'modal-lg',
    };
    this.bsModalRef = this.modalService.show(HistoryComponent, initialState);
    this.bsModalRef.content.closeBtnName = 'Close';
  }
}
