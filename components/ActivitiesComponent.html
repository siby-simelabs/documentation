import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { InterventionService } from 'src/app/providers/project/interventions/intervention.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Validator } from 'src/app/_helpers/validators';
@Component({
  selector: 'app-activities',
  templateUrl: './activities.component.html',
  styleUrls: ['./activities.component.scss']
})
export class ActivitiesComponent implements OnInit {
 
  submitted = false;
  formCreateActivities: FormGroup;
  name: any;
  outcome: any;
  category: any;
  subCategory: any;
  assignTarget:any;
  activityArray:any;
  tempActivity: any;
  projectInfo: any;
  
  constructor(
    public assign:AssigntargetService,
    public common:CommonService,
    public route:Router,
    public project:ProjectService,
    public spinner: NgxSpinnerService,
    public inter:InterventionService,
    public formBuilder: FormBuilder, 
    private activatedRoute: ActivatedRoute,
    public validator: Validator

  ) {
    this.formCreateActivities = this.formBuilder.group({
      name: ['', Validators.required],
      outcome: ['', Validators.required],
      category: ['', Validators.required],
      subCategory: ['', Validators.required],
    });
   }

  ngOnInit(): void {
    this.assignTarget = this.common.getAssingTarget();
    console.log(this.assignTarget)
    this.tempActivity = this.assignTarget.activities;
    this.activityArray = this.assignTarget.activities;

    this.project.getProjectByID(this.assignTarget.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger(this.common.headerEvent,[{
          name : 'Project ID' ,
          value :  this.assignTarget.projectId 
        },{
          name : 'Project Name' ,
          value :  this.projectInfo.name 
        },{
          name : 'intervention Name' ,
          value : this.assignTarget.name
        }])
      }
    })
    
    
    
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateActivities.controls;
  }
  viewSplits(splits:any){
    this.route.navigateByUrl('/dashboard/assigntarget/assignT', { state: { splits:splits, flag: "edit" } });
  }
  goToAssignTarget(grp:any, id:any){
    this.route.navigateByUrl('/dashboard/assigntarget/target', { state: { targets:grp,interId:this.assignTarget, flag: "edit" } });
    this.common.saveActivity(grp);
  }
  getTotalCount(targetSplit:any){
    let target = 0;
    for(let i=0; i<targetSplit.length; i++){
      for(let j=0; j<targetSplit[i]?.blocks?.length; j++){
        target = target+targetSplit[i].blocks[j].target;
      }
    }
    return target;
  }
  clearFilter(){
    this.name = undefined;
    this.outcome = undefined;
    this.activityArray = this.tempActivity;
    this.common.closeActiveFilter();
  }
  onSubmit(value: any){
    this.activityArray = this.tempActivity;
    
    
    if(value["name"] != undefined){
      let cato = this.activityArray.filter((item: { activityCategory: any; }) => item.activityCategory == value["name"]);
      
      this.activityArray = cato;
      if(value["outcome"] != undefined){
        let kpi = cato.filter((item: { kpi: any; }) => item.kpi == value["outcome"]);
        this.activityArray = kpi;
      }
    }
    this.common.closeActiveFilter();
  }

}
