import { Component, OnInit, TemplateRef } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { BsModalService, BsModalRef } from 'ngx-bootstrap/modal';

@Component({
  selector: 'app-distlevel-add',
  templateUrl: './distlevel-add.component.html',
  styleUrls: ['./distlevel-add.component.scss'],
})
export class DistlevelAddComponent implements OnInit {
  crpId: any;
  formCreateCrp: FormGroup;
  submitted: boolean = false;
  datas: any;
  id: any;
  status: any;
  distuser: any;
  crpInfo: any;
  proName: any;
  distCrp: any;
  parms: any;
  distOfficer: any;
  userName: any;
  userNameInput: boolean = false;
  userStatus: boolean = false;
  modalRef?: BsModalRef;
  projectInfo: any = null;
  role:any;
  constructor(
    private modalService: BsModalService,
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public project: ProjectService,
    public budget: BudgetService,
    public spinner: NgxSpinnerService
  ) {
    this.formCreateCrp = this.formBuilder.group({
      userName: [null, Validators.required],
      id: [null],
      status: [null],
    });
  }

  ngOnInit(): void {
    this.access();
    this.parms = history.state.group;
    if (!this.parms) {
      this.common.showError('Please select a district from the list');
      this.route.navigateByUrl('/dashboard/crp/distlevel');
    }
    this.crpId = this.common.getCrpId();
    //
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
        this.common.trigger('updateHeaderDetails', [{
          name: 'Project ID',
          value: this.crpInfo.id
        }, {
          name: "Project Name",
          value: this.crpInfo.name
        }, {
          name: 'State',
          value: this.parms.stateName
        }, {
          name: 'District',
          value: this.parms.blockName || false
        }])
      }
    });
    this.getUser();
  }
  access() {
    this.role = this.common.screenRole("distlevel");
  }
  updateHeader() {
    this.common.trigger(this.common.headerEvent, [
      ...this.projectInfo
    ])
  }
  getUser() {
    this.manager
      .getAllUsers(this.crpId, 'getDistrictUser')
      .subscribe((resp) => {
        if (resp) {
          this.datas = resp.data;
          if (this.parms) {
            this.getDistOfficer();
          }
        }
      });
  }
  onSubmit(value: any, template: any) {
    this.submitted = true;
    if (this.formCreateCrp.invalid) {
      return;
    } else {
      

        if(this.distuser?.status == 'ACTIVE'){
          this.userStatus = true
        }else{
          this.userStatus = false
        }
      this.distCrp = {
        "id": this.parms.projectCovId || null,
        "projectId": this.crpId,
        "locationId": this.parms.locationId,
        "status": this.userStatus,
        "personId": this.distuser.id
      };

      this.saveDist(this.distCrp, template);
    }
  }
  openModal(template: TemplateRef<any>) {
    this.modalRef = this.modalService.show(template);
  }

  saveDist(data: any, template: any) {
    this.spinner.show();
    this.manager.addDistrict(this.distCrp).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == 'SUCCESS') {
        this.common.showSuccess('Successfully assigned district level manager');
        this.route.navigateByUrl('/dashboard/crp/distlevel');
      } else {
        this.common.showError(resp.message);
      }
    });
  }
  onReset(): void {
    this.submitted = false;
    this.formCreateCrp.reset();
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateCrp.controls;
  }

  nameChange(event: any) {
    console.log(event)
    this.id = "";
    this.status = "";
    this.distuser = this.datas.find((obj: { id: number; }) => { return obj.id === event.item.id; });
    this.formCreateCrp.get("id")?.setValue(this.distuser.id);
    this.formCreateCrp.get("status")?.setValue(this.distuser.status);

  }

  getDistOfficer() {
    this.manager.getDistById(this.parms.projectCovId).subscribe((resp) => {
      if (resp) {
        this.distOfficer = resp.data;
        this.formCreateCrp.get("id")?.setValue(this.distOfficer.userModel.id);
        this.distuser = this.distOfficer.userModel;
        this.formCreateCrp
          .get('status')
          ?.setValue(this.distOfficer.userModel.status);
        this.userStatus = this.distOfficer.userModel.status;
        if (this.distOfficer.userModel.userName) {
          this.userNameInput = true;
        }
        this.formCreateCrp.get("userName")?.setValue(this.distOfficer.userModel.userName);
      }
    })
  }

  ChangeName() {
    this.userNameInput = false;
    this.onReset();
  }
}
