import { Component, OnInit, ViewChild } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { ProjectService } from 'src/app/providers/project/project.service';
import { InterventionService } from 'src/app/providers/project/interventions/intervention.service';
import { CommonService } from 'src/app/providers/common.service';
import { Validator } from 'src/app/_helpers/validators';
import { NgxSpinnerService } from 'ngx-spinner';
import * as moment from 'moment';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';

@Component({
  selector: 'app-intervention-add',
  templateUrl: './intervention-add.component.html',
  styleUrls: ['./intervention-add.component.scss'],
})
export class InterventionAddComponent implements OnInit {
  formCreateIntervention: FormGroup;
  submitted = false;
  name: any;
  interId: any = '';
  sector: any = [];
  category: any;
  subCategory: any;
  proposedStartDate: any;
  proposedEndDate: any;
  actualStartDate: any;
  actualEndDate: any;
  commodity: any;
  description: any;
  projectId: any;
  projectInfo: any;
  parms: any;
  intervention: any;
  interview: any;
  sectorlist: any;
  categoryOfInterventionModelList: any;
  subCategoryOfInterventionModelList: any;
  commodityList: any;
  cato: any;
  subcato: any;
  commo: any;
  valueChain: boolean = true;
  PropoValidDateCheck: boolean = false;
  actualValidDateCheck: boolean = false;
  proposedEndminDate: any;
  proposedStartminDate: any;
  actualEndMindate: any;
  proposedStartmaxDate: any;
  proposedEndmaxDate: any;
  data: any;
  role: any;
  roleBase: any;
  actualStartMaxDate: any;
  latestEndMaxDate: any;
  actualStartminDate: any;
  placement: TypeDatePickerPlacement = 'bottom';
  constructor(
    public common: CommonService,
    public formBuilder: FormBuilder,
    public validator: Validator,
    public route: Router,
    public project: ProjectService,
    public inter: InterventionService,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService,
    public dp: DatePickerPlacement
  ) {
    this.projectId = this.common.getProjectId();
    this.formCreateIntervention = this.formBuilder.group({
      name: [
        null,
        Validators.compose([Validators.required, Validators.maxLength(50)]),
      ],
      sector: [null, Validators.required],
      category: [null, Validators.required],
      subCategory: [null],
      description: [null, Validators.maxLength(499)],
      commodity: [null],
      proposedStartDate: [null, Validators.required],
      proposedEndDate: [null, Validators.required],
      actualStartDate: [null],
      actualEndDate: [null],
    });
  }
  _manageFields() {
    this.spinner.show();
    this.inter.getInterById(this.parms).subscribe((resp) => {
      this.spinner.hide();
      this.intervention = resp.data;
      this.interId = this.intervention.id;
      this.formCreateIntervention.get('name')?.setValue(this.intervention.name);
      this.formCreateIntervention.get('id')?.setValue(this.intervention.id);
      this.formCreateIntervention
        .get('sector')
        ?.setValue(this.intervention.sector);
      this.formCreateIntervention
        .get('category')
        ?.setValue(this.intervention.category);
      this.formCreateIntervention
        .get('subCategory')
        ?.setValue(this.intervention.subCategory);
      this.formCreateIntervention
        .get('commodity')
        ?.setValue(this.intervention.commodity);
      this.formCreateIntervention
        .get('description')
        ?.setValue(this.intervention.description);
      this.formCreateIntervention
        .get('proposedStartDate')
        ?.setValue(this.intervention.proposedStartDate);
      this.formCreateIntervention
        .get('proposedEndDate')
        ?.setValue(this.intervention.proposedEndDate);
      this.formCreateIntervention
        .get('actualStartDate')
        ?.setValue(this.intervention.actualStartDate);
      this.formCreateIntervention
        .get('actualEndDate')
        ?.setValue(this.intervention.actualEndDate);
      setTimeout(() => {
        this.fillSectorDdl(
          this.intervention.sector,
          this.intervention.category,
          this.intervention.subCategory
        );
      }, 500);
      if (this.intervention.sector == 'Value Chain') {
        this.formCreateIntervention.get('subCategory')?.enable();
        this.formCreateIntervention.get('commodity')?.enable();
      } else {
        this.formCreateIntervention.get('subCategory')?.disable();
        this.formCreateIntervention.get('commodity')?.disable();
      }
    });
  }
  ngOnInit(): void {
    this.parms = history.state.id;
    this.interview = history.state.flag;
    this.getMaster();
    if (this.parms) {
      this._manageFields();
    }

    this._fetchProject();
  }
  _setValues() {
    if (this.projectInfo.actualStartDate != null) {
      this.actualStartminDate = this.dateForMindate(
        this.projectInfo.actualStartDate
      );
    }
    if (this.projectInfo.actualEndDate != null) {
      this.actualStartMaxDate = this.dateForMindate(
        this.projectInfo.actualEndDate
      );
    }

    if (this.projectInfo.proposedEndDate != null) {
      this.proposedStartmaxDate = this.dateForMindate(
        this.projectInfo.proposedEndDate
      );
    }
  }
  _fetchProject() {
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp?.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);

        if (this.projectInfo.proposedStartDate != null) {
          this.proposedStartminDate = this.dateForMindate(
            this.projectInfo.proposedStartDate
          );
        }
        if (this.projectInfo.actualEndDate != null) {
          this.latestEndMaxDate = this.dateForMindate(
            this.projectInfo.actualEndDate
          );
        }
        if (this.projectInfo.proposedEndDate != null) {
          this.proposedEndmaxDate = this.dateForMindate(
            this.projectInfo.proposedEndDate
          );
        }
        this._setValues();
        this.access();
      }
    });
  }
  dateForMindate(date: any) {
    if (!date || typeof date !== 'string') return;
    let datearray = date.split('/');
    let newdate = datearray[1] + '/' + datearray[0] + '/' + datearray[2];

    return new Date(newdate);
  }
  access() {
    this.role = this.common.screenRole('project_profile');
    this.roleBase = this.common.getRoleBase();

    if (this.roleBase.name != 'pm_central_user_level1_final_approval') {
      this.formCreateIntervention.controls['actualStartDate'].disable();
      this.formCreateIntervention.controls['actualEndDate'].disable();
    } else {
      this.formCreateIntervention.controls['name'].disable();
      this.formCreateIntervention.controls['sector'].disable();
      this.formCreateIntervention.controls['category'].disable();
      this.formCreateIntervention.controls['subCategory'].disable();
      this.formCreateIntervention.controls['commodity'].disable();
      this.formCreateIntervention.controls['description'].disable();
      this.formCreateIntervention.controls['proposedStartDate'].disable();
      this.formCreateIntervention.controls['proposedEndDate'].disable();
      this.formCreateIntervention.controls['actualStartDate'].setValidators(
        Validators.required
      );
      this.formCreateIntervention.controls['actualEndDate'].setValidators(
        Validators.required
      );
      this.formCreateIntervention.controls[
        'actualStartDate'
      ].updateValueAndValidity();
      this.formCreateIntervention.controls[
        'actualEndDate'
      ].updateValueAndValidity();
    }
    if (this.roleBase.name == 'pm_central_user_level2') {
      this.formCreateIntervention.controls['actualStartDate'].enable();
      this.formCreateIntervention.controls['actualEndDate'].enable();
      this.formCreateIntervention.controls['actualStartDate'].setValidators(
        Validators.required
      );
      this.formCreateIntervention.controls['actualEndDate'].setValidators(
        Validators.required
      );
      this.formCreateIntervention.controls[
        'actualStartDate'
      ].updateValueAndValidity();
      this.formCreateIntervention.controls[
        'actualEndDate'
      ].updateValueAndValidity();
    }
  }

  get f(): { [key: string]: AbstractControl } {
    return this.formCreateIntervention.controls;
  }

  getMaster() {
    this.common.getMasterData().subscribe((resp) => {
      if (resp) {
        this.data = resp.data;
      }
    });
    this.common.getAllSectors().subscribe((resp) => {
      if (resp) {
        this.sectorlist = resp.data;
      }
    });
  }

  checkDateValid(startDate: any, EndDate: any) {
    let mStart = moment(startDate, 'DD-MM-YYYYTHH:mm:ss');
    let mEnd = moment(EndDate, 'DD-MM-YYYYTHH:mm:ss');
    return mStart.isBefore(mEnd);
  }

  onSubmit(value: any): void {
    this.spinner.show()
    this.submitted = true;
    let newData: any = {};
    if (this.formCreateIntervention.invalid) {
      this.spinner.hide();
      return;
    } else {
      let commoInter = this.formCreateIntervention.getRawValue().commodity;

      if (commoInter == undefined) {
        commoInter = '';
      }
      let subcatoInter = this.formCreateIntervention.getRawValue().subCategory;
      if (subcatoInter == undefined) {
        subcatoInter = '';
      }
      let actdate: any = (<HTMLInputElement>(
        document.getElementById('actualStartDate')
      )).value;
      if (actdate == '') {
        actdate = null;
      }
      let actEnd: any =
        (<HTMLInputElement>document.getElementById('actualEndDate')).value ||
        null;

      newData = {
        name: this.formCreateIntervention.getRawValue().name,
        sector: this.formCreateIntervention.getRawValue().sector,
        category: this.formCreateIntervention.getRawValue().category,
        description: this.formCreateIntervention.getRawValue().description,
        proposedStartDate: (<HTMLInputElement>(
          document.getElementById('proposedStartDate')
        )).value,
        proposedEndDate: (<HTMLInputElement>(
          document.getElementById('proposedEndDate')
        )).value,
        actualStartDate: actdate,
        actualEndDate: actEnd,
        subCategory: subcatoInter,
        commodity: commoInter,
        outcome: this.formCreateIntervention.getRawValue().outcome,
        projectId: this.projectId,
        id: null,
      };
    }
    this._save(newData);
  }
  _confirmWithUser(newData: any) {
    if (
      confirm(
        'Change in sector/category/sub-category will affect the existing outcomes in indicators. Do you want to continue?'
      ) == true
    ) {
      this.SaveIntervention(JSON.stringify(newData));
    } else {
      return;
    }
  }
  _saveNullIntervention(newData: any) {
    if (
      this.formCreateIntervention.getRawValue().sector !=
        this.intervention.sector ||
      this.formCreateIntervention.getRawValue().category !=
        this.intervention.category ||
      this.formCreateIntervention.getRawValue().subCategory !=
        this.intervention.subCategory
    ) {
      this._confirmWithUser(newData);
    } else {
      this.SaveIntervention(JSON.stringify(newData));
    }
  }
  _save(newData: any) {
    if (this.parms) {
      newData['id'] = this.intervention.id;
      if (this.intervention?.outcome != null) {
        this._saveNullIntervention(newData);
      } else {
        this.SaveIntervention(JSON.stringify(newData));
      }
    } else {
      this.SaveIntervention(JSON.stringify(newData));
    }
  }
  SaveIntervention(data: any) {
    this.spinner.show();
    this.inter.addInterventions(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
       

        if (this.parms) {
          this.common.showSuccess('Successfully Updated Intervention');
          this.route.navigate(['/dashboard/project/intervention']);
        } else {
          this.common.showSuccess('Successfully Created Intervention');
          this.route.navigate(['/dashboard/project/intervention']);
        }
      }
    });
  }

  onReset(): void {
    this.submitted = false;
    this.formCreateIntervention.reset();
  }
  getSubCato = (sector: any, cato: any, subcato: any, resp: any) => {
    if (resp) {
      this.cato = resp.data;
      for (let j: any = 0; j < this.cato.length; j++) {
        if (cato == this.cato[j].masterValue) {
          this.common.getSubCato(this.cato[j]?.id).subscribe((resp) => {
            this._getCommodity(subcato, resp);
          });
        }
      }
    }
  };
  _getCommodity = (subcato: any, resp: any) => {
    if (resp) {
      this.subcato = resp.data;
      for (let k: any = 0; k < this.subcato.length; k++) {
        if (subcato == this.subcato[k].masterValue) {
          this.getCommodity(this.subcato[k].id);
        }
      }
    }
  };
  fillSectorDdl(sector: any, cato: any, subcato: any) {
    for (let i: any = 0; i < this.sectorlist.length; i++) {
      if (sector == this.sectorlist[i]?.masterValue) {
        this.common.getCato(this.sectorlist[i]?.id).subscribe((resp: any) => {
          this.getSubCato(sector, cato, subcato, resp);
        });
      }
    }
  }
  sectorChange(event: any) {
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    if (this.parms && this.intervention.outcomeId != null) {
      if (
        confirm(
          'Change in sector will clear the already assigned outcome in indicators/measures'
        ) == true
      ) {
        this.sectorLoop(SelectVal);
      } else {
        this.formCreateIntervention
          .get('sector')
          ?.setValue(this.intervention.sector);
        this.sectorLoop(SelectVal);
      }
    } else {
      this.sectorLoop(SelectVal);
    }
    if (event.target.value == 'Value Chain') {
      this.formCreateIntervention.get('subCategory')?.enable();
      this.formCreateIntervention.get('commodity')?.enable();
    } else {
      this.formCreateIntervention.get('subCategory')?.disable();
      this.formCreateIntervention.get('commodity')?.disable();
    }
  }

  getCato(SelectVal: number) {
    this.common.getCato(SelectVal).subscribe((resp) => {
      if (resp) {
        this.cato = resp.data;
      }
    });
  }
  sectorLoop(SelectVal: any) {
    this.formCreateIntervention.get('category')?.setValue('');
    this.formCreateIntervention.get('subCategory')?.setValue('');
    this.formCreateIntervention.get('commodity')?.setValue('');
    this.cato = [];
    this.subcato = [];
    this.commo = [];

    this.getCato(SelectVal);
  }
  catoChange(event: any) {
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.getSubcato(SelectVal);
  }
  getSubcato(SelectVal: number) {
    this.common.getSubCato(SelectVal).subscribe((resp) => {
      if (resp) {
        this.subcato = resp.data;
      }
    });
  }
  subCatoChange(event: any) {
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.getCommodity(SelectVal);
  }
  getCommodity(SelectVal: number) {
    this.common.getCommodity(SelectVal).subscribe((resp) => {
      if (resp) {
        this.commo = resp.data;
      }
    });
  }
  onValueChange(event: any, value: any) {
    if (value == 'proposedStartDate') {
      this.formCreateIntervention.get('proposedEndDate')?.setValue('');
      this.proposedEndminDate = event;
    }
    if (value == 'actualStartDate') {
      this.formCreateIntervention.get('actualEndDate')?.setValue('');
      this.actualEndMindate = event;
    }
  }
}
