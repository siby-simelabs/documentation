import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { data } from 'jquery';
import { BsModalRef, BsModalService } from 'ngx-bootstrap/modal';
import { NgxSpinnerService } from 'ngx-spinner';
import { EventService } from 'src/app/providers/admin/kpiconfig/event/event.service';
import { OutcomeService } from 'src/app/providers/admin/kpiconfig/outcome/outcome.service';
import { CommonService } from 'src/app/providers/common.service';
import { IndicatorsService } from 'src/app/providers/project/indicators/indicators.service';
import { InterventionService } from 'src/app/providers/project/interventions/intervention.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Validator } from 'src/app/_helpers/validators';

@Component({
  selector: 'app-addtarget',
  templateUrl: './addtarget.component.html',
  styleUrls: ['./addtarget.component.scss']
})
export class AddtargetComponent implements OnInit {
  formCreateTarget: FormGroup;
  projectInfo: any;
  target: any;
  submitted = false;

  projectId: any;
  yearFiveTarget: any;
  yearFourTarget: any;
  yearOneTarget: any;
  yearSixTarget: any;
  yearThreeTarget: any;
  yearTwoTarget: any;
  yearSevenTarget: any;
  yearEightTarget: any;
  isChecked = false;
  totalNumberCount: any;
  interId: string | undefined;
  kpiid: string | undefined;
  outcomeId: string | undefined;
  outcomename: string | undefined;
  kpiName: string | undefined;
  parms: any;
  Target: any;
  activityDetail: any;
  flag: any;
  indicatorData: any | undefined;

  constructor(
    public int: IndicatorsService,
    public event: EventService,
    public inter: InterventionService,
    public common: CommonService,
    public out: OutcomeService,
    public validator: Validator,
    public route: Router,
    public project: ProjectService,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService,
    public modalService: BsModalService,
    public bsModalRef: BsModalRef
  ) {
    this.formCreateTarget = this.formBuilder.group({
      target: ['', [Validators.required]],
      yearFiveTarget: [''],
      yearFourTarget: [''],
      yearOneTarget: [''],
      yearSixTarget: [''],
      yearThreeTarget: [''],
      yearTwoTarget: [''],
      yearSevenTarget: [''],
      yearEightTarget: [''],
      setYearWiseTarget: [],
    }
    );
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateTarget.controls;
  }
getTabs(){
  let items : any = [{
    name : 'List of interventions & Assigned outcome' ,
    link : '/dashboard/project/indicators'
  },{
    name : 'Assign outcome KPI target to intervention' ,
    link  : '/dashboard/project/edit-indicators'
  },{
    name : "Assign Target" ,
    link : '/dashboard/project/add-target'
  }]
  return items ;
}
  ngOnInit(): void {
    this.formCreateTarget.controls['setYearWiseTarget'].setValue(false);
    this.indicatorData = this.modalService.config.initialState;
    this.parms = this.indicatorData.id;
    this.flag = this.indicatorData.flag;
    
    this.interId = this.indicatorData.interId;
    this.kpiid = this.indicatorData.id;
    this.outcomename = this.indicatorData.outcomename;
    this.outcomeId = this.indicatorData.outcomeId;
    this.kpiName = this.indicatorData.kpi;
    this.projectId = this.common.getProjectId();

    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;

      }
    })
    this.getActivityById();
    if (this.parms) {
      this.int.editAssign(this.outcomeId, this.interId, this.parms).subscribe((resp) => {
        if (resp) {
          this.Target = resp.data;
          this.formCreateTarget.get("yearOneTarget")?.setValue(resp.data?.yearOneTarget);
          this.formCreateTarget.get("yearTwoTarget")?.setValue(resp.data?.yearTwoTarget);
          this.formCreateTarget.get("yearThreeTarget")?.setValue(resp.data?.yearThreeTarget);
          this.formCreateTarget.get("yearFourTarget")?.setValue(resp.data?.yearFourTarget);
          this.formCreateTarget.get("yearFiveTarget")?.setValue(resp.data?.yearFiveTarget);
          this.formCreateTarget.get("yearSixTarget")?.setValue(resp.data?.yearSixTarget);
          this.formCreateTarget.get("setYearWiseTarget")?.setValue(true);
          this.formCreateTarget.controls['target']?.setValue(resp.data?.total);

          if (resp.data?.setYearWise == true) {
            this.isChecked = true
            this.formCreateTarget.get("setYearWiseTarget")?.setValue(true);
          }
          else {
            this.isChecked = false
            this.formCreateTarget.get("setYearWiseTarget")?.setValue(false);
          }
          if (this.projectInfo?.mode == "Project") {
            this.isChecked = false;
            this.yearOneTarget = null;
            this.yearTwoTarget = null;
            this.yearThreeTarget = null;
            this.yearFourTarget = null;
            this.yearFiveTarget = null;
            this.yearSixTarget = null;
            this.yearSevenTarget = null;
            this.yearEightTarget = null;
          }

        }

      });
    }

  }

  getActivityById() {
    this.event.getEventById(this.parms).subscribe((resp) => {
      if (resp) {
        this.activityDetail = resp;
       }
    });
  }

  onSubmit(value: any) {
    this.submitted = true;
    if (this.formCreateTarget.invalid) {
      return;
    }
    if (this.totalNumberCount > this.target) {
      alert("Sum of the year wise target should not exceed the total target!")
      return;
    }

    if (this.isChecked == true) {
      this.formCreateTarget.controls['setYearWiseTarget'].setValue(true);
    }
    else {
      this.formCreateTarget.controls['setYearWiseTarget'].setValue(false);
    }

    let data = {
      "interventionId": this.interId,
      "outcomeId": this.outcomeId,
      "kpi": this.kpiName,
      "target": null,
      "total": value["target"],
      "yearOneTarget": value["yearOneTarget"],
      "yearTwoTarget": value["yearTwoTarget"],
      "yearThreeTarget": value["yearThreeTarget"],
      "yearFourTarget": value["yearFourTarget"],
      "yearFiveTarget": value["yearFiveTarget"],
      "yearSixTarget": value["yearSixTarget"],
      "setYearWise": this.isChecked,
      "activityId": this.kpiid,
      "locationId": "0", //static value
      "userId": 0, //static value
    }
    this.SaveTarget(data);
  }
  SaveTarget(data: any) {
    this.spinner.show();
    this.int.saveTarget(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.common.showSuccess('Successfully Updated');
   
        this.decline("updated");
      }
    })
  }
  changeStatus(event: any) {
    if (event.srcElement.checked == true) {
      alert("When setting year targets, the years should not exceed the project end date.");
      this.isChecked = true;
      this.formCreateTarget.controls['setYearWiseTarget'].setValue(true);
    }
    else {
      this.formCreateTarget.controls['setYearWiseTarget'].setValue(false);
      this.isChecked = false;
      this.yearOneTarget = null;
      this.yearTwoTarget = null;
      this.yearThreeTarget = null;
      this.yearFourTarget = null;
      this.yearFiveTarget = null;
      this.yearSixTarget = null;
      this.yearSevenTarget = null;
      this.yearEightTarget = null;
    }
  }
  onTotalChange(event: any) {
    this.totalNumberCount = (this.formCreateTarget.controls['yearOneTarget'].value || 0) + (this.formCreateTarget.controls['yearTwoTarget'].value || 0) + (this.formCreateTarget.controls['yearThreeTarget'].value || 0) + (this.formCreateTarget.controls['yearFourTarget'].value || 0) + (this.formCreateTarget.controls['yearFiveTarget'].value || 0) + (this.formCreateTarget.controls['yearSixTarget'].value || 0) + (this.formCreateTarget.controls['yearSevenTarget'].value || 0) + (this.formCreateTarget.controls['yearEightTarget'].value || 0);
    if (this.totalNumberCount > this.formCreateTarget.get('target')?.value) {
      alert("Sum of the year wise target should not exceed the total target!")
      return
    }
  }

  decline(re:string): void {
    this.modalService.setDismissReason(re);
    this.bsModalRef.hide();
    this.modalService.hide(0)
  }
}
