import { DatePipe } from '@angular/common';
import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { ReviewService } from 'src/app/providers/project/review/review.service';
import * as moment from 'moment';
import { Api } from 'src/app/providers/api.service';
import { formatDate } from "../../../../../_helpers/dateFormt";
import { saveAs } from 'file-saver';

@Component({
  selector: 'app-pdf',
  templateUrl: './pdf.component.html',
  styleUrls: ['./pdf.component.scss']
})
export class PdfComponent implements OnInit {
  projectId: any;
  vewId: any;
  projectInfo: any;
  pageNo: number = 1;
  pageSize: number = 100;
  datas: any = [];
  prCount: number = 0;

  constructor(public rev: ReviewService, public common: CommonService, public route: Router, public project: ProjectService, public spinner: NgxSpinnerService, public datePipe: DatePipe, private api: Api) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId()
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError("Please Select/Create a Project");
    }
  }

  ngOnInit(): void {
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    })
    this.getReview();
    this.downloadPdf();

  }
  _formatDate = (date: any) => {
    return formatDate(date);
  }

  downloadPdf() {
    let url = 'project/getProposalDocument?projectId=' + this.projectId + '';
    let name = this.projectId + ".pdf";
    this.common.dwnldFile(url, name);
  }

  getReview() {
    this.spinner.show();
    this.rev.getHistory(this.pageNo, this.pageSize, this.projectId).subscribe(resp => {
      this.spinner.hide();

      if (resp) {
        this.datas = resp?.data;
        if(this.datas ==null) this.datas=[];
        this.prCount = resp?.data?.length;
      }
    });
  }

  dateFormat(date: any) {
    let d = new Date(date);
    let fdate = moment(d).format("DD/MM/YYYY")
    return fdate;
  }

}
