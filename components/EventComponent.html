import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { EventService } from 'src/app/providers/admin/kpiconfig/event/event.service';
import { OutcomeService } from 'src/app/providers/admin/kpiconfig/outcome/outcome.service';
import { CommonService } from 'src/app/providers/common.service';

@Component({
  selector: 'app-event',
  templateUrl: './event.component.html',
  styleUrls: ['./event.component.scss']
})
export class EventComponent implements OnInit {
   
  pageNo: number = 1;
  pageSize: number = 1000;
  datas: any = [];
  prCount!: number;
  outcomeId: any;
  vewId: any;
  outcomeData: any;
  flag:string = "add";
  role:any;
  constructor(public eve: EventService, public outcome: OutcomeService, public common: CommonService, public route: Router, public spinner: NgxSpinnerService,  public out: OutcomeService,) {

  }

  ngOnInit(): void {
    this.access();
    this.vewId = this.common.getOutcomeView();
    this.outcomeId = this.common.getOutcomeId();
    if (this.outcomeId == "") {
      this.common.showSuccess("Please add Outcome!");
      this.route.navigate(['/dashboard/admin/outcome']);
    }
    else{
      this.flag = "edit"
    }
    if(this.vewId) this.flag = "view";
    this.getEvent();
    this.getOutcome();
    
  } 
  access() {
    this.role = this.common.screenRole("kpiconfig");
  }
  getEvent() {
    this.spinner.show();
    this.eve.getActivity(this.pageNo, this.pageSize, this.outcomeId).subscribe(resp => {
      this.spinner.hide();
      if (resp) {
        this.datas = resp.data?.kpiActivityList;
        this.prCount = resp.data?.totalCount;
       }
    },
      (error) => {
        console.error(error);
        this.spinner.hide();
      });
  }
  edit(id: any) {
    this.route.navigateByUrl('/dashboard/admin/edit-event', { state: { id: id } });
  }
  view(id: any) {
    this.route.navigateByUrl('/dashboard/admin/edit-event', { state: { id: id, flag: "view" } });
  }
  delete(id: any) {
    this.spinner.show();
    if (confirm('Are you sure you want to delete this record?')) {
      this.eve.delete(id).subscribe(resp => {
        
        this.spinner.hide();
        if (resp) {
          this.common.showSuccess("Successfully Deleted");
          this.getEvent();
        }
      })
    }
    else {
      this.spinner.hide();
    }
  }
  mandatoryProof(status: any) {
    if (status == true) {
      return "Yes";
    }
    else {
      return "No";
    }
  }
  getOutcome() {
    this.out.getOutById(this.outcomeId).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.outcomeData = resp.data;
        
      }
    });
  }

}
