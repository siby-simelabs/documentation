import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-assignshg-add',
  templateUrl: './assignshg-add.component.html',
  styleUrls: ['./assignshg-add.component.scss']
})
export class AssignshgAddComponent implements OnInit {
  crpId: any;
  crpInfo: any;
  proName: any;
  village: any;
  selectedCheck: any = [];
  blkId: any;
  overhide: boolean = true;
  flag: any;
  index: any;
  singleShgData: any;
  shgview: boolean = false;
  state = 16;
  blockList: any;
  districtList: any;
  changeId: any;
  blockCode: any;
  grmList: any = [];
  villageNewList: any = [];
  villageListFilter: any = [];
  selectAllCheck: boolean = false;
  shgModelList:any;
  showFilter:boolean = false;
  geoDatas: any = [];
  role:any;


  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public spinner: NgxSpinnerService,
    public geoData: GeoService) { }
  filterForm = this.formBuilder.group(
    {
      districtName: [null],
      blockName: [null],
      gramPanchayath: [null],
      village: [null],
      shg: [null]
    },
  );

  ngOnInit(): void {
    this.access()
    this.crpId = this.common.getCrpId();
    this.shgModelList = history.state.shgModelList?.shgResponseModel
    this.flag = history.state.flag;
    this.index = history.state.index;
    console.log(this.index)
    if (this.flag == "view") {
      this.shgview = true;
    }
    if (this.index == undefined || null) {
      this.route.navigate(['/dashboard/crp/assignshg']);
      this.common.showError("Select a SHG from SHG List")
    }
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    });
    this.geoData.getGeoCoverage(0, 1, this.crpId).subscribe((resp) => {
      this.geoDatas = resp.data?.geoCoverageList;
    })
    this.listAllVill();
    this.getAllBlockList();
  }
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }
  access() {
    this.role = this.common.screenRole("assignshg");
  }
  listAllVill() {
    this.manager.getAllShg(this.crpId).subscribe((resp) => {
      if (resp) {
        this.selectedCheck = [];
        this.singleShgData = resp.data[this.index];
        let str = JSON.stringify(resp?.data[this.index]?.shgResponseModel);
        this.village = resp?.data[this.index]?.shgResponseModel;
        str = str.replace(/shgName/g, 'villageName');
        str = str.replace(/shgId/g, 'locationCode');
        let object = JSON.parse(str);
        this.village = object;
        this.villageNewList = object;

        for (let i = 0; i < this.village.length; i++) {
          if (this.village[i].selected == true) {
            this.selectedCheck.push(this.village[i]);
          }
        }
        if (this.selectedCheck?.length == this.village?.length) {
          this.selectAllCheck = true;
        }
        this.common.trigger('updateHeaderDetails',[{
          name : 'Project ID' ,
          value : this.crpInfo.id 
        },{
          name : "Project Name" ,
          value : this.crpInfo.name
        },{
          name : 'State' ,
          value : this.geoDatas[0]?.stateName 
        },{
          name : 'CRP Name' ,
          value : history.state.shgModelList?.crpName
        }])
        
      }
    })
  }
  checkValue(event: any, code: any) {
    
    let indexVal;
    
    if (event.currentTarget.checked == true) {
      for (let i = 0; i < this.village.length; i++) {
        if (this.village[i].locationCode == code) {
          indexVal = this.village[i];
          indexVal.selected = true;
          indexVal = {
            "villageName": indexVal.villageName,
            "locationCode": indexVal.locationCode,
            "selected": true,
            "status": true,
          }
          this.selectedCheck.push(indexVal);
          break;

        }
      }
    }
    else {
      this.selectedCheck = this.selectedCheck.filter((item: { locationCode: any; }) => item.locationCode !== code);
    }
  }
  removeCheckd(event: any, code: any) {
    if (event.currentTarget.checked == false) {
      this.selectedCheck = this.selectedCheck.filter((item: { locationCode: any; }) => item.locationCode !== code);
      let obj = this.village.find((village: { locationCode: number; }) => village.locationCode == code);
      if (obj) {
        obj.selected = false;
      }
    }
  }

  saveVillage() {
    if (this.selectedCheck.length > 0) {
      let str = JSON.stringify(this.selectedCheck);
      str = str.replace(/villageName/g, 'shgName');
      str = str.replace(/locationCode/g, 'shgId');
      let object = JSON.parse(str);
      let data = {
        "shgModelList": object,
        "projectId": this.crpId,
        "personId": this.singleShgData.personId
      }
      this.manager.saveshg(data).subscribe((resp) => {
        this.spinner.hide();
        if (resp.message == "SUCCESS") {
          
          this.common.showSuccess('Successfully assign SHG');
          this.route.navigateByUrl('/dashboard/crp/assignshg');
        }
        else {
          this.common.showError(resp.message);
        }
      })
    }
    else {
      this.common.showError('Select any SHG from the list');
    }

  }

  selectAll(event: any) {
    if (event.currentTarget.checked == true) {
      for (let i = 0; i < this.village.length; i++) {
        this.village[i].selected = true;
      }
      this.selectedCheck = this.village;
    }
    else {
      for (let i = 0; i < this.village.length; i++) {
        this.village[i].selected = false;
      }
      this.selectedCheck = [];
    }
  }
  showMore() {
    this.overhide = false;
  }
  getAllBlockList() {
    this.spinner.show();
    this.manager.getBlkList(this.crpId).subscribe(resp => {
      this.spinner.hide();
      
      if (resp) {
        this.blockList = resp.data;
        this.districtList = resp.data;
        const gids = this.districtList.map((o: { districtName: any; }) => o.districtName)
        const grpdata = this.districtList.filter(({ districtName }: any, index: number) => !gids.includes(districtName, index + 1));
        this.districtList = grpdata;
      }
    })
  }

  distChange(event: any) {
    
    this.blockList = [];
    this.grmList = [];
    this.villageListFilter = []
    this.filterForm.get("blockName")?.setValue(null);
    const SelectVal = event.item.districtName;
    
    for (let i = 0; i < this.villageNewList.length; i++) {
      if (this.villageNewList[i].district == SelectVal) {
        this.blockList.push(this.villageNewList[i]);
      }
    }
    const gids = this.blockList.map((o: { block: any; }) => o.block)
    const grpdata = this.blockList.filter(({ block }: any, index: number) => !gids.includes(block, index + 1));
    this.blockList = grpdata;
    

  }

  blockChange(event: any) {
    
    this.grmList = [];
    this.villageListFilter = []
    const SelectVal = event.item.block;
    for (let i = 0; i < this.villageNewList.length; i++) {
      if (this.villageNewList[i].block == SelectVal) {
        this.grmList.push(this.villageNewList[i]);
      }
    }
    const gids = this.grmList.map((o: { panchayat: any; }) => o.panchayat)
    const filteredGram = this.grmList.filter(({ panchayat }: any, index: number) => !gids.includes(panchayat, index + 1));
    this.grmList = filteredGram;
    
  }
  gramChange(event: any) {
    
    this.villageListFilter = [];
    const SelectVal = event.item.panchayat;
    
    for (let i = 0; i < this.villageNewList.length; i++) {
      if (this.villageNewList[i].panchayat == SelectVal) {
        this.villageListFilter.push(this.villageNewList[i]);
      }
    }
    
    const gids = this.villageListFilter.map((o: { village: any; }) => o.village)
    const filteredGram = this.villageListFilter.filter(({ village }: any, index: number) => !gids.includes(village, index + 1));
    this.villageListFilter = filteredGram;
    
  }
 
  find(items: any, text: any) {
    text = text.split(' ');
    return items.filter((item: { villageName: string | any[]; }) => {
      return text.every((el: string) => {
        return item.villageName?.includes(el);
      });
    });
  }

  onSubmit(value: any): void {
    this.showFilter= false;
    this.village = this.villageNewList;
    if(value['districtName'] != null){
      
      this.village = this.village.filter((item: { district: any; }) => item.district == value['districtName']);
      if (value['blockName'] != null) {
        this.village = this.village.filter((item: { block: any; }) => item.block == value['blockName']);
        if (value['gramPanchayath'] != null) {
          this.village = this.village.filter((item: { panchayat: any; }) => item.panchayat == value['gramPanchayath']);
          if ((value['village'] != null)) {
            let tempVillage = this.village.filter((item: { village: any; }) => item.village == value['village']);
            this.village = this.village.filter((item: { village: any; }) => item.village == value['village']);
            
            if (value['shg'] != null) {
              
              this.village = this.find(tempVillage, value['shg']);
              
            }
          }
        }
      }
      this.common.closeActiveFilter();
      
    } 
    
  }

  clearFilter() {
    this.listAllVill();
    this.filterForm.reset();
    this.blockList = [];
    this.showFilter= false;
    this.grmList = [];
    this.villageListFilter = [];
    this.common.closeActiveFilter();
  }

}
