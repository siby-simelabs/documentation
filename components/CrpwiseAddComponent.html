import { ThrowStmt } from '@angular/compiler';
import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { TrackKpiService } from 'src/app/providers/trackKpi/track-kpi.service';

@Component({
  selector: 'app-crpwise-add',
  templateUrl: './crpwise-add.component.html',
  styleUrls: ['./crpwise-add.component.scss']
})
export class CrpwiseAddComponent implements OnInit { 

  monthData: any;
  flag:any;
  crpWise: any = [];
  submitted: boolean = false;
  formCreateCrp: FormGroup;
  dateFo: any;
  activityData: any;
  interventioList: any;
  finalProId: any;
  search: any = null;
  rowData: any;

  constructor(
    public track: TrackKpiService,
    public assign: AssigntargetService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public project: ProjectService
  ) {
    this.formCreateCrp = this.formBuilder.group({
      actualReportingDate: [null],
      crp: [null],
      actuals: [null],
    })
  }

  ngOnInit(): void {
    this.onload()
  }
  onload(){
    this.finalProId = this.common.getFundId();
    this.monthData = history.state.data.monthData;
    this.rowData = history.state.data
    this.flag = history.state.flag;
    
    this.dateFo = new Date(this.monthData.reportedDt);
    
    
    
    this.activityData = this.common.getAssingTarget();
    
    this.formCreateCrp.get("crp")?.setValue(this.rowData.kpi);
    this.formCreateCrp.get("actualReportingDate")?.setValue(this.dateFormat(this.monthData.reportedDt));
    this.formCreateCrp.get("actuals")?.setValue(this.monthData.actualAmount);
  }
  onSubmit(value: any) {
    this.submitted = true;
    if (this.formCreateCrp.invalid) {
      return;
    }
    else {
      
      let crpWise = {
        "interventionActualPostRequests": [{
          "yearId": this.monthData.year,
          "monthId": this.monthData.month,
          "actualReportingDate": this.dateFormat(this.monthData.reportedDt),
          "interventionId": this.activityData.interventionId,
          "userId": this.monthData.crpId,
          "actuals": value["actuals"],
          "approvalFlag": false,
          "activityId" : this.monthData.activityId,
          "personId": this.monthData.shgId
        }]
      };
      
      this.saveCrp(crpWise);
    }
  }
  dateFormat(date: string) {
    const today = new Date(date);
    const yyyy = today.getFullYear();
    let mm: any = today.getMonth() + 1;
    let dd: any = today.getDate();
    if (dd < 10) dd = '0' + dd;
    if (mm < 10) mm = '0' + mm;
    let formattedToday = dd + '/' + mm + '/' + yyyy;
    return formattedToday
  }

  saveCrp(data: any) {
    this.spinner.show();
    this.track.editActual(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        
        this.getKpiInter();
      }
      else {
        this.common.showError(resp.message);
      }

    })
  }
  onReset(): void {
    this.submitted = false;
    this.formCreateCrp.reset();
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateCrp.controls;
  }
  getKpiInter() {
    this.spinner.show();
    this.assign.getAllInterventionForKpi(this.finalProId, this.search).subscribe(resp => {
      
      this.spinner.hide()
      this.interventioList = resp.data.intervention_list;
      

      for (let i = 0; i < this.interventioList.length; i++) {
        
        
        if (this.interventioList[i].interventionId == this.activityData.interventionId) {
          this.common.saveAssingTarget(this.interventioList[i]);
          
          this.common.showSuccess('Successfully Updated');
          this.route.navigateByUrl('/dashboard/track-kpi/crpwise');
        }
      }
    })
  }

}
