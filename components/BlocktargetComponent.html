import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
import * as moment from 'moment';

@Component({
  selector: 'app-blocktarget',
  templateUrl: './blocktarget.component.html',
  styleUrls: ['./blocktarget.component.scss'],
})
export class BlocktargetComponent implements OnInit {
  targetDate: any;
  minDate = new Date();
  blockForm: FormGroup;
  year: any;
  total: any;
  submitted: boolean = false;
  assignBlk: any;
  interventionData: any;

  mdlSm: boolean = false;
  assignTlist: any;
  localblockList: any;
  blockData: any;
  activity: any;
  pageNo: number = 1;
  pageSize: number = 200;
  yearList: any;
  geoData: any;
  prid: any;
  finalProId: any;
  flag: string | undefined;
  interventioList: any;
  projectInfo: any;
  localflag: any;
  stateList: any;
  blockList: any;
  districtList: any;
  changeId: any;
  search: any = null;
  blockCode: any;
  mergeArray: any;
  tempBlockList: any;
  localObj: any;
  totalBlockTargetValue: number = 0;
  localTempList: any;
  deadMinDate: any;
  deadMaxDate: any;
  role: any;
  placement: TypeDatePickerPlacement = 'bottom';

  constructor(
    public common: CommonService,
    public assign: AssigntargetService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public geo: GeoService,
    public project: ProjectService,
    public dp: DatePickerPlacement
  ) {
    this.blockForm = this.formBuilder.group({
      year: [null],
      total: [null, Validators.required],
      targetDate: [null, Validators.required],
    });
  }
  filterForm = this.formBuilder.group({
    stateName: [null],
    districtName: [null],
    blockName: [null],
  });

  ngOnInit(): void {
    this.access();
    this.onload();
  }
  onload() {
    this.assignTlist = this.common.getYearWaise();
    this.localblockList = this.assignTlist.blocks;
    console.log(this.assignTlist.year);
    if (this.localblockList == null) this.localblockList = [];
    this.interventionData = this.common.getAssingTarget();
    this.finalProId = this.common.getFundId();
    this.activity = this.common.getActivity();
    this.prid = this.common.getFundId();
    this.flag = history.state.flag;
    this.localflag = this.common.getFlag();
    this.getState();

    this.blockForm.get('year')?.setValue(this.assignTlist.year);
    this.getGeoCov();

    this.project.getProjectByID(this.prid).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.deadMinDate = this.dateForMindate(
          this.projectInfo.actualStartDate
        );
        this.deadMaxDate = this.dateForMindate(this.projectInfo.actualEndDate);
        this.common.trigger(this.common.headerEvent, [
          {
            name: 'Project ID',
            value: this.projectInfo.id,
          },
          {
            name: 'Project Name',
            value: this.projectInfo.name,
          },
          {
            name: 'Int. Name',
            value: this.interventionData.name,
          },
          {
            name: 'KPI',
            value: this.activity.kpi,
          },
        ]);
      }
    });
  }
  access() {
    this.role = this.common.screenRole('assignblocktarget');
  }

  addCrp(crp: any) {
    this.common.saveBlockCrp(crp);
    this.route.navigateByUrl('/dashboard/assigntarget/crptarget', {
      state: { crp: crp, year: this.assignTlist.year, flag: this.flag },
    });
  }
  dateForMindate(date: any) {
    if (!date || typeof date !== 'string') {
      return;
    }
    let datearray = date.split('/');
    let newdate = datearray[1] + '/' + datearray[0] + '/' + datearray[2];

    return new Date(newdate);
  }

  getGeoCov() {
    this.geo
      .getGeoCoverage(this.pageNo, this.pageSize, this.prid)
      .subscribe((resp) => {
        if (resp) {
          this.geoData = resp.data?.geoCoverageList;
          let blockTemp: any = [];
          for (let g = 0; g < this.geoData.length; g++) {
            let tempData = {
              blockId: this.geoData[g].locationId,
              block_name: this.geoData[g].blockName,
              district_name: this.geoData[g].districtName,
              state_name: this.geoData[g].stateName,
              target: null,
              targetDate: null,
            };
            blockTemp.push(tempData);
          }
          this.geoData = blockTemp;
          this.mergeData();
        }
      });
  }

  mergeData() {
    this.tempBlockList = this.assignTlist.blocks;
    if (this.tempBlockList == null) this.tempBlockList = [];

    let ids = new Set(
      this.localblockList.map((d: { blockId: any }) => d.blockId)
    );
    let merged = [
      this.localblockList,
      this.geoData.filter((d: { blockId: number }) => !ids.has(d.blockId)),
    ];
    let mergedArr = [].concat.apply([], merged);

    this.localblockList = mergedArr;
    this.localTempList = mergedArr;
    this.totalBlockTarget();
  }

  showMdlSm(grp: any, type: string) {
    this.blockForm.get('total')?.setValue(grp.target);
    this.blockForm.get('targetDate')?.setValue(grp.targetDate);
    this.mdlSm = true;
    if (type == 'edit') {
      this.blockData = grp;
    } else {
      grp = {
        blockId: grp.locationId,
        target: null,
        targetDate: null,
        crp: [],
        block_name: grp.blockName,
      };
      this.blockData = grp;
    }
  }
  closeMdlSm() {
    this.mdlSm = false;
  }

  onSubmit(value: any) {
    this.submitted = true;

    if (this.blockForm.invalid) {
      return;
    } else {
      this.assignBlk = {
        interventionId: this.interventionData.interventionId,
        activityId: this.activity.activityId,
        outcomeId: this.interventionData.outcomeId,
        blockId: this.blockData.blockId,
        year: value['year'],
        target: value['total'],
        targetDate: (<HTMLInputElement>document.getElementById('targetDate'))
          .value,
      };

      let bt = this.blockData.target;

      if (bt == null) {
        bt = 0;
      }
      if (value['total'] > bt) {
        let grand;
        let tempTarget = value['total'] - bt;
        grand = this.totalBlockTargetValue + tempTarget;

        if (grand > this.assignTlist.target) {
          this.common.showError(
            'The target amount will not exceed the total amount for the year'
          );
        } else {
          this.saveBlk(this.assignBlk);
        }
      } else {
        this.saveBlk(this.assignBlk);
      }
    }
  }

  totalBlockTarget() {
    let tempTrgt = 0;
    for (let i = 0; i < this.localblockList.length; i++) {
      tempTrgt = tempTrgt + this.localblockList[i].target;
    }
    this.totalBlockTargetValue = tempTrgt;
  }

  saveBlk(data: any) {
    this.spinner.show();
    this.assign.addBlk(this.assignBlk).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == 'SUCCESS') {
        this.common.showSuccess('Successfully assigned block Target');
        this.mdlSm = false;
        this.getInterventionList();
        this.onReset();
      } else {
        this.common.showError(resp.message);
        this.spinner.hide();
      }
    });
  }
  onReset(): void {
    this.submitted = false;
    this.blockForm.reset();
  }
  get f(): { [key: string]: AbstractControl } {
    return this.blockForm.controls;
  }
  _saveActivity(i: number) {
    for (let j = 0; j < this.interventioList[i].activities.length; j++) {
      if (
        this.interventioList[i].activities[j].activityId ==
        this.activity.activityId
      ) {
        this.common.saveActivity(this.interventioList[i].activities[j]);
        for (
          let k = 0;
          k < this.interventioList[i].activities[j].targetSplit.length;
          k++
        ) {
          this.spinner.hide();
          if (
            this.interventioList[i].activities[j].targetSplit[k].year ==
            this.assignTlist.year
          ) {
            this.common.yearWaise(
              this.interventioList[i].activities[j].targetSplit[k]
            );
            this.onload();
          }
        }
      }
    }
  }
  getInterventionList() {
    this.spinner.show();
    this.assign
      .getAllIntervention(this.finalProId, this.search)
      .subscribe((resp) => {
        this.spinner.hide();
        this.interventioList = resp.data.intervention_list;
        for (let i = 0; i < this.interventioList.length; i++) {
          if (
            this.interventionData.interventionId ==
            this.interventioList[i].interventionId
          ) {
            this.common.saveAssingTarget(this.interventioList[i]);
            this._saveActivity(i);
          }
        }
        this.spinner.hide();
      });
  }
  getState() {
    this.common.getState().subscribe((resp) => {
      if (resp) {
        this.stateList = resp.data;
      }
    });
  }
  stateChange(event: any) {
    this.blockList = [];
    this.districtList = [];
    this.filterForm.get('districtName')?.setValue(null);
    this.filterForm.get('blockName')?.setValue(null);
    const SelectVal = event.item.locationCode;
    this.common.getDistrict(SelectVal).subscribe((resp) => {
      this.districtList = resp.data;
      this.changeId = SelectVal;
    });
  }
  distChange(event: any) {
    this.blockList = [];
    this.filterForm.get('blockName')?.setValue(null);
    const SelectVal = event.item.locationCode;
    this.common.getBlock(SelectVal, this.changeId).subscribe((resp) => {
      this.blockList = resp.data;
    });
  }
  blockChange(event: any) {
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.blockCode = SelectVal;
  }

  onSubmitFilter(value: any) {
    this.localObj = this.localTempList;
    if (value['stateName'] != null) {
      let state = this.localObj.filter(
        (item: { state_name: any }) => item.state_name == value['stateName']
      );
      this.localblockList = state;
      if (value['districtName'] != null) {
        let dist = state.filter(
          (item: { district_name: any }) =>
            item.district_name == value['districtName']
        );
        this.localblockList = dist;
        if (value['blockName'] != null) {
          let blk = dist.filter(
            (item: { block_name: any }) => item.block_name == value['blockName']
          );
          this.localblockList = blk;
        }
      }
    }
    this.common.closeActiveFilter();
  }
  clearFilter() {
    this.filterForm.reset();
    if (this.localObj?.length > 0) {
      this.localblockList = this.localTempList;
    }
    this.common.closeActiveFilter();
  }
  getTotalTarget() {
    return this.totalBlockTargetValue;
  }
  geAssignedTarget() {
    return this.totalBlockTargetValue;
  }
}
