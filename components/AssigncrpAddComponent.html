import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-assigncrp-add',
  templateUrl: './assigncrp-add.component.html',
  styleUrls: ['./assigncrp-add.component.scss']
})
export class AssigncrpAddComponent implements OnInit {
  crpId: any;
  formCreateCrp: FormGroup;
  crpInfo: any;
  proName: any;
  blkId = 1601001;
  village: any;
  selectedCheck: any = [];
  overhide: boolean = true;
  searchData: any;
  currentPage = 1;
  personId: any;
  flag: any;
  villageview: boolean = false;
  personData: any;
  submitted: boolean = false;
  id: any;
  userData: any;
  username: string | undefined;
  state = 16;
  districtList: any;
  blockList: any;
  blockCode: any;
  changeId: any;
  grmList: any = [];
  villageNewList: any = [];
  selectAllCheck: boolean = false;
  tempSelect: any = [];
  readonly: boolean = false;
  showFilter:boolean = false;
  geoDatas:any = [];
  villageListFilter: any = [];
  activeCount: number = 0;
  role:any;

  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public spinner: NgxSpinnerService,
    public geoData: GeoService
  ) {
    this.formCreateCrp = this.formBuilder.group({
      userName: [null, Validators.required],
      id: [null]
    })
  }
  filterForm = this.formBuilder.group(
    {
      gramPanchayath: [null],
      districtName: [null],
      blockName: [null],
      villageName: [null]
    },
  );
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }


  ngOnInit(): void {
    this.access();
    this.crpId = this.common.getCrpId();
    this.getDist();
    this.personId = history.state.personId;
    this.flag = history.state.flag;

    if (this.flag == "view") {
      this.villageview = true
    }
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {

        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    });
    this.getUser();
    this.getAllBlockList();
    this.geoData.getGeoCoverage(0, 1, this.crpId).subscribe((resp) => {
      this.geoDatas = resp.data?.geoCoverageList;
    })

  }
  access() {
    this.role = this.common.screenRole("assigncrp");
  }
  userSelectedValue(username: any) {
    if (username == this.username) {

      return true;
    }
    else {
      return false
    }
  }
  getUser() {
    this.manager.getCrpUserKerala(this.crpId).subscribe((resp) => {
      if (resp) {
        let usrnm: any;

        this.personData = resp.data;
        if ((this.flag == "edit") || (this.flag == "view")) {
          this.readonly = true;
          for (let i = 0; i < this.personData.length; i++) {
            if (this.personData[i].id == this.personId) {
              this.formCreateCrp.get("id")?.setValue(this.personData[i].id);
              this.formCreateCrp.get("userName")?.setValue(this.personData[i].userName);
              usrnm = this.personData[i].userName

            }
          }
          setTimeout(() => {
            this.username = usrnm;
          }, 1500);
        }


        this.listAllVill();
        this.common.trigger('updateHeaderDetails', [{
          name: 'Project ID',
          value: this.crpInfo.id
        }, {
          name: "Project Name",
          value: this.crpInfo.name
        }, {
          name: 'State',
          value: this.geoDatas[0]?.stateName
        }])
      }
    })
  }
  listAllVill() {
    this.village = [];
    this.villageNewList = [];
    this.selectedCheck = [];
    this.manager.listAllVillagebyAssign(this.crpId, this.flag, this.personId).subscribe((resp) => {
      if (resp) {
        this.village = resp.data;
        this.villageNewList = resp.data;
        this.getLengthOfActiveUser(resp.data)
        if ((this.flag == "edit") || (this.flag == "view")) {
          for (let i = 0; i < this.village.length; i++) {
            if (this.village[i].selected == true) {
              this.selectedCheck.push(this.village[i]);
            }
          }
        }
        if (this.selectedCheck?.length == this.village?.length) {
          this.selectAllCheck = true;
        }
      }
    })
  }
  checkValue(event: any, code: any) {

    let indexVal;

    if (event.currentTarget.checked == true) {
      for (let i = 0; i < this.village.length; i++) {
        if (this.village[i].locationCode == code) {
          indexVal = this.village[i];
          indexVal.selected = true;
          indexVal = {
            "villageName": indexVal.villageName,
            "locationCode": indexVal.locationCode,
            "selected": true,
            "disabled": indexVal.disabled
          }
          this.selectedCheck.push(indexVal);
          this.tempSelect.push(indexVal);

        }
      }
    }
    else {
      this.selectedCheck = this.selectedCheck.filter((item: { locationCode: any; }) => item.locationCode !== code);
      this.tempSelect = this.tempSelect.filter((item: { locationCode: any; }) => item.locationCode !== code);
    }
  }
  removeCheckd(event: any, code: any) {
    if (event.currentTarget.checked == false) {
      this.selectedCheck = this.selectedCheck.filter((item: { locationCode: any; }) => item.locationCode !== code);

      let obj = this.village.find((village: { locationCode: number; }) => village.locationCode == code);
      if (obj) {
        obj.selected = false;
      }


    }
  }
  getDist() {
    this.common.getDistrict(this.state).subscribe((resp) => {
      this.districtList = resp.data;
    });
  }

  blockChange(event: any) {

    this.grmList = [];
    this.villageListFilter = []
    const SelectVal = event.item.blockName; 
    for (let i = 0; i < this.villageNewList.length; i++) {
      if (this.villageNewList[i].blockName == SelectVal) {
        this.grmList.push(this.villageNewList[i]);
      }
    }
    const gids = this.grmList.map((o: { gramapanchayat: any; }) => o.gramapanchayat)
    const filteredGram = this.grmList.filter(({ gramapanchayat }: any, index: number) => !gids.includes(gramapanchayat, index + 1));
    this.grmList = filteredGram;

  }

  gramChange(event: any) {
    console.log(event.item);
    this.villageListFilter = [];
    const SelectVal = event.item.panchayat;
    
    for (let i = 0; i < this.villageNewList.length; i++) {
      if (this.villageNewList[i].panchayat == SelectVal) {
        this.villageListFilter.push(this.villageNewList[i]);
      }
    }
    
    const gids = this.villageListFilter.map((o: { villageName: any; }) => o.villageName)
    console.log(gids);
    const filteredGram = this.villageListFilter.filter(({ villageName }: any, index: number) => !gids.includes(villageName, index + 1));
    console.log(filteredGram);
    this.villageListFilter = filteredGram;
    console.log(this.villageListFilter);
    
  }

  saveVillage() {
    if (this.formCreateCrp.get("id")?.value != null) {
      if (this.selectedCheck.length > 0) {
        let str = JSON.stringify(this.selectedCheck);
        str = str.replace(/locationCode/g, 'villageId');
        let object = JSON.parse(str);

        let data = {
          "villageList": object,
          "projectId": this.crpId,
          "personId": this.personId,
          "status": true,
        }
        if(this.activeCount>=this.villageNewList.length){
          this.common.showError("The selection exceeds the no of villages already identified!");
        }
        else{
          this.assignVillage(data);
        }
      }
      else {
        this.common.showError("Please select vilage(s)");
      }
    }
    else {
      this.common.showError("Please select a CRP");
    }
  }
  getLengthOfActiveUser(villageList:any) {
    const active = villageList.filter((village: { disabled: boolean; }) => village.disabled == true
    );
    this.activeCount = active.length;
     if(this.activeCount>=this.villageNewList.length){
      this.common.showError("CRPs have already been assigned to the target number of villages!");
    }
  }
  assignVillage(data: any) {
    this.manager.assignVillages(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {

        this.common.showSuccess('Successfully assign CRPs');
        this.route.navigateByUrl('/dashboard/crp/assigncrp');
      }
      else {
        this.common.showError(resp.message);
      }
    })
  }

  selectAll(event: any) {
    if (event.currentTarget.checked == true) {
      for (let i = 0; i < this.village.length; i++) {
        this.village[i].selected = true;
      }
      if (this.selectedCheck.length > 0) {

        this.selectedCheck = this.selectedCheck.concat(this.village)
      }
      else {

        this.selectedCheck = this.village;
      }

    }
    else {
      for (let i = 0; i < this.village.length; i++) {
        this.village[i].selected = false;
      }
      this.selectedCheck = [];
      this.tempSelect = [];
    }
  }
  nameChange(event: any) {
    this.id = "";

    this.userData = event.item;
    this.personId = this.userData.id;
    this.formCreateCrp.get("id")?.setValue(this.userData.id);
  }
  clearFilter() {
    this.listAllVill();
    this.filterForm.reset();
    this.searchData = [];
    this.currentPage = 1;
    this.showFilter = false;
  }
  getAllBlockList() {
    this.spinner.show();
    this.manager.getBlkList(this.crpId).subscribe(resp => {
      this.spinner.hide();

      if (resp) {
        this.blockList = resp.data;

      }
    })
  }
  onSubmit(value: any): void {
    this.showFilter = false;
    this.village = this.villageNewList;
    if (value['blockName'] != null) {
      this.village = this.village.filter((item: { blockName: any; }) => item.blockName == value['blockName']);
      if (value['gramPanchayath'] != null) {
        this.village = this.village.filter((item: { gramapanchayat: any; }) => item.gramapanchayat == value['gramPanchayath']);
        let villageNameValue = value['villageName'];
        if (villageNameValue == "") { villageNameValue = undefined }
        if ((villageNameValue != null) || (villageNameValue != "") || (villageNameValue != undefined)) {
          this.village = this.find(this.village, villageNameValue?.toUpperCase());
        }
      }

    }
  }
  find(items: any, text: any) {
    text = text.split(' ');
    return items.filter((item: { villageName: string | any[]; }) => {
      return text.every((el: string) => {
        return item.villageName?.includes(el);
      });
    });
  }
}
