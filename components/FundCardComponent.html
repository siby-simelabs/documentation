import { Component, OnInit, Input } from '@angular/core';
import { CommonService } from 'src/app/providers/common.service';
@Component({
  selector: 'app-fund-card',
  templateUrl: './fund-card.component.html',
  styleUrls: ['./fund-card.component.scss'],
})
export class FundCardComponent {
  cards: FundCard[] = [];
  constructor(private common: CommonService) {
    this.common.subscribe(this.onCards, 'cards');
  }
  onCards = (cards: FundCard[]) => {
    this.cards = cards.filter(( item )=>item && item.value )
  };
  ngOnDestroy() {
    this.common.subscribe(this.onCards, 'cards');
  }
}
