import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { BsModalService, BsModalRef, ModalOptions } from 'ngx-bootstrap/modal';
import { HistoryComponent } from '../distlevel/history/history.component';

@Component({
  selector: 'app-blocklevel',
  templateUrl: './blocklevel.component.html',
  styleUrls: ['./blocklevel.component.scss']
})
export class BlocklevelComponent implements OnInit { 
  pageNo: number = 1;
  currentPage = 1;
  searchData: any = [];
  submitted: boolean = false;
  crpId:any; 
  datas: any;
  crpInfo: any;
  proName: any;
  id:any;
  blockList:any = [];
  districtList:any = [];
  changeId: any;
  blockCode: any;
  districtName:any;
  stateList: any;
  bsModalRef?: BsModalRef;
  role: any;
  constructor(
    public manager: ManagerService,
    public common:CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public project:ProjectService,
    public budget:BudgetService,
    public spinner: NgxSpinnerService,
    private modalService: BsModalService,
  ) { }
  filterForm = this.formBuilder.group(
    {
      stateName: [null],
      districtName: [null],
      blockName:[null]
    },
  );
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }

  ngOnInit(): void {
    this.spinner.hide();
    this.access();
    this.crpId = this.common.getCrpId();
    this.getState()
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    })
    this.getAllBlockList();
  }
  access() {
    this.role = this.common.screenRole("blocklevel");
  }
  getAllBlockList(){
    this.spinner.show();
    this.manager.getBlkList(this.crpId).subscribe(resp=>{
      this.spinner.hide();
      if(resp){
        this.datas=resp.data;
        this.common.trigger('updateHeaderDetails',[{
          name : 'Project ID' ,
          value : this.crpInfo.id 
        },{
          name : "Project Name" ,
          value : this.crpInfo.name
        },{
          name : 'State' ,
          value : this.datas[0]?.stateName 
        }])
       }
    })
  }
  onSubmit(value:any):void {
    this.submitted = true;
    if (this.filterForm.invalid) {
      return;
    }
    else {
      this.searchData =
      {
        "searchModel": {
          "projectId":this.crpId,
          "stateName": value["stateName"],
          "districtName":value["districtName"],
          "blockName":value["blockName"]
        },
        "pageNumber": 0,
        "pageSize": 100
      }
      this.searchFile(this.searchData);
    }

  }
  searchFile(data: any) {
    this.spinner.show();
    this.manager.filterBlock(data).subscribe((resp) => {
      this.common.closeActiveFilter();
      this.spinner.hide();
      if (resp) {
        this.datas = resp.data;
       }
    })
  }
  clearFilter() {
    this.getAllBlockList();
    this.filterForm.reset();
    this.searchData = [];
    this.currentPage = 1;
    this.common.closeActiveFilter();
  }
  editBlk(group:any){
    this.route.navigateByUrl('/dashboard/crp/add-blocklevel', { state: { group:group } });
  }
  getState(){
    this.common.getState().subscribe((resp) => {
      if (resp) {
        this.stateList = resp.data;
      } 
    })
  }
  stateChange(event: any) {
    this.blockList = [];
    this.districtList = [];
    this.filterForm.get("districtName")?.setValue(null);
    this.filterForm.get("blockName")?.setValue(null);

    const SelectVal = event.item.locationCode;
    this.common.getDistrict(SelectVal).subscribe((resp) => {
      this.districtList = resp.data;
      this.changeId = SelectVal;
    });
  }
  distChange(event: any) {
    
    this.blockList = [];
    this.filterForm.get("blockName")?.setValue(null);
    const SelectVal = event.item.locationCode;
    this.common.getBlock(SelectVal, this.changeId).subscribe((resp) => {
      this.blockList = resp.data;
    })
  }
  blockChange(event:any){
    const SelectVal = event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.blockCode = SelectVal;
    
  }
  openModalWithComponent(id: any) {
    const initialState: ModalOptions = {
      initialState: {
        locId: id,
        endpoint: "getBlockLevelHistory"
      },
      class:"modal-lg"
    };
    this.bsModalRef = this.modalService.show(HistoryComponent, initialState);
    this.bsModalRef.content.closeBtnName = 'Close';
  }

}
