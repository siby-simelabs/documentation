import { Component, OnInit, ViewChild } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { ProjectService } from 'src/app/providers/project/project.service';
import { CommonService } from 'src/app/providers/common.service';
import { Validator } from 'src/app/_helpers/validators';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
@Component({
  selector: 'app-profile',
  templateUrl: './profile.component.html',
  styleUrls: [],
})
export class ProfileComponent implements OnInit {
  placement : TypeDatePickerPlacement = 'bottom' ;
  searchResult = [];
  projects: any = [];
  submitted = false;
  // formCreateProject: FormGroup;
  proposedStartDate: any;
  proposedEndDate: any;
  actualStartDate: any;
  actualEndDate: any;
  ecApprovalDate: any;
  sanctionLetterIssuedDate: any;
  searchData: any = [];
  linkedParentProject: any;
  linkedParentProjectDesc: any;
  projectId: any;
  projectFlag: any;
  name: any;
  mode: any;
  fundingSource: any;
  description: any;
  proId: any = '';
  keyword: any = 'name';
  vewId: any;
  modeOfProjectModelList: any;
  fundingSourceModelList: any;
  datas: any;
  formArray: any;
  proposedEndminDate: any;
  actualStartminDate: any;
  proposedStartminDate: any;
  actualEndminDate: any;
  PropoValidDateCheck: boolean = false;
  role: any;
  roleBase: any;
  linkedParentProjectId: any;
  ecMinDate: any;
  ecMaxDate: any;
  sanMinDate: any;
  sanMaxDate: any;
  actualStartmaxDate: any;
  @ViewChild('aziendaAutocomplete') aziendaAutocomplete: any;
  code='en';

  constructor(
    public common: CommonService,
    public formBuilder: FormBuilder,
    public validator: Validator,
    public route: Router,
    public project: ProjectService,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService ,
    public dp : DatePickerPlacement
  ) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId();
    this.activatedRoute.data.subscribe((value) => {
      this.projectFlag = value;
    });
  }
  ngOnInit() {
    this.access();
    this.editFlag();
    this.proposedStartminDate = new Date();
    this.ecMinDate = new Date();
    this.sanMinDate = new Date();
    this.project.getMode(this.code).subscribe((resp) => {
      if (resp) {
        this.modeOfProjectModelList =
          resp.data;
        // this.fundingSourceModelList =
        //   resp.data.fundingSourceModelList.FundingSource.toString().split(',');
      }
    });
    this.project.getFunding(this.code).subscribe((resp) => {
      if (resp) {
        this.fundingSourceModelList =
          resp.data;
      }
    });
  }
  access() {
    this.role = this.common.screenRole('project_profile');
    this.roleBase = this.common.getRoleBase();
    if (this.roleBase.name != 'pm_central_user_level1_final_approval') {
      this.formCreateProject.controls['actualStartDate'].disable();
      this.formCreateProject.controls['actualEndDate'].disable();
      this.formCreateProject.controls['ecApprovalDate'].disable();
      this.formCreateProject.controls['sanctionLetterIssuedDate'].disable();
    } else {
      this.formCreateProject.controls['name'].disable({ onlySelf: true });
      this.formCreateProject.controls['mode'].disable({ onlySelf: true });
      this.formCreateProject.controls['fundingSource'].disable({
        onlySelf: true,
      });
      this.formCreateProject.controls['description'].disable({
        onlySelf: true,
      });
      this.formCreateProject.controls['proposedStartDate'].disable({
        onlySelf: true,
      });
      this.formCreateProject.controls['proposedEndDate'].disable({
        onlySelf: true,
      });
      this.formCreateProject.controls['linkedParentProjectDesc'].disable({
        onlySelf: true,
      });
      this.formCreateProject.controls['linkedParentProject'].disable({
        onlySelf: true,
      });

      this.formCreateProject.controls['actualStartDate'].setValidators(
        Validators.required
      );
      this.formCreateProject.controls['actualEndDate'].setValidators(
        Validators.required
      );
      this.formCreateProject.controls['ecApprovalDate'].setValidators(
        Validators.required
      );
      this.formCreateProject.controls['sanctionLetterIssuedDate'].setValidators(
        Validators.required
      );

      this.formCreateProject.controls[
        'actualStartDate'
      ].updateValueAndValidity();
      this.formCreateProject.controls['actualEndDate'].updateValueAndValidity();
      this.formCreateProject.controls[
        'ecApprovalDate'
      ].updateValueAndValidity();
      this.formCreateProject.controls[
        'sanctionLetterIssuedDate'
      ].updateValueAndValidity();
    }
    if (this.roleBase.name == 'pm_central_user_level2') {
      this.formCreateProject.controls['actualStartDate'].enable();
      this.formCreateProject.controls['actualEndDate'].enable();
      this.formCreateProject.controls['ecApprovalDate'].enable();
      this.formCreateProject.controls['sanctionLetterIssuedDate'].enable();

      this.formCreateProject.controls['actualStartDate'].setValidators(
        Validators.required
      );
      this.formCreateProject.controls['actualEndDate'].setValidators(
        Validators.required
      );
      this.formCreateProject.controls['ecApprovalDate'].setValidators(
        Validators.required
      );
      this.formCreateProject.controls['sanctionLetterIssuedDate'].setValidators(
        Validators.required
      );

      this.formCreateProject.controls[
        'actualStartDate'
      ].updateValueAndValidity();
      this.formCreateProject.controls['actualEndDate'].updateValueAndValidity();
      this.formCreateProject.controls[
        'ecApprovalDate'
      ].updateValueAndValidity();
      this.formCreateProject.controls[
        'sanctionLetterIssuedDate'
      ].updateValueAndValidity();
    }
  }

  formCreateProject = this.formBuilder.group({
    name: [null, Validators.required],
    mode: [null, Validators.required],
    fundingSource: [null, Validators.required],
    description: [],
    proposedStartDate: [null, Validators.required],
    proposedEndDate: [null, Validators.required],
    actualStartDate: [null],
    actualEndDate: [null],
    ecApprovalDate: [null],
    sanctionLetterIssuedDate: [null],
    linkedParentProject: [null],
    linkedParentProjectDesc: [null],
  });
  editFlag() {
    if (this.projectFlag.flag == 'edit' || this.projectId) {
      this.spinner.show();
      this.project.getProjectByID(this.projectId).subscribe((resp) => {
        if (resp) {
          this.spinner.hide();

          this.datas = resp.data;
          this.common.trigger('updateHeaderDetails', [
            { name: 'Project ID', value: resp.data.id },
            { name: 'Project Name', value: resp.data.name },
          ]);
          this.proId = this.datas.id;
          this.linkedParentProjectId = this.datas.linkedParentProjectId;
          this.formCreateProject.get('name')?.setValue(this.datas.name);
          this.formCreateProject.get('mode')?.setValue(this.datas.mode);
          this.formCreateProject
            .get('fundingSource')
            ?.setValue(this.datas.fundingSource);
          this.formCreateProject
            .get('description')
            ?.setValue(this.datas.description);
          this.formCreateProject
            .get('proposedStartDate')
            ?.setValue(this.datas.proposedStartDate);
          this.formCreateProject
            .get('proposedEndDate')
            ?.setValue(this.datas.proposedEndDate);
          this.formCreateProject
            .get('actualStartDate')
            ?.setValue(this.datas.actualStartDate);
          if (this.datas.actualStartDate) {
            this.ecMaxDate = this.dateForMindate(this.datas.actualStartDate);
            this.sanMaxDate = this.dateForMindate(this.datas.actualStartDate);
          }
          if (this.datas.proposedEndDate != null) {
            // let date = new Date(this.datas.actualStartDate);
            this.actualStartmaxDate = this.dateForMindate(
              this.datas.proposedEndDate
            );
          }

          if (this.datas.proposedStartDate != null) {
            this.actualStartminDate = this.dateForMindate(
              this.datas.proposedStartDate
            );

            this.proposedEndminDate = this.dateForMindate(
              this.datas.proposedStartDate
            );
          }
          this.formCreateProject
            .get('actualEndDate')
            ?.setValue(this.datas.actualEndDate);
          this.formCreateProject
            .get('ecApprovalDate')
            ?.setValue(this.datas.ecApprovalDate);
          if (this.datas.ecApprovalDate) {
            // ecMinDate
          }
          this.formCreateProject
            .get('sanctionLetterIssuedDate')
            ?.setValue(this.datas.sanctionLetterIssuedDate);
          this.formCreateProject
            .get('linkedParentProject')
            ?.setValue(this.datas.linkedParentProject);
          this.formCreateProject
            .get('linkedParentProjectDesc')
            ?.setValue(this.datas.linkedParentProjectDesc);
         }
      });
    } else {
      this.project.clearProject();
    }
  }

  get f(): { [key: string]: AbstractControl } {
    return this.formCreateProject.controls;
  }

  dateForMindate(date: any) {
    if (!date || typeof date !== 'string') return;
    let dateParts: any = date.split('/');
    let dateObject = new Date(+dateParts[2], dateParts[1] - 1, +dateParts[0]);
    dateObject.setDate(dateObject.getDate());
    let localDate = dateObject.toLocaleDateString();
    return new Date(localDate);
  }

  onSubmit(value: any): void {
    this.submitted = true;

    if (this.formCreateProject.invalid) {
      return;
    } else {
      let act: any = (<HTMLInputElement>(
        document.getElementById('actualStartDate')
      )).value;
      if (act == '') {
        act = null;
      }
      let actend: any = (<HTMLInputElement>(
        document.getElementById('actualEndDate')
      )).value;
      if (actend == '') {
        actend = null;
      }
      let ecapp: any = (<HTMLInputElement>(
        document.getElementById('ecApprovalDate')
      )).value;
      if (ecapp == '') {
        ecapp = null;
      }
      let sanc: any = (<HTMLInputElement>(
        document.getElementById('sanctionLetterIssuedDate')
      )).value;
      if (sanc == '') {
        sanc = null;
      }
      let newData = {
        name: this.formCreateProject.getRawValue()?.name,
        mode: this.formCreateProject.getRawValue()?.mode,
        fundingSource: this.formCreateProject.getRawValue()?.fundingSource,
        description: this.formCreateProject.getRawValue()?.description,
        proposedStartDate: (<HTMLInputElement>(
          document.getElementById('proposedStartDate')
        )).value,
        proposedEndDate: (<HTMLInputElement>(
          document.getElementById('proposedEndDate')
        )).value,
        actualStartDate: act,
        actualEndDate: actend,
        ecApprovalDate: ecapp,
        sanctionLetterIssuedDate: sanc,
        linkedParentProject:
          this.formCreateProject.getRawValue()?.linkedParentProject,
        linkedParentProjectDesc: (<HTMLInputElement>(
          document.getElementById('linkedParentProjectDesc')
        )).value,
        id: null,
        linkedParentProjectId: this.linkedParentProjectId,
      };

      if (this.projectFlag.flag == 'edit' || this.projectId) {
        this.formCreateProject.addControl(
          'id',
          new FormControl('', Validators.required)
        );
        newData['id'] = this.projectId;
      }
      this.SaveProjectInfo(JSON.stringify(newData));
    }
  }

  onReset(): void {
    this.submitted = false;
    this.formCreateProject.reset();
  }

  SaveProjectInfo(data: any) {
    this.spinner.show();
    this.project.addProject(data).subscribe((resp) => {
      if (resp) {
        this.spinner.hide();
        if (this.projectFlag.flag == 'edit' || this.projectId) {
          this.common.showSuccess('Project successfully Updated');
        } else {
          this.common.showSuccess('Project successfully created');
        }
        this.common.storeProjectId(resp.data.id);
        this.route.navigate(['/dashboard/project/intervention']);
      }
    });
  }

  onChangeSearch(event: any) {
    this.linkedParentProject = '';
    if (event.target.value.length >= 1) {
      this.project.searchProjectByName(event.target.value).subscribe((resp) => {
        if (resp) {
          this.searchResult = resp.data;
        }
      });
    }
  }

  selectEvent(event: any) {
    this.formCreateProject
      .get('linkedParentProjectDesc')
      ?.setValue(event.item.description);
    this.linkedParentProjectId = event.item.id;
  }

  onCleared(event: any) {
    this.linkedParentProjectDesc = null;
    this.linkedParentProject = null;
    this.linkedParentProjectId = null;
  }

  onValueChange(event: any, value: any) {
    if (value == 'proposedStartDate') {
      this.formCreateProject.get('proposedEndDate')?.setValue('');
      this.proposedEndminDate = event;
      this.actualStartminDate = event;
    }
    if (value == 'actualStartDate') {
      this.formCreateProject.get('actualEndDate')?.setValue('');
      this.actualEndminDate = event;
      this.ecMaxDate = event;
      this.sanMaxDate = event;
    }
    if (value == 'proposedEndDate') {
      this.actualStartmaxDate = event;
    }
  }

  modeChnage(e: any) {
    if (this.projectFlag.flag == 'edit' || this.projectId) {
      if (
        confirm(
          'Changing the mode of project will affect the Beneficiary Coverage and indicators/measures configurations!'
        )
      ) { 
      }
    }
  }
  unlinkPr() {
    this.formCreateProject.get('linkedParentProjectDesc')?.setValue(null);
    this.linkedParentProjectId = null;
    this.formCreateProject.get('linkedParentProject')?.setValue(null);
  }
}
