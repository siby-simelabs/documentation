import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { OutcomeService } from 'src/app/providers/admin/kpiconfig/outcome/outcome.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Validator } from 'src/app/_helpers/validators';

@Component({
  selector: 'app-outcome',
  templateUrl: './outcome.component.html',
  styleUrls: [],
})
export class OutcomeComponent implements OnInit {
  formCreateOutcome: FormGroup;
  submitted = false;
  id: any;
  sector: any;
  interventionCategory: any;
  interventionSubCategory: any;
  outComeLevel: any;
  outcome: any;
  status: any;
  parms: any;
  outCome: any;
  data: any;
  cato: any;
  subcato: any;
  outcomeId: any;
  outview: any;
  expOutString: string = 'Select Project Level';
  sectorlist: any;
  flag: string = 'add';
  role: any;

  constructor(
    public out: OutcomeService,
    public common: CommonService,
    public formBuilder: FormBuilder,
    public validator: Validator,
    public route: Router,
    public project: ProjectService,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService
  ) {
    this.outview = this.common.getOutcomeView();
    this.outcomeId = this.common.getOutcomeId();
    this.parms = history.state.id;

    this.formCreateOutcome = this.formBuilder.group({
      sector: ['', Validators.required],
      interventionCategory: ['', Validators.required],
      interventionSubCategory: [null],
      outComeLevel: ['', Validators.required],
      outcome: [
        '',
        Validators.compose([Validators.required, Validators.maxLength(200)]),
      ],
      status: ['', Validators.required],
      description: [''],
    });
  }

  ngOnInit(): void {
    this.access();
    this.spinner.hide();
    this.outcomeId = this.common.getOutcomeId();

    this.getMaster();
    if (this.parms || this.outcomeId) {
      this.flag = 'edit';
      if (this.outview) this.flag = 'view';
      this.spinner.show();
      this.out.getOutById(this.outcomeId).subscribe((resp) => {
        this.spinner.hide();

        if (resp) {
          this.outCome = resp.data;
          this.formCreateOutcome.get('sector')?.setValue(this.outCome.sector);
          this.formCreateOutcome
            .get('interventionCategory')
            ?.setValue(this.outCome.interventionCategory);
          this.formCreateOutcome
            .get('interventionSubCategory')
            ?.setValue(this.outCome.interventionSubCategory);
          this.formCreateOutcome
            .get('outComeLevel')
            ?.setValue(this.outCome.outComeLevel);
          if (this.outCome.outComeLevel)
            this.changeOutLevel(this.outCome.outComeLevel);
          this.formCreateOutcome.get('outcome')?.setValue(this.outCome.outcome);
          this.formCreateOutcome.get('status')?.setValue(this.outCome.status);
          this.formCreateOutcome
            .get('description')
            ?.setValue(this.outCome.description);
          setTimeout(() => {
            this.fillSectorDdl(
              this.outCome.sector,
              this.outCome.interventionCategory,
              this.outCome.subCategory
            );
          }, 500);
        }
      });
    }
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateOutcome.controls;
  }
  access() {
    this.role = this.common.screenRole('kpiconfig');
  }

  onSubmit() {
    this.submitted = true;
    if (this.formCreateOutcome.invalid) {
      return;
    }
    if (this.outcomeId) {
      this.formCreateOutcome.addControl(
        'id',
        new FormControl('', Validators.required)
      );
      this.formCreateOutcome.controls['id'].setValue(this.outcomeId);
    }

    if (this.formCreateOutcome.get('interventionSubCategory')?.value == '') {
      this.formCreateOutcome.get('interventionSubCategory')?.enable();
      this.formCreateOutcome.get('interventionSubCategory')?.setValue('nill');
    }
    this.formCreateOutcome.addControl(
      'lastUpdatedBy',
      new FormControl('', Validators.required)
    );
    this.formCreateOutcome.controls['lastUpdatedBy'].setValue('user');

    this.SaveOut(JSON.stringify(this.formCreateOutcome.value, null, 2));
  }
  SaveOut(data: any) {
    this.spinner.show();
    this.out.saveOutcome(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.common.storeOutcome(resp.data.id);

        if (this.parms || this.outcomeId) {
          this.common.showSuccess('Successfully Updated Outcome');
          this.route.navigate(['/dashboard/admin/event']);
        } else {
          this.common.showSuccess('Successfully Created Outcome');
          this.route.navigate(['/dashboard/admin/event']);
        }
      }
    });
  }

  onReset(): void {
    this.submitted = false;
    this.formCreateOutcome.reset();
  }

  getMaster() {
    this.common.getMasterData().subscribe((resp) => {
      if (resp) {
        this.data = resp.data;
      }
    });
    this.common.getAllSectors().subscribe((resp) => {
      if (resp) {
        this.sectorlist = resp.data;
      }
    });
  }
  sectorChange(event: any) {
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    if (this.parms || this.outcomeId) {
      if (
        confirm(
          'Change in sector will affect the already existing event/activity'
        ) == true
      ) {
        this.sectorChnageEvent(SelectVal);
      } else {
        this.formCreateOutcome.get('sector')?.setValue(this.outCome.sector);
      }
    } else {
      this.sectorChnageEvent(SelectVal);
    }
    if (event.target.value == 'Value Chain') {
      this.formCreateOutcome.get('interventionSubCategory')?.enable();
    } else {
      this.formCreateOutcome.get('interventionSubCategory')?.disable();
    }
  }

  sectorChnageEvent(SelectVal: number) {
    this.formCreateOutcome.get('interventionCategory')?.setValue('');
    this.formCreateOutcome.get('interventionSubCategory')?.setValue('');
    this.cato = [];
    this.subcato = [];
    this.getCato(SelectVal);

  }
  getCato(SelectVal: number) {
    this.common.getCato(SelectVal).subscribe((resp) => {
      if (resp) {
        this.cato = resp.data;
      }
    });
  }
  catoChange(event: any) {
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.getSubcato(SelectVal);
  }
  getSubcato(SelectVal: number) {
    this.common.getSubCato(SelectVal).subscribe((resp) => {
      if (resp) {
        this.subcato = resp.data;
      }
    });
  }
  projectLevel(event: any) {
    this.changeOutLevel(event.target.value);
  }

  changeOutLevel(plan: any) {
    if (plan == 'Project level') {
      this.expOutString = 'Expected Outcome';
    } else if (plan == 'Planning Adoption') {
      this.expOutString = 'Expected Practice';
    }
  }
  _getSubCategory(cato: any) {
    for (let i: any = 0; i < this.cato.length; i++) {
      if ((cato == this.cato[i].masterValue)) {
        this.common.getSubCato(this.cato[i]?.id).subscribe((resp) => {
          if (resp) {
            this.subcato = resp.data;
          }
        });
      }
    }
  }
  fillSectorDdl(sector: any, cato: any, subcato: any) {
    for (let i: any = 0; i < this.sectorlist.length; i++) {
      if (sector == this.sectorlist[i]?.masterValue) {
        this.common.getCato(this.sectorlist[i]?.id).subscribe((resp) => {
          if (resp) {
            this.cato = resp.data;
            this._getSubCategory(cato);
          }
        });
      }
    }
  }
}
