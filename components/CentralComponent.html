import { Component, OnInit } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { CommonService } from 'src/app/providers/common.service';
import { FundService } from 'src/app/providers/fund/fund.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { UtilizationService } from 'src/app/providers/utilization/utilization.service';

@Component({
  selector: 'app-central',
  templateUrl: './central.component.html',
  styleUrls: ['./central.component.scss']
})
export class CentralComponent implements OnInit {
    
  projId: any;
  proName: any;
  proStatus: any;
  sector: any;
  budgetId:any;
  datas: any;
  id: any;
  fundId: any;
  total: any;
  fundInfo: any;
  budgetData: any;
  role:any;

  constructor(
    public fund : FundService,
    public utilization:UtilizationService,
    public common:CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public project:ProjectService,
    public budget:BudgetService,
    public spinner: NgxSpinnerService
  ) { }

  ngOnInit(): void {
    this.access();
    this.spinner.hide();
    this.fundId = this.common.getFundId();
    this.project.getProjectByID(this.fundId).subscribe((resp) => {
      if (resp) {
        this.fundInfo = resp.data;
        this.proName=resp.data.name;
        this.sector=resp.data.sector;
        this.common.trigger(this.common.headerEvent,[{
          name : 'Project ID' ,
          value : this.fundId
        },{
          name  :'Project Name' ,
          value : resp.data.name
        },{
          name : 'Sector' ,
          value : resp.data.sector
        }])
      }
    })
   
    this.getBudget();
   
  } 
  access() {
    this.role = this.common.screenRole("fund_release");
  }
  getBudget(){
    this.budget.getBudget(this.fundId).subscribe((resp)=>{
      if(resp){
        
        this.budgetData= resp; 
        this.budgetId = resp.data?.id;
        if(!this.budgetId){
          this.route.navigateByUrl('/dashboard/fund/list');
          this.common.showError("This project does'nt contain budget data!");
        }
        else{
          this.getAllCental();
        }
        
        }
        
    })
  }
  getAllCental(){
    this.spinner.show();
    this.fund.getCentralList(this.budgetId).subscribe((resp)=>{
      this.spinner.hide();
      
      if(resp){
        this.datas= resp.data;
        this.id=resp.data.id;
       }
    })
  }
  addNew(){
    this.route.navigateByUrl('/dashboard/fund/add-central', { state: { id:this.id } });
  }
  edit(id: any) {
    this.route.navigateByUrl('/dashboard/fund/edit-add-central', { state: { id: id, flag: "edit" } });
  }
  view(id: any) {
    this.route.navigateByUrl('/dashboard/fund/edit-add-central', { state: { id: id,  flag: "view" } });
  }
  delete(id: any) {
    if (confirm('Are you sure you want to delete this record?')) {
      this.fund.getDelete(id).subscribe(resp => {
        

        if (resp) {
          this.common.showSuccess("Successfully Deleted");
          this.getAllCental();
        }
      })
    }

  }

}
