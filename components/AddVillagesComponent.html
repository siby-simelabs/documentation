import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-add-villages',
  templateUrl: './add-villages.component.html',
  styleUrls: ['./add-villages.component.scss']
})
export class AddVillagesComponent implements OnInit {
  crpId: any;
  crpInfo: any;
  proName: any;
  village: any;
  selectedCheck: any = [];
  blkId: number | undefined;
  overhide: boolean = true;
  villageList: any;
  flag: string | undefined;
  blk: string | undefined;
  dist: string | undefined;
  mainVillageList: any;
  searchData: any;
  villageNew: any;
  gramList: any;
  searchString: any;
  villageview: boolean = false;
  editArray: any = [];
  selectAllCheck: boolean = false;
  submitted: boolean = false;
  villages: any;
  gram: any;
  villageData: any;
  showFilter: boolean = false;
  totalVillagecount: number = 0;
  totalCountSelectedCheck: number = 0;
  role:any;

  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public spinner: NgxSpinnerService
  ) { }
  filterForm = this.formBuilder.group(
    {
      gramapanchayat: [null],
      villageName: [null]
    },
  );
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }

  ngOnInit(): void {
    this.access()
    this.crpId = this.common.getCrpId();
    this.blkId = history.state.blockid;
    this.villageData = history.state.data;
    if (this.blkId == undefined || null) {
      this.route.navigate(['/dashboard/crp/assignvillages']);
      this.common.showError("Select a Village")
    }
    this.villageList = history.state.villages;
    this.flag = history.state.flag;
    if (this.flag == "view") {
      this.villageview = true;
    }
    this.blk = history.state.blk;
    this.dist = history.state.dist;
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    });
    this.listAllVill();
    this.getVillList();
  }
  access() {
    this.role = this.common.screenRole("assignvillage");
  }
  listAllVill() {
    this.manager.listAllVillage(this.blkId).subscribe((resp) => {
      if (resp) {
        this.village = resp.data;

        this.villageNew = resp.data;
        if (this.villageList != undefined || null) {
          this.village = this.villageList;
          for (let i = 0; i < this.villageList.length; i++) {
            if (this.villageList[i].selected == true) {
              this.selectedCheck.push(this.villageList[i]);
            }
          }
          this.totalCountSelectedCheck = this.selectedCheck?.length;
          if (this.selectedCheck?.length == this.village?.length) {
            this.selectAllCheck = true;
          }
        }
        this.common.trigger('updateHeaderDetails', [{
          name: 'Project ID',
          value: this.crpInfo?.id
        }, {
          name: "Project Name",
          value: this.crpInfo?.name
        }, {
          name: 'State',
          value: this.villageData?.stateName
        }, {
          name: 'District',
          value: this.villageData?.districtName
        }, {
          name: 'Block',
          value: this.villageData?.blockName
        }])
      }
    })
  }
  checkValue(event: any, code: any) {
    let indexVal;
    if (event.currentTarget.checked == true) {
      for (let i = 0; i < this.village.length; i++) {
        if (this.village[i].locationCode == code) {
          indexVal = this.village[i];
          indexVal.selected = true;
          this.village[i].selected = true;
          indexVal = {
            "villageName": indexVal.villageName,
            "locationCode": indexVal.locationCode,
            "selected": true
          }
          this.selectedCheck.push(indexVal);
        }
      }

    }
    else {
      for (let i = 0; i < this.village.length; i++) {
        if (this.village[i].locationCode == code) {
          this.village[i].selected = false;
        }
      }

      this.selectedCheck = this.selectedCheck.filter((item: { locationCode: any; }) => item.locationCode !== code);

    }
  }
  removeCheckd(event: any, code: any) {
    let indexVal;
    if (event.currentTarget.checked == false) {
      this.selectedCheck = this.selectedCheck.filter((item: { locationCode: any; }) => item.locationCode !== code);
      let obj = this.village.find((village: { locationCode: number; }) => village.locationCode == code);
      if (obj) {
        obj.selected = false;
      }

    }
  }

  saveVillage() {
    let data: any = {};
    if ((this.selectedCheck.length > 0)) {
      let endpoints = "updateVillage";
      if (this.flag == "edit") {
        this.totalVillagecount = this.selectedCheck.length - this.totalCountSelectedCheck;
        data = {
          "villageListModelList": this.village,
          "projectId": this.crpId,
          "personId": 122390
        }
      }
      else {
        this.totalVillagecount = this.selectedCheck.length;
        data = {
          "villageListModelList": this.selectedCheck,
          "projectId": this.crpId,
          "personId": 122390
        }
      }



      let alcount: number = this.totalVillagecount + parseInt(this.villageData.noOfVillagesAlreadyIdentified);
      console.log(alcount)
      if (alcount > this.villageData.noOfVillage) {
        this.common.showError("No. of villages already identified cannot exceed the target no of villages");
      }
      else {
        this.saveVillageData(data, endpoints);
      }
    }
    else {
      this.common.showError('Select any village from the list');
    }
  }

  saveVillageData(data: any, endpoints: any) {
    this.manager.saveVillages(data, endpoints).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        this.common.showSuccess('Successfully added villages');
        this.backTo();
      }
      else {
        this.common.showError(resp.message);
      }
    })
  }

  selectAll(event: any) {
    let indexVal;
    if (event.currentTarget.checked == true) {
      this.selectedCheck = [];
      for (let i = 0; i < this.village.length; i++) {
        this.village[i].selected = true;
        this.selectedCheck.push(this.village[i]);
      }

    }
    else {
      for (let i = 0; i < this.village.length; i++) {
        this.village[i].selected = false;
      }

      this.selectedCheck = [];
    }
  }
  backTo() {
    history.go(-1)
  }
  getVillList() {
    this.spinner.show();
    this.manager.getAllVillById(this.crpId, this.blkId).subscribe(resp => {
      this.spinner.hide();
      if (resp) {
        this.mainVillageList = resp.data?.villageModels;
      }
    })
  }
  onSubmit(value: any): void {
    this.submitted = true;
    if (this.filterForm.invalid) {
      return;
    }
    else {
      this.showFilter = false;
      let gramValue = this.mainVillageList.filter((item: { gramapanchayat: any; }) => item.gramapanchayat == this.gram);

      let gramArr = gramValue[0]?.villageListModelList;
      this.village = gramArr;
      const villagename = value["villageName"].toUpperCase()
      if (villagename) {
        this.searchData = this.find(gramArr, villagename);
        this.village = this.searchData;
      }
      console.log(this.village)
    }
  }

  gramChange(event: any) {
    console.log(event);
    let gramValue = this.mainVillageList.filter((item: { gramapanchayat: any; }) => item.gramapanchayat == this.gram);

    let gramArr = gramValue[0]?.villageListModelList;
    this.village = gramArr;
  }

  find(items: any, text: any) {
    text = text.split(' ');
    return items.filter((item: { villageName: string | any[]; }) => {
      return text.every((el: string) => {
        return item.villageName?.includes(el);
      });
    });
  }
  clearFilter() {
    this.getVillList();
    this.village = this.villageNew;
    this.filterForm.reset();
    this.gramList = "";
    this.searchString = ""
    this.showFilter = false;
  }
}
