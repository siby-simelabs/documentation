import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { FundService } from 'src/app/providers/fund/fund.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
@Component({
  selector: 'app-central-add',
  templateUrl: './central-add.component.html',
  styleUrls: ['./central-add.component.scss']
})
export class CentralAddComponent implements OnInit {
  submitted: boolean = false;
  formCreateFund: FormGroup;
  installment: any;
  tranche: any;
  amount: any;
  releaseDate: any;
  fileList: any;
  releaseFromAap: any;
  comments: any;
  budgetId: any;
  fileData: any;
  minDate = new Date();
  fundview: any;
  id: any;
  central: any;
  fundId: any;
  budgetData: any;
  total: any;
  deadMinDate: any;
  deadMaxDate: any;
  projectInfo: any;
  trancheData: any;
  isView : any  ;
  placement : TypeDatePickerPlacement= 'bottom';
  role:any;
  constructor(
    public fund: FundService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public budget: BudgetService,
    public project: ProjectService,
    public dp :DatePickerPlacement
  ) {
    this.formCreateFund = this.formBuilder.group({
      installment: ['', Validators.required],
      tranche: ['', Validators.required],
      releaseDate: ['', Validators.required],
      amount: ['', Validators.required],
      fileList: [],
      comments: []
    })
  }

  ngOnInit(): void {
    this.access();
    this.fundId = this.common.getFundId();
    this.getInstallment();
    this.getTranche();
    this.id = history.state.id;
    this.isView = history.state.flag === 'view' ;
    if (history.state.flag == "view") {
      this.fundview = true;
    }
    if (this.id) {
      this.spinner.show();
      this.fund.getCentralById(this.id).subscribe((resp) => {
        this.spinner.hide();
        
        if (resp) {
          this.central = resp.data;
          this.formCreateFund.get("installment")?.setValue(this.central.installment);
          this.formCreateFund.get("tranche")?.setValue(this.central.tranche);
          this.formCreateFund.get("amount")?.setValue(this.central.amount);
          this.formCreateFund.get("releaseDate")?.setValue(this.central.releaseDate);
          this.formCreateFund.get("comments")?.setValue(this.central.comments);
        }
      })
    }
    else {
      this.id = history.state.id;
    }
    this.getBudget();
    this.getTotalShare();
    this.project.getProjectByID(this.fundId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.deadMinDate = this.dateForMindate(this.projectInfo.actualStartDate);
        this.deadMaxDate = this.dateForMindate(this.projectInfo.actualEndDate);
        this.common.trigger( this.common.headerEvent , [{
          name : "Project ID" ,
          value : resp.data.id
        },{
          name : 'Project Name' ,
          value : resp.data.name 
        },{
          name: 'Sector' ,
          value : resp.data.sector
        }])
      }
    })
  }
  dateForMindate(date: any) {
    if(!date || typeof date !== "string")  return ;
    let datearray = date.split("/");
    let newdate = datearray[1] + '/' + datearray[0] + '/' + datearray[2];
    
    return new Date(newdate);
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateFund.controls;
  }
  getTotalShare() {
    this.fund.getTotal(this.fundId).subscribe(resp => {
      if (resp) {
        this.total = resp.data?.fundReleaseTotalModel;
        let total = this.total ;
        this.common.trigger( 'cards' , [{
          name : 'Centre share' ,
          value : total?.budgetCentralAmount,
        },{
          name : 'State share' ,
          value :total?.budgetStateAmount ,
        },{
          name : 'Total budget' ,
          value : total?.budgetTotalAmount,
        },{
          name : 'Other. contri.' ,
          value : total?.budgetOtherAmount,
        },{
          name : 'Grand total' ,
          value : total?.grandTotalAmount,
        },{
          name : 'State release tot.' ,
          value : total?.stateFundTotal,
        },{
          name : 'Centre release tot.' ,
          value : total?.centralFundTotal,
        },{
          name : '% Utilized as on date' ,
          value : this.splitAmount(total?.utilizedAmount),
        }])
      }
    })
  }
  getBudget() {
    this.budget.getBudget(this.fundId).subscribe((resp) => {
      if (resp) {
        
        this.budgetData = resp;
        this.budgetId = resp.data.id;
      }
      
    })
  }
  onSubmit(value: any) {
    this.submitted = true;
    if (this.formCreateFund.invalid) {
      return;
    }
    else {
      let centralFund = {
        "installment": value["installment"],
        "releaseDate": (<HTMLInputElement>document.getElementById("releaseDate")).value,
        "budgetId": this.budgetId,
        "tranche": value["tranche"],
        "amount": value["amount"],
        "budgetType": "1",
        "comments": value["comments"],
        "id": this.id

      };
      this.saveFund(centralFund);
      
    }
  }
  saveFund(data: any) {
    let formData: any = new FormData();
    if (this.fileData) {
      formData.append("sanctionLetter", this.fileData, this.fileData.name);
    }
    else {
      formData.append("sanctionLetter", '');
    }
    formData.append("centralFund", new Blob([JSON.stringify(data, null, 2)], {
      type: "application/json"
    }));
    this.spinner.show();
    this.fund.addCentral(formData).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        
        this.common.showSuccess('Successfully Added Central Share');
        this.route.navigateByUrl('/dashboard/fund/central', { state: { id: this.id } });
      }
      else {
        this.common.showError(resp.message);
      }
    })
  }
  onReset(): void {
    this.submitted = false;
    this.formCreateFund.reset();
  }
  onChange(event: any) {
    if (event.target.files[0].size >= this.common.maxFileSize) {
      this.common.showError("File size too large. Please upload a file less than 50mb");
      this.formCreateFund.get("fileList")?.setValue(null);
      return
    }
    else if (event.target.files[0].name.length >= this.common.maxFileNameLength) {
      this.common.showError("File name length restricted to maximum " + this.common.maxFileNameLength + " characters");
      this.formCreateFund.get("fileList")?.setValue(null);
      return
    }
    else {
      this.fileData = event.target.files[0];
    }
  }
  getInstallment(){
    this.common.getInstallmentData().subscribe((resp) => {
      if (resp) {
         
         this.installment = resp.data;
      }
   });
  }
  getTranche(){
    this.common.getTrancheData().subscribe((resp) => {
      if (resp) {
         
         this.trancheData = resp.data;
      }
   });
  }
  splitAmount(amt: number) {
    if (amt != 0) {
      return amt.toFixed(2);
    }
    else {
      return 0
    }
  }
  download(url: string, docname:string) {
    this.common.dwnldFile(url, docname);
  }
  access() {
    this.role = this.common.screenRole("fund_release");
  }
}
