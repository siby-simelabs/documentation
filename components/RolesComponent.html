import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { AdminService } from '../admin.service';
import { CommonService } from 'src/app/providers/common.service';
import { NgxSpinnerService } from 'ngx-spinner';
@Component({
  selector: 'app-roles',
  templateUrl: './roles.component.html',
  styleUrls: ['./roles.component.scss'],
})
export class RolesComponent implements OnInit {
  selectedRole: any | null = null;
  permissons: any = [];
  step: number = 1;
  loading: boolean = true;
  hasChanges: boolean = false;
  isUpdating: boolean = false;
  initialSelection: string = '';
  constructor(
    private admin: AdminService,
    private common: CommonService,
    private spinner: NgxSpinnerService,
    private route: ActivatedRoute
  ) {
    this.route.paramMap.subscribe((params) => {
      this.initialSelection = params.get('name') || '';
    });
  }
  ngOnInit(): void {
    this.admin.getRoles().subscribe((res) => {
      this.permissons = res.data;
      this.loading = false;
      if (this.initialSelection && 'all' !== this.initialSelection) {
        this.editRole({}, this.initialSelection);
      }
    });
    this.common.subscribe(this.onReset, 'reset_role_page');
  }
  onReset = () => {
    this.reset();
  };
  get isAdmin() {
    return this.selectedRole.role == 'pm_admin';
  }
  reset = () => {
    this.step = 1;
    this.selectedRole = null;
    this.hasChanges = false;
    window.history.replaceState({}, '', `/dashboard/admin/roles/all`);
  };
  saveChanges($event: any) {
    this.spinner.show();
    this.selectedRole &&
      (this.isUpdating = true) &&
      this.admin.updateRole(this.selectedRole).subscribe((res) => {
        this.isUpdating = false;
        this.reset();
        this.common.showSuccess('Permissons updated');
        this.spinner.hide();
        this.common.fetchUserRole(() => {});
      });
  }
  getBreadcumbs() {
    let item: any = [
      {
        name: 'Roles',
        link: '/dashboard/admin/roles/all/',
        onClick: this.reset,
      },
    ];
    if (this.step === 2 && this.selectedRole) {
      item.push({
        name: `Edit ${this.selectedRole.role}`,
        link: `/dashboard/admin/roles/${this.selectedRole.role}/`,
      });
    }
    return item;
  }
  editRole($event: any, name: string) {
    this.step = 2;
    this.permissons.map((item: any) => {
      if (name === item.role) {
        this.selectedRole = item;
      }
    });

    window.history.replaceState(
      {},
      '',
      `/dashboard/admin/roles/${this.selectedRole.role}`
    );
  }
  getScreens(): ScreenItem[] {
    return this.selectedRole &&
      this.selectedRole.rolePrivilege &&
      this.selectedRole.rolePrivilege.screen
      ? this.selectedRole.rolePrivilege.screen
      : [];
  }
  onRoleChangeHandler($event: any) {
    let value = $event.target.value;
    this.permissons.map((item: any) => {
      if (item.name === value) {
        this.selectedRole = item;
      }
    });
  }
  onChangeHanlder($event: any) {
    let [screen, role] = $event.target.name.split('|');
    let isChecked = $event.target.checked;
    let screens = this.getScreens().map((item, index) => {
      if (item.name === screen) {
        const ROLES = {
          ...item.roles[0],
          [role]: isChecked,
        };
        let auth: boolean = Object.values(ROLES).indexOf(true) > -1;
        return {
          ...item,
          auth,
          roles: [ROLES],
        };
      } else {
        return item;
      }
    });
    if (!this.hasChanges) {
      this.hasChanges = true;
    }
    if (this.selectedRole) {
      this.selectedRole.rolePrivilege.screen = screens;
    }
  }
}
