import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { CommonService } from 'src/app/providers/common.service';
import { CrpService } from 'src/app/providers/project/crp/crp.service';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { InterventionService } from 'src/app/providers/project/interventions/intervention.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { UserService } from 'src/app/providers/user.service';
import { Validator } from 'src/app/_helpers/validators';
@Component({
  selector: 'app-shg',
  templateUrl: './shg.component.html',
  styleUrls: ['./shg.component.scss']
})
export class ShgComponent implements OnInit {
  formCreateShg: FormGroup;
  totalNumber: any;
  submitted = false;
  setYearWise = true;
  projectId: any;
  yearFiveTarget: any;
  yearFourTarget: any;
  yearOneTarget: any;
  yearSixTarget: any;
  yearThreeTarget: any;
  yearTwoTarget: any;
  projectInfo: any;
  vewId: any;
  isChecked = false;
  totalNumberCount: any;
  role: any;
  shgData: any;
  constructor(
    public common: CommonService,
    public formBuilder: FormBuilder,
    public validator: Validator,
    public route: Router,
    public project: ProjectService,
    public inter: InterventionService,
    public geo: GeoService,
    public crp: CrpService,
    private activatedRoute: ActivatedRoute,
    public user: UserService
  ) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId()
    this.formCreateShg = this.formBuilder.group({
      totalNumber: ['', Validators.required],
      yearFiveTarget: [],
      yearFourTarget: [],
      yearOneTarget: [],
      yearSixTarget: [],
      yearThreeTarget: [],
      yearTwoTarget: [],
      setYearWise: [],

    }
    );
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateShg.controls;
  }

  ngOnInit(): void {
    this.access();
    this.formCreateShg.controls['setYearWise'].setValue(false);
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);

      }
    });
    this.getShg();
  }

  access() {
    this.role = this.common.screenRole("beneficiary_coverage");
    
    if ((this.user.profile.roleName == "pm_central_user_level1_final_approval")) {
      this.vewId = this.projectId;
    }
  }

  onSubmit() {
    this.submitted = true;
    this.formCreateShg.addControl('projectId', new FormControl('', Validators.required));
    this.formCreateShg.controls['projectId'].setValue(this.projectId);
    if (this.formCreateShg.invalid) {
      return;
    }
    if (this.totalNumberCount > this.totalNumber) {
      alert("Sum of the year wise target should not exceed the total number of SHG Members to be covered!")
      return;
    }
    
    this.SaveShg(
      JSON.stringify(this.formCreateShg.value, null, 2)
    );
  }
  SaveShg(data: any) {
    this.crp.saveShg(data).subscribe((resp) => {
      if (resp) {
        
        if (this.shgData != null) {
          this.common.showSuccess('Successfully Updated SHG Members to be Covered ');
          this.route.navigate(['/dashboard/project/coverage/shg']);
        }
        else{
          this.common.showSuccess('Successfully Created SHG Members to be Covered ');
          this.route.navigate(['/dashboard/project/coverage/shg']);
        }
      }
    });
  }
  getShg() {
    if (this.projectId) {
      let data = {
        "projectId": this.projectId,
        "type": "shg",
      }
      this.crp.getCrp(data).subscribe((resp) => {
        
        this.shgData = resp.data;
        this.totalNumber = resp.data?.totalNumber;
        this.yearOneTarget = resp.data?.yearOneTarget;
        this.yearTwoTarget = resp.data?.yearTwoTarget;
        this.yearThreeTarget = resp.data?.yearThreeTarget;
        this.yearFourTarget = resp.data?.yearFourTarget;
        this.yearFiveTarget = resp.data?.yearFiveTarget;
        this.yearSixTarget = resp.data?.yearSixTarget;
        if (resp.data.setYearWise == true) {
          this.isChecked = true
          this.formCreateShg.controls['setYearWise'].setValue(true);
        }
        else {
          this.formCreateShg.controls['setYearWise'].setValue(false);
          this.isChecked = false
        }
      })
    }
  }
  changeStatus(event: any) {
    
    if (event.srcElement.checked == true) {
      this.isChecked = true;
      this.formCreateShg.controls['setYearWise'].setValue(true);
    }
    else {
      this.formCreateShg.controls['setYearWise'].setValue(false);
      this.isChecked = false;
      this.yearOneTarget = "";
      this.yearTwoTarget = "";
      this.yearThreeTarget = "";
      this.yearFourTarget = "";
      this.yearFiveTarget = "";
      this.yearSixTarget = "";
    }
  }
  onTotalChange(event: any) {
    
    this.totalNumberCount = (this.formCreateShg.controls['yearOneTarget'].value || 0) + (this.formCreateShg.controls['yearTwoTarget'].value || 0) + (this.formCreateShg.controls['yearThreeTarget'].value || 0) + (this.formCreateShg.controls['yearFourTarget'].value || 0) + (this.formCreateShg.controls['yearFiveTarget'].value || 0) + (this.formCreateShg.controls['yearSixTarget'].value || 0);
    if (this.totalNumberCount > this.totalNumber) {
      alert("Sum of the year wise target should not exceed the total number of SHG Members to be covered!")
      return
    }
  }

}
