import { Component } from '@angular/core';
import { UserService } from './providers/user.service';
import { TranslateService } from '@ngx-translate/core';
import { getDefaultLanguage } from './_helpers/language';
@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent {
  title = 'techxers';
  constructor(public user:UserService , public translates : TranslateService ){
    translates.setDefaultLang(getDefaultLanguage());
  }
  toggleLang(){
    let name = prompt("") ;
    if( name ){
      this.translates.setDefaultLang( name );
    }
  }
  ngOnInit(){
    this.user.loadUser();
    
  }
}
