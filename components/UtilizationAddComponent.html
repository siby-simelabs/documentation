import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { UtilizationService } from 'src/app/providers/utilization/utilization.service';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
@Component({
  selector: 'app-utilization-add',
  templateUrl: './utilization-add.component.html',
  styleUrls: ['./utilization-add.component.scss'],
})
export class UtilizationAddComponent implements OnInit {
  submitted: boolean = false;
  formCreateUtil: FormGroup;
  amount: any;
  utilizedDt: any;
  docUrl: any;
  parms: any;
  utilview: any;
  util: any;
  fileData: any = '';
  minDate = new Date();
  expenseHead: any;
  amountUtilized: any;
  total: any;
  expenseProvisionId: any;
  expenseIdAdd: any;
  flag: any;
  expenseId: any;
  projectInfo: any;
  deadMinDate: any;
  deadMaxDate: any;
  projectId: any | null;
  path!: string;
  placement : TypeDatePickerPlacement= 'bottom';
  role:any;

  constructor(
    public utilization: UtilizationService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public project: ProjectService ,
    public dp :DatePickerPlacement
  ) {
    this.formCreateUtil = this.formBuilder.group({
      amount: ['', Validators.required],
      utilizedDt: ['', Validators.required],
      docUrl: [''],
    });
  }

  ngOnInit(): void {
    this.access();
    this.path = window.location.pathname;
    this.spinner.hide();
    this.projectId = this.common.getUtilId();
    this.expenseId = history.state.expenseId;
    this.parms = history.state.id;
    this.flag = (this.parms) ?  "edit" : "add";
    this.expenseHead = history.state.head;
    this.amount = history.state.total;
    this.amountUtilized = history.state.amountUtilized;

    this.common.trigger('cards', [
      {
        name: 'ID',
        value: this.expenseId,
      },
      {
        name: 'Head',
        value: this.expenseHead,
      },
      {
        name: 'Total',
        value: this.amount,
      },
      {
        name: 'Utilized',
        value: this.amountUtilized,
      },
    ]);
    if (history.state.flag == 'view') {
      this.utilview = true;
      this.flag = 'view';
    }
    if (history.state.flag == 'viewonce') {
      this.utilview = true;
    }
    if (!history.state.flag) {
      this.common.showError('Select project!');
      this.route.navigateByUrl('/dashboard/utilization/list');
    }

    if (this.parms) {
      this.spinner.show();
      this.utilization.getFundById(this.parms).subscribe((resp) => {
        this.spinner.hide();

        if (resp) {
          this.util = resp.data?.fundUtilizationModel;
          this.formCreateUtil.get('amount')?.setValue(this.util.amount);
          this.formCreateUtil.get('utilizedDt')?.setValue(this.util.utilizedDt);
        }
      });
    }
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          {
            name: 'Project ID',
            value: this.projectInfo.id,
          },
          {
            name: 'Project Name',
            value: this.projectInfo.name,
          }
        ]);
        this.deadMinDate = this.dateForMindate(
          this.projectInfo.actualStartDate
        );
        this.deadMaxDate = this.dateForMindate(this.projectInfo.actualEndDate);
      }
    });
  }


  dateForMindate(date: any) {
    if (!date || typeof date !== 'string') return '';
    let datearray = date.split('/');
    let newdate = datearray[1] + '/' + datearray[0] + '/' + datearray[2];
    return new Date(newdate);
  }

  get f(): { [key: string]: AbstractControl } {
    return this.formCreateUtil.controls;
  }

  onSubmit(value: any) {
    this.submitted = true;
    if (this.formCreateUtil.invalid) {
      return;
    } else {
      let fundUtilization = {
        amount: value['amount'],
        utilizedDt: (<HTMLInputElement>document.getElementById('utilizedDt'))
          .value,
        expenseProvisionId: this.expenseId,
        id: this.parms,
      };
      this.saveUtil(fundUtilization);
    }
  }

  saveUtil(data: any) {
    let formData: any = new FormData();
    if (this.fileData) {
      formData.append('fundDocument', this.fileData, this.fileData.name);
    } else {
      formData.append('fundDocument', '');
    }
    formData.append(
      'fundUtilization',
      new Blob([JSON.stringify(data, null, 2)], {
        type: 'application/json',
      })
    );
    this.spinner.show();
    this.utilization.saveUtilization(formData).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == 'SUCCESS') {
        this.common.showSuccess('Successfully Added Utilization');

        this.route.navigateByUrl('/dashboard/utilization/add-component', {
          state: {
            expenseHead: this.expenseHead,
            amount: this.amount,
            amountUtilized: this.amountUtilized,
            expenseProvisionId: this.expenseId,
            flag: 'edit',
          },
        });
      } else {
        this.common.showError('Utilized amount is exceeded than total amount');
      }
    });
  }

  onReset(): void {
    this.submitted = false;
    this.formCreateUtil.reset();
  }

  onChange(event: any) {
    if (event.target.files[0].size >= this.common.maxFileSize) {
      this.common.showError(
        'File size too large. Please upload a file less than 50mb'
      );
      this.formCreateUtil.get('docUrl')?.setValue(null);
      return;
    } else if (
      event.target.files[0].name.length >= this.common.maxFileNameLength
    ) {
      this.common.showError(
        'File name length restricted to maximum ' +
        this.common.maxFileNameLength +
        ' characters'
      );
      this.formCreateUtil.get('docUrl')?.setValue(null);
      return;
    } else {
      let file = event.target.files[0];
      this.fileData = event.target.files[0];
    }
  }

  backtoAdd() {
    window.history.go(-1);
  }
  download(url: string, docname: string) {
    this.common.dwnldFile(url, docname);
  }
  access() {
    this.role = this.common.screenRole("utilization");
  }
}
