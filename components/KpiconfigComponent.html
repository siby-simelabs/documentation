import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { OutcomeService } from 'src/app/providers/admin/kpiconfig/outcome/outcome.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-kpiconfig',
  templateUrl: './kpiconfig.component.html',
  styleUrls: ['./kpiconfig.component.scss']
})
export class KpiconfigComponent implements OnInit {
  
  pageNo: number = 1;
  pageSize: number = 10;
  datas: any = [];
  projectId: any;
  projectInfo: any;
  currentPage = 1;
  prCount!: number;
  searchData:any;
  data: any;
  master = [];
  submitted: boolean = false;
  cato: any;
  interventionCategory: any;
  role:any; 
  issearch:boolean=false ;
  load:boolean=false;
  constructor(
    public outcome: OutcomeService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public project: ProjectService
  ) {
  }
  filterForm = this.formBuilder.group({
    sector:[null] ,
    interventionCategory:[null],
  },);

  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }

  ngOnInit(): void {
    this.access();
    this.common.clearOutcomeId();
    this.common.clearOutcomeViewId();
    this.getKpiAll();
    this.getMaster()
    this.common.clearProjectId()
    this.common.clearProjectViewId()
  }
 
  access() {
    this.role = this.common.screenRole("kpiconfig");
  }
  onSubmit(value:any): void {
    this.issearch = true;
    this.project.trigger("onFilterApply");
    this.submitted = true;
    this.pageNo = 1;
    this.pageSize = 10;
    this.currentPage = 1;
    if (this.filterForm.invalid) {
      return;
    }
    this.searchData = {
      "searchModel": {
        "status": true,
        "interventionCategory": this.filterForm.get("interventionCategory")?.value,
        "sector" : this.filterForm.get("sector")?.value
      },
      "pageNumber": this.pageNo,
      "pageSize": this.pageSize

    }
    this.filterdata(this.searchData);
  }
  filterdata(data: any) {
    this.load=false;
    this.spinner.show();
    this.datas = [];
    this.outcome.filterKpiOutcome(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.load=true;
        this.datas = resp.data?.kpiOutcomeList;
        this.prCount = resp.data?.totalCount;
      }
    });
  }

  getKpiAll() {
    this.load=false;
    this.spinner.show();
    this.datas = []; 
    
    this.outcome.getKpi(this.pageNo, this.pageSize, this.projectId).subscribe(resp => {
      this.spinner.hide();
      if (resp) {
        this.load=true;
        this.datas = resp.data?.kpiOutcomeList;
        this.prCount = resp.data?.totalCount;

       }
    },
      (error) => {
        this.load=true;
        this.common.showError(error.statusText);
        this.spinner.hide();
      });
  }
  edit(id: any) {
    this.route.navigateByUrl('/dashboard/admin/edit-outcome', { state: { id: id } });
    this.common.storeOutcome(id);
  }
  view(id: any) {
    this.route.navigateByUrl('/dashboard/admin/edit-outcome', { state: { id: id, flag: "view" } });
    this.common.storeOutcomeView(id);
    this.common.storeOutcome(id)
  }
  delete(id: any) {
    this.spinner.show();
    if (confirm('Are you sure you want to delete this record?')) {
      this.outcome.delete(id).subscribe(resp => {
        
        this.spinner.hide();
        if (resp) {
          this.common.showSuccess("Successfully Deleted");
          this.getKpiAll();
        }
      })
    }
    else {
      this.spinner.hide();
    }
  }
  getMaster() {
    this.common.getMasterData().subscribe((resp) => {
      if (resp) {
        
        this.data = resp.data;
      }
    });
  }
  sectorChange(event: any) {
    this.filterForm.get("interventionCategory")?.setValue("");
    this.cato = [];
    
    for (let i: any = 0; i < this.data.sector.length; i++) {
      if (event.target.value == this.data.sector[i]?.name) {
        this.cato = this.data.sector[i].cato;
      }
    }
  }


  outcomeStatus(status: any) {
    if (status == true) {
      return "Active";
    }
    else {
      return "Inactive";
    }
  }
  clearFilter() {
    this.datas = []; 
    this.cato = [];
    this.getKpiAll();
    this.issearch = false;
    this.searchData = [];
    this.currentPage = 1;
    this.filterForm.reset();
    this.project.trigger("onFilterApply");
  }
  pageChanged(event: any) {
    if (event.page != this.pageNo) {
      this.pageNo = event.page;
      this.pageSize = event.itemsPerPage;
      
      if (this.issearch==true) {
        this.searchData.pageNumber = event.page;
        this.searchData.pageSize = event.itemsPerPage;
        this.searchData.pageNumber = event.page;
        this.filterdata(this.searchData);

      }
      else {
        this.getKpiAll();
      }
    }

  }
}
