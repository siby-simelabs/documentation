import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { UserService } from 'src/app/providers/user.service';
import { Validator } from 'src/app/_helpers/validators';
import { AuthService } from 'src/app/auth.service';
@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent implements OnInit {
  formLogin: FormGroup;
  loginId: any;
  password: any;
  submitted = false;
  data: any;
  username:any;

  constructor(
    public auth: AuthService,
    public common: CommonService,
    public formBuilder: FormBuilder,
    public validator: Validator,
    public route: Router,
    public project: ProjectService,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService,
    public user: UserService
  ) {
    this.formLogin = this.formBuilder.group({
      username: ['', Validators.required],
      password: ['', Validators.required],
    })
  }

  ngOnInit(): void {
    localStorage.clear()
  }


  get f(): { [key: string]: AbstractControl } {
    return this.formLogin.controls;
  }
  onSubmit(value: any) {
    this.submitted = true;
    if (this.formLogin.invalid) {
      return;
    }
    else {
      this.saveLogin(
        JSON.stringify(this.formLogin.value, null, 2)
      )
    }
  }
  saveLogin(data: any) {
    this.spinner.show();
    this.auth.login(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp.data.statusMessage == "Login Success") {
        this.user.updateLoginData(resp);
        this.common.showSuccess('You have successfully logged in');
        this.common.updatePermisson(resp.data.role || {});
        setTimeout(() => {
          if(resp.data.role.name == "pm_manager_district_level" || resp.data.role.name == "pm_manager_block_level"){
            this.route.navigate(['/dashboard/crp/list']);
          }
          else{
            this.route.navigate(['/dashboard/project/list']);
          }          
        }, 600);
        

      }
      else {
        this.common.showError('Invalid credentials');
      }
    },
      (error: any) => {
        console.error(error);
        this.spinner.hide();
        this.common.showError("Network error! Please try again later.");
      });
  }
  trim(event:any){
    this.username = event.target.value.trim()
  }


}
