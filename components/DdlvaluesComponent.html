import { THIS_EXPR } from '@angular/compiler/src/output/output_ast';
import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { CommonService } from 'src/app/providers/common.service';

@Component({
  selector: 'app-ddlvalues',
  templateUrl: './ddlvalues.component.html',
  styleUrls: ['./ddlvalues.component.scss']
})
export class DdlvaluesComponent {
  repdata: any;
  parms: any;
  flag: any;
  lastUrlSegment: any;
  childData: any;
  endpoint: any;
  path!: string;
  child: boolean = false;
  segment: any;
  view:boolean = false;
  role:any;

  constructor(public common: CommonService, public route: Router, public aRoute: ActivatedRoute) {
    aRoute.params.subscribe(val => {
      this.parms = history.state.data;
      this.view = val['view'];
      if(!this.parms){
        this.route.navigateByUrl('/dashboard/admin/master-values', { state: { flag: this.endpoint,  edit:"back" } });
      }
      this.flag = history.state.flag;
      this.childData = history.state.childData;
      this.lastUrlSegment = this.route.url.split('?')[0].split('/').pop();
      
      this.loadData()
    });
  }
  access() {
    this.role = this.common.screenRole("adminvalues");
  }

  
  loadData() {
    this.access();
    this.path = window.location.pathname;
    if (window.location.href.indexOf("view=true") > -1) {
      this.view = true;
    }
    else{
      this.view = false;
    }
    if (window.location.href.indexOf("child") > -1) {
      this.getddlbyid();
      this.segment = "ddl-child";
    }
    else {
      this.common.saveFlag(this.lastUrlSegment);
      this.getmasterDDls();
      this.segment = "master-child";
    }
  }

  add(data: any) {
    this.child = true;
     this.route.navigateByUrl('/dashboard/admin/child/' + data.id + '?view='+this.view+'', { state: { data: data, flag: this.endpoint,  edit:"add" } });
  }
  addNew() {
   
    this.route.navigateByUrl('/dashboard/admin/'+this.segment+'/' + this.lastUrlSegment + '', { state: { flag: this.flag,  masters: this.parms,  edit:"add"}, skipLocationChange: false });
  }
  edit(data: any) {

    this.route.navigateByUrl('/dashboard/admin/'+this.segment+'/' + this.lastUrlSegment + '', { state: { data: data, flag: this.endpoint,  masters: this.parms, edit:"edit" }, skipLocationChange: false });
  }
  
  delete(data: any) {
    if (confirm("Are you sure you want to delete this record?") == true) {
      let deleteId = {"id":data.id}
      this.common.deleteMaster(deleteId, "deleteMasterDdl").subscribe((resp) => {
        if (resp) {
          this.loadData();
        }
      });
    }
  }
  back() {
    history.go(-1)
  }
  getddlbyid() {
    this.common.getChildDdlById(this.lastUrlSegment).subscribe((resp) => {
      if (resp) {
        
        this.repdata = resp.data;
      }
    });
  }
  getmasterDDls() {
    this.common.getMasterDdlById(this.lastUrlSegment).subscribe((resp) => {
      if (resp) {
        
        this.repdata = resp.data;
      }
    });
  }
  getStatus(status:boolean){    
    if(status == true){
      return "Active";
    }
    else{
      return "Inactive";
    }
  }
}
