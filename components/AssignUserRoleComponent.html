import { Component, OnInit } from '@angular/core';
import { AdminService } from '../../pages/admin/admin.service';
import { FormBuilder, Validators, AbstractControl } from '@angular/forms';
import { CommonService } from 'src/app/providers/common.service';
import { EventEmitter } from '@angular/core';
@Component({
  selector: 'app-assign-user-role',
  templateUrl: './assign-user-role.component.html',
  styleUrls: ['./assign-user-role.component.scss'],
})
export class AssignUserRoleComponent implements OnInit {
  user: any;
  isEditing: boolean = false;
  roles: any = [];
  form: any;
  submitted: boolean = false;
  public event: EventEmitter<any> = new EventEmitter();
  constructor(
    private admin: AdminService,
    private fb: FormBuilder,
    private common: CommonService
  ) {}

  ngOnInit(): void {
    this.form = this.fb.group({
      userName: ['', !this.isEditing ? Validators.required : null],
      password: ['', !this.isEditing ? Validators.required : null],
      role: [
        this.user?.role || '',
        !this.isEditing ? Validators.required : null,
      ],
    });
    this.admin.getRoles().subscribe((res) => {
      this.roles = res.data;
    });
  }

  get f(): { [key: string]: AbstractControl } {
    return this.form.controls;
  }

  onSubmit($event: any) {
    $event.preventDefault();
    this.submitted = true;
    if (this.form.invalid) {
      return;
    }
    let payload: any = this.form.value;
    if (this.isEditing) {
      let _payload: any = {};
      Object.keys(payload).forEach((keyName) => {
        if (payload[keyName] && payload[keyName].trim()) {
          _payload[keyName] = payload[keyName];
        }
      });
      payload = _payload;
      payload['id'] = this.user.id;
      payload['userName'] = this.user.userName;
      if (!payload['password']) {
        payload['password'] = null;
      }
    }
    this.update(payload);
  }
  update(payload: any) {
    this.admin.updateUser(payload).subscribe((data: any) => {
      if (data && data.code === 'TEX0400') {
        this.common.showError(data.message || '');
      } else {
        this.common.showSuccess(
          this.isEditing ? 'User profile updated' : 'User Created'
        );
        this.close(true);
      }
    });
  }
  close(created: boolean = false) {
    this.event.emit({ close: true, created });
  }
}
