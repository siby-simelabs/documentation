import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { OutcomeService } from 'src/app/providers/admin/kpiconfig/outcome/outcome.service';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { TrackKpiService } from 'src/app/providers/trackKpi/track-kpi.service';

@Component({
  selector: 'app-crpwise',
  templateUrl: './crpwise.component.html',
  styleUrls: ['./crpwise.component.scss'],
})
export class CrpwiseComponent implements OnInit {
  submitted: boolean = false;
  filterForm: FormGroup;
  activityData: any;
  finalProId: any;
  projectInfo: any;
  actData: any = [];
  deleteData: any;
  interId: any;
  interventioList: any;
  interData: any;
  search: any = null;
  activityFilterData: any;
  kpiFilterData: any;
  crpFilterDtata: any;
  monthYearData: any;
  tempActData: any;
  outcomeData: any;
  role: any;
  constructor(
    public track: TrackKpiService,
    public assign: AssigntargetService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public out: OutcomeService
  ) {
    this.filterForm = this.formBuilder.group({
      crp: [null],
      kpi: [null],
      activity: [null],
      year: [null],
    })
  }


  ngOnInit(): void {
    this.access();
    this.onload();
  }
  onload() {
    this.finalProId = this.common.getFundId();
    if (!this.finalProId) {
      this.route.navigate(['/dashboard/track-kpi/list']);
      this.common.showError("Please Select a Project");
    }

    this.activityData = this.common.getAssingTarget();
    this.project.getProjectByID(this.finalProId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.getActivityTargets();
        this.activityNewData();
      }
    })
    this.interData = this.common.getAssingTarget();
    this.getOutcome();
    this.interId = this.activityData.interventionId;
    this.activityData = this.activityData?.activities;
    console.log(this.activityData)
  }
  access() {
    this.role = this.common.screenRole("trackkpi");

  }
  getOutcome() {
    this.out.getOutById(this.activityData.outcomeId).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.outcomeData = resp.data;
      }
    });
  }
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }
  clearFilter() {
    this.actData = this.tempActData;
    this.filterForm.reset();
    this.project.trigger("onFilterApply");
  }
  onSubmit(value: any): void {
    this.submitted = true;
    if (this.filterForm.invalid) {
      this.spinner.hide();
      return;
    }
    else {
      this.actData = this.tempActData;
      let inter: any = this.actData;
      if (value["activity"] != null) {
        inter = inter.filter((item: { activity: any; }) => item.activity == value["activity"]);
      }
      if (value["crp"] != null) {
        inter = inter.filter((item: { crp: any; }) => item.crp == value["crp"]);
      }
      if (value["kpi"] != null) {
        inter = inter.filter((item: { kpi: any; }) => item.kpi == value["kpi"]);
      }
      if (value["year"] != null) {
        inter = inter.filter((item: { year: any; }) => item.year == value["year"]);
      }
      this.actData = inter;
    }
    this.project.trigger("onFilterApply");
  }
  getMonthanme(month: any) {
    if (month == 1) return "Janaury";
    if (month == 2) return "February";
    if (month == 3) return "March";
    if (month == 4) return "April";
    if (month == 5) return "May";
    if (month == 6) return "June";
    if (month == 7) return "July";
    if (month == 8) return "August";
    if (month == 9) return "September";
    if (month == 10) return "October";
    if (month == 11) return "November";
    else return "December";
  }
  getPryear(year: string) {
    return year;
  }
  activityNewData() {
    this.actData = [];
    let activityArr: any = this.activityData;
    for (let i = 0; i < activityArr.length; i++) {
      for (let j = 0; j < activityArr[i].actual.monthly?.length; j++) {
        let obj = {
          "activity": activityArr[i].activityName,
          "kpi": activityArr[i].kpi,
          "year": activityArr[i].actual.monthly[j]?.year,
          "month": activityArr[i].actual.monthly[j]?.month,
          "crp": activityArr[i].actual.monthly[j]?.crp,
          "target": activityArr[i].targetSplit,
          "actual": activityArr[i].actual.monthly[j].actualAmount,
          "monthData": activityArr[i].actual.monthly[j],
          "approved": activityArr[i].actual.monthly[j].approved,
          "activityId": activityArr[i].activityId,
          "crpStatus": activityArr[i].actual.monthly[j]?.crpStatus,
          "personId": activityArr[i].actual.monthly[j]?.shgId
        }
        this.actData.push(obj);
      }
    }
    this.tempActData = this.actData;
    const act = this.actData.map((o: { activity: any; }) => o.activity)
    this.activityFilterData = this.actData.filter(({ activity }: any, index: number) => !act.includes(activity, index + 1));

    const crpd = this.actData.map((o: { crp: any; }) => o.crp)
    this.crpFilterDtata = this.actData.filter(({ crp }: any, index: number) => !crpd.includes(crp, index + 1));

    const kpid = this.actData.map((o: { kpi: any; }) => o.kpi)
    this.kpiFilterData = this.actData.filter(({ kpi }: any, index: number) => !kpid.includes(kpi, index + 1));

    const yeard = this.actData.map((o: { year: any; }) => o.year)
    this.monthYearData = this.actData.filter(({ year }: any, index: number) => !yeard.includes(year, index + 1));
    console.log(this.interData)
    this.common.trigger("updateHeaderDetails", [{
      name: 'Project ID',
      value: this.projectInfo.id
    },
    {
      name: 'Project Name',
      value: this.projectInfo.name
    },
    {
      name: 'Int. Name',
      value: this.interData.name
    },
    {
      name: 'Outcome',
      value: this.interData.outcome
    }])
  }
   
  getCrpTotal(group: any) {
    let year = group.year;
    if (group.target[0]?.year == 9999) year = 9999;
    let grandtotalCrp = 0;
    for (let i = 0; i < group.target.length; i++) {
      if (year == group.target[i].year) {
        for (let j = 0; j < group.target[i]?.blocks?.length; j++) {
          for (let k = 0; k < group.target[i].blocks[j]?.crp?.length; k++) {
            grandtotalCrp = grandtotalCrp + group.target[i].blocks[j].crp[k].target;
          }
        }
      }
    }
    return grandtotalCrp;
  }
  monthData(grp: any) {
    this.route.navigateByUrl('/dashboard/track-kpi/add-crpwise', { state: { flag: 'edit', data: grp } });
  }
  view(grp: any) {
    this.route.navigateByUrl('/dashboard/track-kpi/add-crpwise', { state: { flag: 'view', data: grp } });
  }
  delete(group: any) {
    this.deleteData = {
      "interventionActualPostRequests": [{
        "yearId": group.year,
        "monthId": group.monthData.month,
        "actualReportingDate": this.dateFormat(group.monthData.reportedDt),
        "interventionId": this.interId,
        "userId": group.monthData.crpId,
        "actuals": group.monthData.actual,
        "activityId": group.monthData.activityId,
        "personId": group.monthData.shgId
      }]
    }
    if (confirm('Are you sure you want to delete this record?')) {
      this.track.deleteActual(this.deleteData).subscribe(resp => {
        if (resp) {
          this.common.showSuccess("Successfully Deleted");
          this.getKpiInter();
        }
      })
    }
  }
  dateFormat(date: string) {
    const today = new Date(date);
    const yyyy = today.getFullYear();
    let mm: any = today.getMonth() + 1;
    let dd: any = today.getDate();
    if (dd < 10) dd = '0' + dd;
    if (mm < 10) mm = '0' + mm;
    let formattedToday = dd + '/' + mm + '/' + yyyy;
    return formattedToday
  }
  approve(group: any) {
    if (confirm('Are you sure you want to approve this record?')) {
      let crpWise = {
        "interventionActualPostRequests": [{
          "yearId": group.year,
          "monthId": group.monthData.month,
          "actualReportingDate": this.dateFormat(group.monthData.reportedDt),
          "interventionId": this.interId,
          "userId": group.monthData.crpId,
          "actuals": group.monthData.actual,
          "approvalFlag": true,
          "activityId": group.monthData.activityId,
          "personId": group.monthData.shgId
        }]
      }
      this.saveCrp(crpWise);
    }

  }
  saveCrp(data: any) {
    this.spinner.show();
    this.track.editActual(data).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        this.getKpiInter();
      }
      else {
        this.common.showError(resp.message);
      }

    })
  }
  getKpiInter() {
    this.spinner.show();
    this.assign.getAllInterventionForKpi(this.finalProId, this.search).subscribe(resp => {
      this.spinner.hide()
      this.interventioList = resp.data.intervention_list;
      for (let i = 0; i < this.interventioList.length; i++) {
        if (this.interventioList[i].interventionId == this.interData.interventionId) {
          this.common.saveAssingTarget(this.interventioList[i]);
          this.onload();
        }
      }
    })
  }
  getActivityTargets() {
    console.log(this.interData);
    console.log(this.activityData.activityId)
  }
  yearCalc(year: any) {
    if (year == 9999) {
      return "12 Months";
    }
    else {
      return year;
    }
  }
  status(st: boolean) {
    if (st == true) {
      return "Active";
    }
    else {
      return "Inactive";
    }
  }
}
