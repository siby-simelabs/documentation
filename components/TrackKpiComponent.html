import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { CommonService } from 'src/app/providers/common.service';
import { FundService } from 'src/app/providers/fund/fund.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { UtilizationService } from 'src/app/providers/utilization/utilization.service';
@Component({
  selector: 'app-track-kpi',
  templateUrl: './track-kpi.component.html',
  styleUrls: ['./track-kpi.component.scss'],
})
export class TrackKpiComponent implements OnInit {

  pageNo: number = 1;
  currentPage = 1;
  pageSize: number = 10;
  datas: any;
  role: any;
  id: any;
  nameOfProject: any;
  sector: any;
  submitted: boolean = false;
  searchData: any = [];
  total: any;
  issearch:boolean=false ;
  constructor(
    public fund: FundService,
    public utilization: UtilizationService,
    public common: CommonService,
    public project: ProjectService,
    public formBuilder: FormBuilder,
    public route: Router,
    private activatedRoute: ActivatedRoute,
    public spinner: NgxSpinnerService,
  ) { }
  filterForm = this.formBuilder.group(
    {
      id: [null],
      sector: [null],
      nameOfProject: [null],
      status: ["Approved"]
    },
  );
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }


  ngOnInit(): void {
    this.common.clearProjectId();
    this.spinner.hide();
    this.common.getAllSectors().subscribe((resp) => {
      if (resp) {
        this.sector = resp.data;
      }
    });
    this.getList();
    this.access()

  }
  access() {
    this.role = this.common.screenRole("trackkpi");

  }
  getList() {
    this.spinner.show();
    this.utilization.getApproved(this.pageNo, this.pageSize).subscribe(resp => {
      this.spinner.hide();
      if (resp) {
        this.datas = resp.data?.fundProjectModelsList;
        this.total = resp.data?.totalCount;
      }
    })
  }

  onSubmit(value: any): void {
    this.issearch = true;
    this.submitted = true;
    this.pageNo = 1;
    this.pageSize = 10;
    this.currentPage = 1;
    if (this.filterForm.invalid) {
      return;
    }
    else {
      this.searchData =
      {
        "searchModel": {
          "projectId": value["id"],
          "nameOfProject": value["nameOfProject"],
          "status": "Final Approved and locked",
          "sector": value["sector"]
        },
        "pageNumber": this.pageNo,
        "pageSize": this.pageSize
      }
      this.searchFile(this.searchData);
    }

  }
  searchFile(data: any) {
    this.spinner.show();
    this.utilization.filterProject(data).subscribe((resp) => {
      this.project.trigger("onFilterApply");
      this.spinner.hide();
      if (resp) {
        this.datas = resp.data?.fundProjectModelsList;
        this.total = resp.data?.totalCount;
      }
    })
  }
  clearFilter() {
    this.getList();
    this.filterForm.reset();
    this.searchData = [];
    this.issearch = false;
    this.currentPage = 1;
    this.project.trigger("onFilterApply");
  }

  editFund(fId: any) {
    this.common.storeFundId(fId);
    this.route.navigate(['/dashboard/track-kpi/interlist'])
  }
  pageChanged(event: any) {
    if (event.page != this.pageNo) {
      this.pageNo = event.page;
      this.pageSize = event.itemsPerPage;
     
      if (this.issearch==true) {
        this.searchData.pageNumber = event.page;
        this.searchFile(this.searchData);
      }
      else {
        this.getList();
      }
    }

  }

}
