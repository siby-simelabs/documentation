import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Api } from 'src/app/providers/api.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { ReviewService } from 'src/app/providers/project/review/review.service';
import { UserService } from 'src/app/providers/user.service';

@Component({
  selector: 'app-review-add',
  templateUrl: './review-add.component.html',
  styleUrls: ['./review-add.component.scss']
})
export class ReviewAddComponent implements OnInit {
  re_view: any;
  reviewAddForm: any;
  projectId: any;
  projectInfo: any;
  vewId: any;
  submitted: boolean = false;
  id: any;
  stage: any;
  comment: any;
  workFlowDate: any;
  fileList: any;
  parms: any;
  pageNo: number = 1;
  pageSize: number = 100;
  datas: any;
  cmnt: any;
  role: any;
  roleBase: any;
  updatestatus: boolean = false;
  submitPropsal: boolean = false;
  unlockforState: boolean = false;
  unlockforCenter: boolean = false;
  username: any;
  statusText: any;
  file:any;

  constructor(public rev: ReviewService, public common: CommonService, public formBuilder: FormBuilder, public route: Router, public project: ProjectService, public spinner: NgxSpinnerService, public user: UserService, private api:Api) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId()
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError("Please Select/Create a Project");
    }
    this.reviewAddForm = this.formBuilder.group({
      stage: ['', Validators.required],
      comment: ['',  Validators.compose([Validators.required, Validators.maxLength(1000)])],
      fileList: ['', Validators.required]
    });
  }

  ngOnInit(): void {
    this.access();
    this.spinner.hide();
    this.parms = history.state.id;
    this.file = history.state.file;
    
   
    if (this.user.hasLoggedIn) {
      this.username = this.user.profile.userName;
    }
    if (history.state.msg) {
      this.reviewAddForm.get("comment")?.setValue(history.state.msg);
     
    }
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    });

  }
  get f(): { [key: string]: AbstractControl } {
    return this.reviewAddForm.controls;
  }

  onSubmit(value: any) {
    let validation = false;
    if ((this.roleBase.name == "pm_central_user_level1_final_approval")&&(this.statusText == "Final approval")) {
      validation = true;
    }
    this.submitted = true;
    if (this.reviewAddForm.invalid) {
      return;
    }
    
    else {
      let reviewdata = {
        "stage": value["stage"],
        "comment": value["comment"],
        "projectId": this.projectId,
        "userName": this.username,
        "validation": validation
      }
      if ((this.user.profile.roleName == "pm_central_user_level1_final_approval")&&(this.statusText == "Final approval")) {
        if ((this.projectInfo.actualEndDate == null) && (this.projectInfo.actualStartDate == null) && (this.projectInfo.ecApprovalDate == null) && (this.projectInfo.sanctionLetterIssuedDate == null)) {
          alert("Please fill in the required dates in project profile and intervention tabs!");
        }
        else{
          this.saveRev(reviewdata);
        }
      }
      else{
        this.saveRev(reviewdata);
      }
      
    }
  }

  access() {
    this.role = this.common.screenRole("review");
    this.roleBase = this.common.getRoleBase();
    
    if (this.roleBase.name == "pm_central_user_level1_scrutiny") {
      this.updatestatus = true;
      this.unlockforState = false;
    }
    if ((this.roleBase.name == "pm_state_user_level2") || (this.roleBase.name == "pm_admin")) {
      this.updatestatus = false;
      this.unlockforState = false;
      this.submitPropsal = true;
    }
  }

  saveRev(data: any) {
    
    let formData: any = new FormData();
    if (this.fileList) {
      formData.append("documentFile", this.fileList, this.fileList.name);
    }
    formData.append("proposal", new Blob([JSON.stringify(data, null, 2)], {
      type: "application/json"
    }));
    this.spinner.show();
    this.rev.addReview(formData).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
          this._handleSuccess();
      }
      else{
        this.common.showError(resp.message)
      }
    },
    (error: any) => {
      console.error(error.error.message);
      this.spinner.hide();
      this.common.showError(error.error.message);
    });
  }
  _handleSuccess(){
    this.sendEmail();
    if ((this.unlockforState == true) && (this.updatestatus == true)) {
      this.common.showSuccess('Proposal status updated successfully.');
      this.route.navigate(['/dashboard/project/review']);
    }
    else {
      this.common.showSuccess('Proposal status updated successfully.');
      this.route.navigate(['/dashboard/project/review']);
    }
    if ((this.user.profile.roleName == "pm_state_user_level1") || (this.user.profile.roleName == "pm_state_user_level2")) {
      if (this.statusText == "Submission by state") {
        this.common.storeProjectViewId(this.projectId);
      }
    }
    if ((this.user.profile.roleName == "pm_central_user_level2")) {
      if (this.statusText == "Unlock") {
        this.common.clearProjectViewId();
      }
    }
    if ((this.user.profile.roleName == "pm_central_user_level2")) {
      if ((this.statusText == "Reject to merge proposal")||(this.statusText == "Project Closed")||(this.statusText == "Rejected")) {
        this.common.storeProjectViewId(this.projectId);
      }
    } 
  }

  sendEmail() {
    let data = {
      projectId: this.projectId,
      projectName: this.projectInfo.name,
      status: this.statusText,
    };
    this.rev.sendMail(data).subscribe((resp) => {
      if (resp) {
        
        this.common.showSuccess(resp.data);
      }
    });
  }
  veiwDoc(){
    this.common.dwnldFile(this.file[0].url, this.file[0].name);
  }

  onReset(): void {
    this.submitted = false;
    this.reviewAddForm.reset();
  }
  onChange(event: any) {
    let file = event.target.files[0];
    this.fileList = event.target.files[0];
    
    if(this.fileList.size >= this.common.maxFileSize){
      this.common.showError("File size too large. Please upload a file less than 50mb");
      this.reviewAddForm.get("fileList")?.setValue(null);
      return
    }
    if(this.fileList.name.length >= this.common.maxFileNameLength){
      this.common.showError("File name length restricted to maximum "+this.common.maxFileNameLength+" characters");
      this.reviewAddForm.get("fileList")?.setValue(null);
      return
    }
  } 
  stageChange(event: any) {
    
    this.statusText = event.target.value;
    if (event.target.value == "Scrutiny") {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
    else if (event.target.value == 'Submission by state') {
      this.unlockforState = false;
      this.unlockforCenter = false;
    }

    else if (event.target.value == 'Returned back to state') {
      this.unlockforState = true;
      this.updatestatus = false;
      this.unlockforCenter = false;
    }

    else if (event.target.value == 'Appraisal') {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
    else if (event.target.value == 'Pre-EC') {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
    else if (event.target.value == 'EC') {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
    else if (event.target.value == 'Rejected') {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
    else if (event.target.value == 'Project Closed') {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
    else if (event.target.value == 'Reject to merge proposal') {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
    else if (event.target.value == 'Rejected') {
      this.unlockforState = false;
      this.updatestatus = false;
      this.unlockforCenter = true;
    }
    else if (event.target.value == 'Unlock') {
      this.unlockforState = false;
      this.updatestatus = false;
      this.unlockforCenter = true;
    }
    else if (event.target.value == 'Final approval') {
      this.unlockforState = false;
      this.updatestatus = true;
      this.unlockforCenter = false;
    }
 

  }
  unlockState() {
    this.common.showSuccess('Proposal unlocked to State.');
    this.route.navigate(['/dashboard/project/review']);

  }
}
