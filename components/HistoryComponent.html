import { Component, OnInit } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { Router, ActivatedRoute } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { BsModalService, BsModalRef, ModalOptions } from 'ngx-bootstrap/modal';

@Component({
  selector: 'app-history',
  templateUrl: './history.component.html',
  styleUrls: ['./history.component.scss']
})
export class HistoryComponent implements OnInit {
  locId: any;
  crpId: any;
  historyData:any;
  history: any = [];
  crpInfo: any;
  proName: any;
  ep:any;
  data:any;
 
  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public project: ProjectService,
    public budget: BudgetService,
    public modalService: BsModalService,
    public spinner: NgxSpinnerService,
    public bsModalRef: BsModalRef
  ) { }

  ngOnInit(): void {
    this.crpId = this.common.getCrpId();
    this.historyData = this.modalService.config.initialState;
    this.locId=this.historyData.locId;
    this.ep = this.historyData.endpoint;
    this.data = this.historyData.data;
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    })

    
    this.getHistory();
  }
  mngStatus(status:boolean){
    if(status == true){
      return "Active";
    }
    else{
      return "Inactive";
    }
  }
  getHistory(){
    this.manager.history(this.crpId,this.locId, this.ep, this.data).subscribe((resp)=>{
      if(resp){
        
        this.history = resp.data;
        if(this.history === null){
          this.history = [];
        }
       
      }
    })
  }
  closeMdl(){
    
    this.bsModalRef.hide();
    this.modalService.hide(0)
  }

}
