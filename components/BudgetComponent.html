import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,

  Validators,
} from '@angular/forms';

import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';

import { ProjectService } from 'src/app/providers/project/project.service';
import { UserService } from 'src/app/providers/user.service';

@Component({
  selector: 'app-budget',
  templateUrl: './budget.component.html',
  styleUrls: ['./budget.component.scss'],
})
export class BudgetComponent implements OnInit {

  central: any;
  state: any;

  budgetForm: FormGroup;
  projectInfo: any;
  isChecked = false;
  submitted: boolean = false;

  budgetFileData: any;
  readonly: boolean = true;

  otherContributionType: any;
  role: any;

  constructor(
    public common: CommonService,
    public formBuilder: FormBuilder,
    public route: Router,
    public project: ProjectService,
    public budget: BudgetService,
    public spinner: NgxSpinnerService,
    public user: UserService,
    public disburs: DisbursService,
  ) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId();
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError('Please Select/Create a Project');
    }
    this.budgetForm = this.formBuilder.group({
      totalBudgetProposed: ['', [Validators.required]],
      budgetComment: [],
      grandTotalAmount: ['', [Validators.required]],
      budgetUtilizationApp: [],
      otherContributionAmount: [],
      shareDistribution: ['', [Validators.required]],
      centralShare: ['', [Validators.required]],
      stateShare: ['', [Validators.required]],
      otherContributionType: [],
      proposalFile: [],
      budgetFile: []
    });
  }
  get f(): { [key: string]: AbstractControl } {
    return this.budgetForm.controls;
  }

  ngOnInit(): void {
    this.access();
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp?.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
        this.budgetForm.get("name")?.setValue(resp?.data);
      }
    });
    this.budget.getShare(this.code).subscribe((resp) => {
      if(resp){
        this.shareDis = resp.data;
      }
    })

    this.getBudget();
    this.getContribute();
    this.getDisbursement();
  }
  access() {
    this.role = this.common.screenRole("budget");
    if ((this.user.profile.roleName == "pm_central_user_level1_final_approval")) {
      this.vewId = this.projectId;
    }
  }
  getBudget() {
    this.spinner.show();
    this.budget.getBudget(this.projectId).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.budgetData = resp.data;
        if (this.budgetData) {
          this.budgetId = this.budgetData.id;
          this.budgetForm.controls['grandTotalAmount'].setValue(this.budgetData.grandTotalAmount);
          this.budgetForm.controls['budgetComment'].setValue(this.budgetData.budgetComment);
          this.budgetForm.controls['totalBudgetProposed'].setValue(this.budgetData.totalBudgetProposed);
          this.isChecked = this.budgetData.budgetUtilizationApp == true  ?   true :   false;
          this.budgetForm.controls['otherContributionType'].setValue(this.budgetData.otherContributionType);
          this.budgetForm.controls['otherContributionAmount'].setValue(this.budgetData.otherContributionAmount);
          this.budgetForm.controls['shareDistribution'].setValue(this.budgetData.shareDistribution);
          this.budgetForm.controls['centralShare'].setValue(this.budgetData.centralShare);
          this.budgetForm.controls['stateShare'].setValue(this.budgetData.stateShare);
        }
      }
    })
  }
  onSubmit(value: any) {
    this.submitted = true;
    this.budgetForm.addControl(
      'projectId',
      new FormControl('', Validators.required)
    );
    this.budgetForm.controls['projectId'].setValue(this.projectId);
    if (this.budgetForm.invalid) {
      return;
    }
    else {
      let docdata = {
        "id": this.budgetId,
        "projectId": this.projectId,
        "budgetComment": value["budgetComment"],
        "grandTotalAmount": value["grandTotalAmount"],
        "totalBudgetProposed": value["totalBudgetProposed"],
        "budgetUtilizationApp": this.isChecked,
        "shareDistribution": value["shareDistribution"],
        "otherContributionType": value["otherContributionType"],
        "otherContributionAmount": value["otherContributionAmount"],
        "stateShare": value["stateShare"],
        "centralShare": value["centralShare"]
      }
      this.saveBudget(docdata);
    }
  }
  saveBudget(docdata: any) {

    let formData: any = new FormData();
    if (this.proposalFileData) {
      formData.append("proposalFile", this.proposalFileData, this.proposalFileData.name);
    }
    else {
      formData.append("proposalFile", "");
    }
    if (this.budgetFileData) {
      formData.append("budgetFile", this.budgetFileData, this.budgetFileData.name);
    }
    else {
      formData.append("budgetFile", "")
    }

    formData.append("budget", new Blob([JSON.stringify(docdata, null, 2)], {
      type: "application/json"
    }));

    if (this.budgetData != null) {
      if (this.disbursement.length > 0) {
        if (confirm('Changing the budget data will affect disbursement configurations!')) {
          this.saveBudgetFormdata(formData);
        }
      }
      else {
        this.saveBudgetFormdata(formData);
      }
    }
    else {
      this.saveBudgetFormdata(formData);
    }

  }

  getDisbursement() {
    this.disburs.getDis(this.pageNo, this.pageSize, this.projectId).subscribe(resp => {
      if (resp) {
        this.disbursement = resp.data;
      }
    });
  }

  saveBudgetFormdata(formData: any) {
    this.spinner.show();
    this.budget.saveBudget(formData).subscribe((resp) => {
      if (resp) {
        this.spinner.hide();
        if (this.budgetData != null) {
          this.common.showSuccess('Successfully Updated Budget');
          this.route.navigate(['/dashboard/project/headwise']);
        }
        else {
          this.common.showSuccess('Successfully Created Budget');
          this.route.navigate(['/dashboard/project/headwise']);
        }

      }
    });
  }

  fileData(arg0: string, fileData: any, name: any) {
    throw new Error('Method not implemented.');
  }

  onTotalChange(event: any) {
    let shareid: any;
    this.selectedOption = this.budgetForm.get('shareDistribution')?.value;
    this.totalBudgetProposed = this.budgetForm.get('totalBudgetProposed')?.value;
    this.otherContributionAmount = this.budgetForm.get('otherContributionAmount')?.value;
    if (this.selectedOption != 4) {
      for (let i: any = 0; i <= this.shareDis.length; i++) {
        if (this.shareDis[i]?.id == this.selectedOption) {
          let stateValue: any = this.shareDis[i]?.share[0].state;
          let centralValue: any = this.shareDis[i]?.share[0].central;

          this.state = this.totalBudgetProposed * stateValue / 100;
          this.central = this.totalBudgetProposed * centralValue / 100;
          this.grandTotalAmount = this.totalBudgetProposed + this.otherContributionAmount;

          this.budgetForm.controls['stateShare'].setValue(this.state);
          this.budgetForm.controls['centralShare'].setValue(this.central);
          this.budgetForm.controls['grandTotalAmount'].setValue(this.grandTotalAmount);
        }
      }
    }
  }
  changeStatus(event: any) {
    if (event.srcElement.checked == true) {
      this.isChecked = true;
      this.budgetForm.controls['budgetUtilizationApp'].setValue(true);
    }
    else {
      this.budgetForm.controls['budgetUtilizationApp'].setValue(false);
      this.isChecked = false;
    }
  }
  onChange(event: any, type: string) {
    if (type == "proposalFile") {
      if (event.target.files[0].size >= this.common.maxFileSize) {
        this.common.showError("File size too large. Please upload a file less than 50mb");
        this.budgetForm.get("proposalFile")?.setValue(null);
        return
      }
      else if (event.target.files[0].name.length >= this.common.maxFileNameLength) {
        this.common.showError("File name length restricted to maximum " + this.common.maxFileNameLength + " characters");
        this.budgetForm.get("proposalFile")?.setValue(null);
        return
      }
      else {
        this.proposalFileData = event.target.files[0];
      }

    }
    if (type == "budgetFile") {
      if (event.target.files[0].size >= this.common.maxFileSize) {
        this.common.showError("File size too large. Please upload a file less than 50mb");
        this.budgetForm.get("budgetFile")?.setValue(null);
        return
      }
      else if (event.target.files[0].name.length >= this.common.maxFileNameLength) {
        this.common.showError("File name length restricted to maximum " + this.common.maxFileNameLength + " characters");
        this.budgetForm.get("budgetFile")?.setValue(null);
        return
      }
      else {
        this.budgetFileData = event.target.files[0];
      }
    }

  }
  shareChange(event: any) {
    this.onTotalChange(event);
    this.readonly = (event == 4) ?   false :  true;
  }

  otherContriValue(event: any) {
    this.selectedOption = this.budgetForm.get('shareDistribution')?.value;
    if (this.selectedOption == 4) {
      this.StateCentral(event);
    }
    else {
      this.onTotalChange(event);
    }
  }
  otherContri(event: any) {
    this.otherAmountInput = (!event) ?  true :  false;
  }

  StateCentral(event: any) {
    this.selectedOption = this.budgetForm.get('shareDistribution')?.value;
    this.otherContributionAmount = this.budgetForm.get('otherContributionAmount')?.value;
    this.totalBudgetProposed = this.budgetForm.get('totalBudgetProposed')?.value;
    if (this.selectedOption == 4) {
      this.state = this.budgetForm.get('stateShare')?.value;
      this.central = this.budgetForm.get('centralShare')?.value;
      let sharevalue = this.state + this.central;
      this.budgetForm.controls['totalBudgetProposed'].setValue(sharevalue);
      this.grandTotalAmount = this.totalBudgetProposed + this.otherContributionAmount;
      this.budgetForm.controls['grandTotalAmount'].setValue(this.grandTotalAmount);
    }
  }
  getContribute() {
    this.common.getContribute().subscribe((resp) => {
      if (resp) {
        this.contribute = resp.data;
      }
    });
  }
  download(url: string, docname: string) {
    this.common.dwnldFile(url, docname);
  }
}
