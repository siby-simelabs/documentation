import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-trackinter',
  templateUrl: './trackinter.component.html',
  styleUrls: ['./trackinter.component.scss']
})
export class TrackinterComponent implements OnInit {

  submitted: boolean = false;

  targetTotal: any;
  finalProId: any;
  interventioList: any;
  search: any = null;
  projectInfo: any;
  interList: any;
  interData: any;
  outcomeData: any;
  newInterlist: any;
  role: any;
  level:string | undefined;

  constructor(
    public assign: AssigntargetService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public project: ProjectService
  ) {

  }

  ngOnInit(): void {
    this.access();
    this.finalProId = this.common.getFundId();
    this.project.getProjectByID(this.finalProId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        
      }
    })
    this.getKpiInter();
  }
  access() {
    this.role = this.common.screenRole("trackkpi");

  }
  
  levelChange(value:string){
    console.log(value)
  }

  filterData() {
    let inter = this.interventioList?.intervention_list;


    if (this.interData != null) {
      inter = inter.filter((item: { name: any; }) => item.name == this.interData);
      this.interList = inter;
      if (this.outcomeData != null) {
        inter = inter.filter((item: { outcome: any; }) => item.outcome == this.outcomeData);
        this.interList = inter;
        if (this.level != undefined) {
          inter = inter.filter((item: { level: any; }) => item.level == this.level);
          this.interList = inter;
        }
      }
    }  
    this.project.trigger("onFilterApply");
  }


  getKpiInter() {
    this.spinner.show();
    this.assign.getAllInterventionForKpi(this.finalProId, this.search).subscribe(resp => {

      this.spinner.hide()
      this.interventioList = resp.data;
      this.interList = this.interventioList?.intervention_list;
      let temp: any = [];
      for (let i = 0; i < this.interList?.length; i++) {
        if (this.interList[i].outcome != null) {
          temp.push(this.interList[i]);
        }
      }
      this.interList = temp;
      this.newInterlist = temp;
      console.log(this.newInterlist)
      this.common.trigger("updateHeaderDetails", [{
        name: 'Project ID',
        value: this.projectInfo.id
      },
      {
        name: 'Project Name',
        value: this.projectInfo.name
      } ])

     })
  }
  view(grp: any) {
    this.common.saveAssingTarget(grp);
    this.route.navigateByUrl("/dashboard/track-kpi/crpwise");
  }
  target(activity: any) {
    if (activity != null) {
      let targetTotal = 0;
      for (let i = 0; i < activity.length; i++) {
        targetTotal = targetTotal + activity[i].total
      }
      return targetTotal;
    }
    else {
      return 0;
    }

  }
  actualTotal(activity: any) {

    let actualtotal = 0;
    if (activity != null) {
      for (let i = 0; i < activity.length; i++) {
        if (activity[i].actual.monthly != null) {
          for (let j = 0; j < activity[i].actual.monthly.length; j++) {
            actualtotal = activity[i].actual.monthly[j].actualAmount + actualtotal;
          }
        }
      }
      return actualtotal;
    }
    else {
      return 0;
    }
  }

  clearFilter() {
    this.outcomeData = null;
    this.interData = null;
    this.interList = this.newInterlist;
    this.level = "";
    this.project.trigger("onFilterApply");
  }

}
