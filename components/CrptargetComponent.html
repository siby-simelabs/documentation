import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
@Component({
  selector: 'app-crptarget',
  templateUrl: './crptarget.component.html',
  styleUrls: ['./crptarget.component.scss']
})
export class CrptargetComponent implements OnInit {
  targetDate: any;
  minDate = new Date();
  crpForm: FormGroup;
  year: any;
  total: any;
  submitted: boolean = false;
  assignCrp: any;
  assignTlist: any;
  crpList: any = [];
  interventionData: any;
  activity: any;
  prid: any;
  crpData: any = [];
  projectInfo: any;
  localflag: any;
  newCrplist: any;
  finalProId: any;
  search: any = null;
  blockId: any;

  mdlSm: boolean = false;
  changeId: any;
  blockCode: any;
  crpId: any;
  crpStateData: any;
  interventioList: any;
  localCrp: any = [];

  state = 16;
  districtList: any;
  blockList: any;
  grmList: any;
  villageListFilter: any;
  villageNewList: any;
  parentBlock: any;
  selectedYear:any;
  totalCrp: number = 0;
  tempTarget: any;
  tempLocalCrp: any;
  flag: any;
  deadMinDate: any;
  deadMaxDate: any;
  role: any; 
  placement : TypeDatePickerPlacement= 'bottom'

  constructor(public common: CommonService,
    public assign: AssigntargetService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public manager: ManagerService, 
    public dp :DatePickerPlacement
  ) {
    this.crpForm = this.formBuilder.group({
      year: ['',],
      total: [null, Validators.required],
      targetDate: [null, Validators.required],
    })
  }
  filterForm = this.formBuilder.group(
    {      
      name: [null],
    },
  );

  ngOnInit(): void {
    this.access();
    this.onload()
  }
  onload() {
    this.assignTlist = this.common.getYearWaise();
    this.interventionData = this.common.getAssingTarget();
    this.activity = this.common.getActivity();
    this.prid = this.common.getFundId();
    this.localflag = this.common.getFlag();
    this.finalProId = this.common.getFundId();
    this.parentBlock = this.common.getBlockCrp();
    
    this.crpStateData = this.parentBlock?.crp;
    console.log(this.activity.crpMonthlyTotal)
    this.selectedYear = history.state.year;
    this.flag = history.state.flag
    this.crpForm.get("year")?.setValue(this.selectedYear);
    this.blockId = this.parentBlock.blockId


    this.project.getProjectByID(this.prid).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.deadMinDate = this.dateForMindate(this.projectInfo.actualStartDate);
        this.deadMaxDate = this.dateForMindate(this.projectInfo.actualEndDate);
        this.common.trigger(this.common.headerEvent,[{
          name : 'Project ID' ,
          value :  this.projectInfo.id
        },{
          name : 'Project Name' ,
          value :  this.projectInfo.name
        },{
          name : 'Int. Name' ,
          value :  this.interventionData?.name
        } ,{
          name : 'KPI' ,
          value :  this.activity.kpi
        } ])
      }
    })
    this.getListCrp();
  }
  access() {
    this.role = this.common.screenRole("assigncrptarget");
    
  }
  
  dateForMindate(date: any) {
    if(!date || typeof date !== "string") {
      return;
    }
      let datearray = date.split('/');
      let newdate = datearray[1] + '/' + datearray[0] + '/' + datearray[2];
  
      return new Date(newdate);
    }

  getDist() {
    this.common.getDistrict(this.state).subscribe((resp) => {
      this.districtList = resp.data;
    });
  }
  distChange(event: any) {
    this.blockList = [];
    this.filterForm.get("blockName")?.setValue(null);
    const SelectVal = event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.common.getBlock(SelectVal, this.changeId).subscribe((resp) => {
      this.blockList = resp.data;
    })
  }
  blockChange(event: any) {
    let gramArray: any = [];
    const SelectVal = event.item.blockName;
    for (let i = 0; i < this.newCrplist.length; i++) {
      for (let j = 0; j < this.newCrplist[i].villageList.length; j++) {
        if (this.newCrplist[i].villageList[j].blockName == SelectVal) {
          gramArray = gramArray.concat(this.newCrplist[i].villageList[j])
        }
      }
    }
    const gids = gramArray.map((o: { gramapanchayat: any; }) => o.gramapanchayat)
    const filteredGram = gramArray.filter(({ gramapanchayat }: any, index: number) => !gids.includes(gramapanchayat, index + 1));
    this.grmList = filteredGram;
    this.blockCode = SelectVal;
  }
  gramChange(event: any) {
    this.villageListFilter = [];
    const SelectVal = event.item.panchayat;
    for (let i = 0; i < this.villageNewList.length; i++) {
      if (this.villageNewList[i].panchayat == SelectVal) {
        this.villageListFilter.push(this.villageNewList[i]);
      }
    }
    const gids = this.villageListFilter.map((o: { village: any; }) => o.village)
    const filteredGram = this.villageListFilter.filter(({ village }: any, index: number) => !gids.includes(village, index + 1));
    this.villageListFilter = filteredGram;
  }

  getListCrp() {
    this.crpList = [];
    this.spinner.show();
    this.manager.getCrpList(this.prid).subscribe(resp => {
      this.spinner.hide();
      if (resp) {
        this.newCrplist = resp.data;
        for (let i = 0; i < this.newCrplist?.length; i++) {
          let tempData = {
            "name": this.newCrplist[i].crpName,
            "crpId": String(this.newCrplist[i].personId),
            "blockId": this.blockId,
            "target": null,
            "targetDate": null,
            "status": this.newCrplist[i].status,
          }
          this.crpList.push(tempData);
          
          this.mergeData();
        }
      }
    })
  }
  mergeData() {
    let tempData: any = [];
    this.localCrp = []
    let ids = new Set(this.crpStateData.map((d: { crpId: any; }) => d.crpId));
    let merged = [this.crpStateData, this.crpList.filter((d: { crpId: number; }) => !ids.has(d.crpId))];
    let mergedArr = [].concat.apply([], merged);
    
    this.localCrp = mergedArr;
    this.tempLocalCrp = mergedArr;
    this.totalCrpTarget();
  }

  get f(): { [key: string]: AbstractControl } {
    return this.crpForm.controls;
  }
  showMdlSm(grp: any, type: string) {
    console.log(grp)
    this.mdlSm = true;
    this.crpId = grp.crpId;
    this.tempTarget = grp.target | 0;
    console.log(this.tempTarget)
    let tempData = {
      "blockid": this.blockId,
      "crpid": this.crpId
    }
    this.crpData = tempData;
    this.crpForm.get("total")?.setValue(grp.target);
    this.crpForm.get("targetDate")?.setValue(grp.targetDate);
  }
  closeMdlSm() {
    this.mdlSm = false;
  }
  onSubmitFilter(value: any) {
    let temp = this.tempLocalCrp;
    
    if (value["name"] != null) {
      this.localCrp = this.find(temp, value["name"]?.toLowerCase());
      
    }
    this.common.closeActiveFilter();
  }
  find(items: any, text: any) {
    text = text.split(' ');
    return items.filter((item: { name: any | any[]; }) => {
      return text.every((el: any) => {
        let result = item.name;
        let caps = result.toLowerCase()
        return caps?.includes(el);
      });
    });
  }
  clearFilter() {
    this.localCrp = this.tempLocalCrp;
    this.filterForm.get("name")?.setValue(null);
    this.common.closeActiveFilter();
  }
  onSubmit(value: any) {
    this.submitted = true;
    if (this.crpForm.invalid) {
      return;
    }
    else {
      this.assignCrp = {
        "interventionId": this.interventionData.interventionId,
        "activityId": this.activity.activityId,
        "outcomeId": this.interventionData.outcomeId,
        "blockId": this.blockId,
        "crpId": this.crpId,
        "year": value["year"],
        "target": value["total"],
        "targetDate": (<HTMLInputElement>document.getElementById("targetDate")).value
      };
       
      if (value["total"] > this.totalCrp) {
        this.common.showError("The total assigned target exceeded the block target!");
      }
      else {
        
        this.saveBlk(this.assignCrp);
      }
    }
  }
  totalCrpTarget() {
    this.totalCrp = parseInt(this.parentBlock.target) - parseInt(this.activity.crpMonthlyTotal);
  }
  saveBlk(data: any) {
    this.spinner.show();
    this.assign.addCrp(this.assignCrp).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        this.common.showSuccess('Successfully assigned CRP Target');
        this.mdlSm = false;
        this.getInterventionList();
        this.onReset();
      }
      else {
        this.common.showError(resp.message);
        this.mdlSm = false;
      }
    })
  }
  onReset(): void {
    this.submitted = false;
    this.crpForm.reset();
  }

  backToBlock(){
    this.route.navigateByUrl('/dashboard/assigntarget/blocktarget', { state: { flag:this.flag} });
  }

  status(st:boolean){
    if(st == true){
      return "Active";
    }
    else{
      return "Inactive";
    }
  }
_saveActivity(i : number , j : number  ){
  if (this.interventioList[i].activities[j].activityId == this.activity.activityId) {
    this.common.saveActivity(this.interventioList[i].activities[j]);
    for (let k = 0; k < this.interventioList[i].activities[j].targetSplit.length; k++) {
      this.spinner.hide();
      if (this.interventioList[i].activities[j].targetSplit[k].year == this.assignTlist.year) {
        this.common.yearWaise(this.interventioList[i].activities[j].targetSplit[k]);
        for (let l = 0; l < this.interventioList[i].activities[j].targetSplit[k].blocks.length; l++) {
          if (this.interventioList[i].activities[j].targetSplit[k].blocks[l].blockId == this.blockId) {
            this.common.saveBlockCrp(this.interventioList[i].activities[j].targetSplit[k].blocks[l]);
            this.onload()
          }
        }

      }
    }
  }
}
  getInterventionList() {
    this.spinner.show();
    this.assign.getAllIntervention(this.finalProId, this.search).subscribe(resp => {
      this.spinner.hide()
      this.interventioList = resp.data.intervention_list;
      for (let i = 0; i < this.interventioList.length; i++) {
        if (this.interventionData.interventionId == this.interventioList[i].interventionId) {
          this.common.saveAssingTarget(this.interventioList[i]);
          for (let j = 0; j < this.interventioList[i].activities.length; j++) {
              this._saveActivity( i  , j )
          }
        }
