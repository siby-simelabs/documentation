import { Component, OnInit, Input, Renderer2, ViewChild } from '@angular/core';
import { ProjectService } from 'src/app/providers/project/project.service';
@Component({
  selector: 'app-page-header',
  templateUrl: './page-header.component.html',
  styleUrls: ['./page-header.component.scss'],
})
export class PageHeaderComponent implements OnInit {
  @Input() title: string = '';
  @Input() breadcumbs: BreadCumbType[] = [];
  @Input() showHome: boolean = true;
  @Input() hasFilter: boolean = false;
  @ViewChild('toggleButton') toggleButton: any;
  @ViewChild('filterContainer') filterContainer: any;
  @Input() onClose : ()=>void = ()=>{} ;
  @Input() 
  hideFilter: boolean = true;
  constructor(private renderer: Renderer2 , private   project : ProjectService  ) {
    this.renderer.listen('window', 'click', this.onClickHandler.bind(this));
  }
  ngOnDestroy(){
    this.project.unsubscribe(this.onFilterComplete , 'onFilterApply');
  }
  onFilterComplete = ( value : any )=>{
    this.hideFilter = true;
  }
  onClickHandler(e: Event) {
    if (this.toggleButton && this.filterContainer) {
      if (
        this.toggleButton.nativeElement != e.target &&
        !this.filterContainer.nativeElement.contains(e.target)
      ) {
        this.hideFilter = true;
      }
    }
  }
  ngOnInit(): void {
    this.project.subscribe(this.onFilterComplete , 'onFilterApply');
  }
  toggleFilter() {
    this.hideFilter = !this.hideFilter;
  }
}
