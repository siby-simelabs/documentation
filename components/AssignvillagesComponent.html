import { Component, OnInit } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { GeoService } from 'src/app/providers/project/geo/geo.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { UserService } from 'src/app/providers/user.service';

@Component({
  selector: 'app-assignvillages',
  templateUrl: './assignvillages.component.html',
  styleUrls: ['./assignvillages.component.scss'],
})
export class AssignvillagesComponent implements OnInit {
  crpId: any;
  crpInfo: any;
  proName: any;
  projectId: any;
  pageNo: number = 1;
  pageSize: number = 100;
  datas: any; 
  roleBase: any;
  role:any;
  constructor(
    public geo: GeoService,
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public project: ProjectService,
    public budget: BudgetService,
    public user: UserService,
    public spinner: NgxSpinnerService
  ) { }

  ngOnInit(): void {
    this.access();
    this.spinner.hide();
    this.roleBase = this.common.getRoleBase();
    this.crpId = this.common.getCrpId();
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    });
    this.getListVill();
  }
  access() {
    this.role = this.common.screenRole("assignvillage");
  }
  getListVill() {
    this.spinner.show();
    this.manager.getVillList(this.crpId).subscribe(resp => {
      this.spinner.hide();
      
      if (resp) {
        this.datas = resp.data?.geoCoverageList;
        this.common.trigger('updateHeaderDetails',[{
          name : 'Project ID' ,
          value : this.crpInfo.id 
        },{
          name : "Project Name" ,
          value : this.crpInfo.name
        },{
          name : 'State' ,
          value : this.datas[0]?.stateName 
        }])
      }
    })
  }
  editVill(group: any) {
    this.route.navigateByUrl('/dashboard/crp/add-assignvillages', { state: { data: group, id: group.id, locationId: group.locationId, districtName: group.districtName, blockName: group.blockName, blockid: group.blockId } });
    this.common.storeLocal(group.blockId, "blockId");
  }
  approveVillage(group: any) {  
    if (confirm('Are you sure you want to approve this record?')) {
      this.spinner.show();
      this.manager.approveVillage(group.blockId, this.crpId).subscribe(resp => {
        this.spinner.hide(); 
        if (resp) {
          this.getListVill();
        }
      })
    }
  }
}
