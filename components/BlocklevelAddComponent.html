import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-blocklevel-add',
  templateUrl: './blocklevel-add.component.html',
  styleUrls: ['./blocklevel-add.component.scss']
})
export class BlocklevelAddComponent implements OnInit {
  crpId: any;
  formCreateBlk: FormGroup;
  submitted: boolean = false;
  datas: any;
  id: any;
  status: any;
  crpInfo: any;
  proName: any;
  parms: any;
  blkCrp: any;
  blkuser: any;
  blockOfficer: any;
  userNameInput: boolean = false;
  userStatus: any;
  role:any;

  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public project: ProjectService,
    public budget: BudgetService,
    public spinner: NgxSpinnerService
  ) {
    this.formCreateBlk = this.formBuilder.group({
      userName: [null, Validators.required],
      id: [null],
      status: [null],

    })
  }
  ngOnInit(): void {
    this.access();
    this.parms = history.state.group;
    this.crpId = this.common.getCrpId();
    if (!this.parms) {
      this.common.showError('Please select a district from the list');
      this.route.navigateByUrl('/dashboard/crp/blocklevel');
    }
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    })
    this.getUser();
  }
  access() {
    this.role = this.common.screenRole("blocklevel");
  }
  getUser() {
    this.manager.getAllUsers(this.crpId, "getBlockUser").subscribe((resp) => {
      if (resp) {
        this.datas = resp.data;
        this.common.trigger('updateHeaderDetails',[{
          name : 'Project ID' ,
          value : this.crpInfo.id 
        },{
          name : "Project Name" ,
          value : this.crpInfo.name
        },{
          name : 'State' ,
          value : this.parms.stateName 
        },{
          name : 'District' ,
          value : this.parms.blockName 
        }])
        this.getBlockOfficer();
      }
    })
  }
  onSubmit(value: any) {
    this.submitted = true;
    if (this.formCreateBlk.invalid) {
      return;
    }
    else {
      this.blkCrp = {
        "id": this.parms.projectCovId,
        "projectId": this.crpId,
        "locationId": this.parms.locationId,
        "status": true,
        "personId": this.blkuser.id
      };
      this.saveDist(this.blkCrp);
    }
  }
  saveDist(data: any) {
    this.spinner.show();
    this.manager.addBlk(this.blkCrp).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == "SUCCESS") {
        
        this.common.showSuccess('Successfully assigned block level manager');
        this.route.navigateByUrl('/dashboard/crp/blocklevel');
      }
      else {
        this.common.showError(resp.message);
      }
    })
  }
  onReset(): void {
    this.submitted = false;
    this.formCreateBlk.reset();
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateBlk.controls;
  }
  nameChange(event: any) {
    this.id = "";
    this.status = "";
    
    this.blkuser = this.datas.find((obj: { id: number; }) => {return obj.id === event.item.id;});
    this.formCreateBlk.get("id")?.setValue(this.blkuser.id);
    this.formCreateBlk.get("status")?.setValue(this.blkuser.status);
    this.formCreateBlk.get("name")?.setValue(this.blkuser.userName);
  }

  getBlockOfficer(){
    this.manager.getBlockById(this.parms.projectCovId).subscribe((resp) => {
      if (resp) {
        
        this.blockOfficer = resp.data;
        
        this.formCreateBlk.get("id")?.setValue(this.blockOfficer.userModel.id);
        this.blkuser = this.blockOfficer.userModel;
        
        this.formCreateBlk.get("status")?.setValue(this.blockOfficer.userModel.status);
        this.userStatus = this.blockOfficer.userModel.status;
        if(this.blockOfficer.userModel.userName){
          this.userNameInput = true;
        }
        this.formCreateBlk.get("userName")?.setValue(this.blockOfficer.userModel.userName);
      }
    })
  }
  ChangeName(){
    this.userNameInput = false;
    this.onReset();
  }



}
