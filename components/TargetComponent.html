import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { AssigntargetService } from 'src/app/providers/assigntarget/assigntarget.service';
import { CommonService } from 'src/app/providers/common.service';
import { InterventionService } from 'src/app/providers/project/interventions/intervention.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-target',
  templateUrl: './target.component.html',
  styleUrls: ['./target.component.scss']
})
export class TargetComponent implements OnInit {
 
  targets: any = [];
  interId: any;
  projectInfo: any;
  prid: any;
  activity: any;
  interventionRemoteData: any;
  interventionData: any;
  targetLength: any;
  lastYear: boolean = true;
  finalProId:any;
  search:any = null;
  interventioList:any;
  role: any;

  constructor(public route: Router, public common: CommonService, public project: ProjectService, public inter: InterventionService, public assign: AssigntargetService, public spinner: NgxSpinnerService,) { }

  ngOnInit(): void {
    this.onload();
  }
  onload(){
    this.common.saveFlag("");
    this.access();
    this.finalProId = this.common.getFundId();
    this.activity = this.common.getActivity();
    this.interventionData = this.common.getAssingTarget();
    this.interId = this.interventionData.interventionId;
    this.targets = this.activity?.targetSplit;
    
    this.targetLength = this.targets.length;
    
    
    
    this.prid = this.common.getFundId();
    if (!this.targets) {
      this.route.navigateByUrl('/dashboard/assigntarget/activities', { state: { flag: "backtoactivities" } });
    }
    this.inter.getInterById(this.interId).subscribe((resp) => {
      this.interventionRemoteData = resp.data;
    });
    this.project.getProjectByID(this.prid).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger(this.common.headerEvent,[{
          name : 'Project ID' ,
          value :  this.projectInfo.id
        },{
          name : 'Project Name' ,
          value :  this.projectInfo.name
        },{
          name : 'Int. Name' ,
          value :  this.interventionData?.name
        } ,{
          name : 'KPI' ,
          value :  this.activity.kpi
        } ])
      }
    });
    
    if (this.targets[0].year == 9999) {
      this.lastYear = false;
    }
    if (this.targets.length == 0) {
      this.targets = [{
        "year": 9999,
        "target": this.activity.total,
        "blocks": [],
        "activityId": this.activity.activityId
      }]
      
      
    }
  }
  access() {
    this.role = this.common.screenRole("assigntarget");
    
  }
  edit(grp: any) {
    this.route.navigateByUrl('/dashboard/assigntarget/blocktarget', { state: { targetSplit: grp, flag: "edit" } });
    this.common.yearWaise(grp);
    this.common.saveFlag("edit");
  }
  view(grp: any) {
    this.route.navigateByUrl('/dashboard/assigntarget/blocktarget', { state: { targetSplit: grp, flag: "view" } });
    this.common.yearWaise(grp);
    this.common.saveFlag("edit");
  }
  addNewRec() {
    this.route.navigateByUrl('/dashboard/assigntarget/blocktarget', { state: { flag: "add" } });
    this.common.saveFlag("add");
  }
  getSplit(blk: any) {
    
    if (blk != null) {
      let target = 0;
      for (let i = 0; i < blk.length; i++) {
        target = target + blk[i].target;
      }
      return target;
    }
    else {
      return 0
    }

  }
  delete(grp: any) {
    if (confirm('Are you sure you want to delete this record?')) {
      
      let postData = {
        "interventionId": this.interventionData.interventionId,
        "activityId": this.activity.activityId,
        "outcomeId": this.interventionData.outcomeId,
        "year": grp.year
      }
      
      this.assign.deleteBlock(postData).subscribe((resp) => {
        if (resp) {
          this.common.showSuccess("Successfully deleted!");
          this.getInterventionList();
        }
      })
    }
  }

  getInterventionList() {
    this.spinner.show();
    this.assign.getAllIntervention(this.finalProId, this.search).subscribe(resp => {
      this.spinner.hide()
      this.interventioList = resp.data.intervention_list;
      for (let i = 0; i < this.interventioList.length; i++) {
        if (this.interventionData.interventionId == this.interventioList[i].interventionId) {
          this.common.saveAssingTarget(this.interventioList[i]);
          for (let j = 0; j < this.interventioList[i].activities.length; j++) {
            if (this.interventioList[i].activities[j].activityId == this.activity.activityId) {
              this.common.saveActivity(this.interventioList[i].activities[j]);
              for (let k = 0; k < this.interventioList[i].activities[j].targetSplit.length; k++) {
                this.spinner.hide();
                this.onload();
              }
            }
          }
        }
      }
      this.spinner.hide();
    })
  }

}
