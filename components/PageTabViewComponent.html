import ProjectTabs from '../../../const/projectMenus';

const FUND_TABS = {
  'Central Share': {
    link: '/dashboard/fund/central',

const KPI_CONFIG = {

    link : '/dashboard/admin/outcome' ,
    sub : ['/dashboard/admin/edit-outcome']

    link : '/dashboard/admin/event' ,

const getTabs = (name: string) => {
  let menu = {};
  switch (name) {
    case 'project':
      menu = ProjectTabs;
      break;
    case 'crp':
      menu = CRPTabs;
      break;
    case 'block':
      menu = BLOCK_TABS;
      break;
    case 'fund':
      menu = FUND_TABS;
      break;
      case 'kpi_config' :
        menu = KPI_CONFIG ;
  }
  return menu;
};
@Component({
  selector: 'app-page-tab-view',
  templateUrl: './page-tab-view.component.html',
  styleUrls: ['./page-tab-view.component.scss'],
})
export class PageTabViewComponent implements OnInit {
  @Input() tabs: any;
  @Input() name: string = '';
  @Input() showTabs: boolean = true;
  @Input() hideProjectInfo: boolean = false;
  @Input() showAll : boolean = true ;
  @ViewChild('tab') tabContainer: any;
  @ViewChild('moreItems') moreButton: any;
  @HostListener('window:resize', ['$event'])
  onResize(event: any) {
    this.debounce && clearTimeout(this.debounce);
    this.debounce = setTimeout(this.onResizeEnd, 100);
  }
  maxItems = this.getMaxAllowed();
  debounce: any = null;
  

  constructor(public route: Router) {}
  onResizeEnd = () => {
    this.maxItems = Object.keys(this.tabs).length;
    setTimeout(() => {
      this.adjustTabs();
    });
  };
  ngOnInit(): void {
    this.tabs = getTabs(this.name);
    this.maxItems = this.getMaxAllowed();
  }
  getWidth(ele: any) {
    return ele && ele.getBoundingClientRect
      ? ele.getBoundingClientRect().width
      : 0;
  }
  getMaxAllowed() {
    let screenSize = localStorage.getItem('screen');
    if (screenSize && screenSize == this.getScreenWidth()) {
      let limit = localStorage.getItem('menuAllowed');
      if (limit && !isNaN(parseInt(limit))) {
        return parseInt(limit);
      }
    }

    return this.tabs ? Object.keys(this.tabs).length : 0;
  }
  getScreenWidth() {
    return `${window.innerWidth}`;
  }
  updateToStorage() {
    localStorage.setItem('screen', this.getScreenWidth());
  }
  updateLimit(maxAllowed: number) {
    this.updateToStorage();
    localStorage.setItem('menuAllowed', `${maxAllowed}`);
  }
  adjustTabs() {
    if (
      this.tabContainer &&
      this.tabContainer.nativeElement &&
      this.moreButton &&
      this.moreButton.nativeElement
    ) {
      let wrapper = this.tabContainer.nativeElement,
        moreButton = this.moreButton.nativeElement,
        buttonWidth = this.getWidth(moreButton) || 84.9375,
        outerWidth = this.getWidth(wrapper),
        childNodes = wrapper.childNodes,
        totalWidth = 0;
      for (let i = 0; i < childNodes.length; i++) {
        let item = childNodes[i];
        if (item.nodeType == 8) continue;
        let width = this.getWidth(item);
        if (totalWidth + buttonWidth + width > outerWidth) {
          setTimeout(() => {
          
            this.maxItems = i - 1;
          });
          break;
        }
        totalWidth += width;
      }
    }
  }
  getTabs() {
    return Object.keys(this.tabs);
  }
  hasMoreButton() {
    return this.maxItems < Object.keys(this.tabs).length;
  }
  getLink(tab: string) {
    if (this.tabs.hasOwnProperty(tab)) {
      return this.tabs[tab as any].link;
    }
    return '';
  }
  isHidden(index: number) {
    return index > this.maxItems;
  }
  getMoreItems() {
    return Object.keys(this.tabs).slice(
      this.maxItems + 1,
      Object.keys(this.tabs).length
    );
  }
  ngAfterViewInit() {
    this.adjustTabs();
  }
  isActive(tab: string) {
    let url = this.route.url;
    if (this.tabs.hasOwnProperty(tab)) {
      let item = this.tabs[tab];
      if (item.sub.concat([item.link]).indexOf(url) > -1) {
        return true;
      }
    }
    return false;
  }
}
