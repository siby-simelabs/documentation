import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { FundService } from 'src/app/providers/fund/fund.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
@Component({
  selector: 'app-other',
  templateUrl: './other.component.html',
  styleUrls: ['./other.component.scss'],
})
export class OtherComponent implements OnInit {
  submitted: boolean = false;
  formCreateOther: FormGroup;
  otherAmountReleasedSoFar: any;
  installment: any;
  trancheData: any;
  tranche: any;
  amount: any;
  releaseDate: any;
  fileList: any;
  releaseFromAap: any;
  comments: any;
  budgetId: any;
  fileData: any;
  minDate = new Date();
  otherview: any;
  otherData: any;
  fundId: any;
  total: any;
  budgetData: any;
  otherType: any;
  projectInfo: any;
  deadMinDate: any;
  deadMaxDate: any;
  ocb: any;
  placement : TypeDatePickerPlacement= 'bottom';
  role:any;
  constructor(
    public fund: FundService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    public budget: BudgetService,
    public project: ProjectService,
    public dp :DatePickerPlacement
  ) {
    this.formCreateOther = this.formBuilder.group({
      installment: ['', Validators.required],
      tranche: ['', Validators.required],
      amount: ['', Validators.required],
      releaseDate: ['', Validators.required],
      fileList: [],
      otherAmountReleasedSoFar: [''],
      comments: [],
      budgetType: [],
    });
  }

  ngOnInit(): void {
    this.access();
    this.spinner.hide();
    this.fundId = this.common.getFundId();
    this.getInstallment();
    this.getTranche();
    this.getBudget();
    this.getTotalShare();
    this.project.getProjectByID(this.fundId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.deadMinDate = this.dateForMindate(
          this.projectInfo.actualStartDate
        );
        this.deadMaxDate = this.dateForMindate(this.projectInfo.actualEndDate);
        this.common.trigger(this.common.headerEvent, [
          {
            name: 'Project ID',
            value: resp.data.id,
          },
          {
            name: 'Project Name',
            value: resp.data.name,
          },
          {
            name: 'Sector',
            value: resp.data.sector ,
          },
        ]);
      }
    });
  }
  dateForMindate(date: any) {
  if(!date || typeof date !== "string"){
    return ;
  }
    let datearray = date.split('/');
    let newdate = datearray[1] + '/' + datearray[0] + '/' + datearray[2];

    return new Date(newdate);
  }

  get f(): { [key: string]: AbstractControl } {
    return this.formCreateOther.controls;
  }
  getTotalShare() {
    this.fund.getTotal(this.fundId).subscribe((resp) => {
      if (resp) {
        this.total = resp.data?.fundReleaseTotalModel;
        let total = this.total;
        this.common.trigger('cards', [
          {
            name: 'Centre share',
            value: total?.budgetCentralAmount,
          },
          {
            name: 'State share',
            value: total?.budgetStateAmount,
          },
          {
            name: 'Total budget',
            value: total?.budgetTotalAmount,
          },
          {
            name: 'Other. contri.',
            value: total?.budgetOtherAmount,
          },
          {
            name: 'Grand total',
            value: total?.grandTotalAmount,
          },
          {
            name: 'State release tot.',
            value: total?.stateFundTotal,
          },
          {
            name: 'Centre release tot.',
            value: total?.centralFundTotal,
          },
          {
            name: '% Utilized as on date',
            value: this.splitAmount(total?.utilizedAmount),
          },
        ]);
      }
    });
  }
  getBudget() {
    this.budget.getBudget(this.fundId).subscribe((resp) => {
      if (resp) {
        this.budgetData = resp;
        this.budgetId = resp.data?.id;
        this.ocb = resp.data?.otherContributionBudgetType;
        const otherCtr = resp.data?.otherContributionType;
        if (otherCtr == 'Community') this.otherType = 3;
        else if (otherCtr == 'Line Dept(State)') this.otherType = 4;
        else if (otherCtr == 'Other Centre Dept') this.otherType = 5;
        else if (otherCtr == 'NGO') this.otherType = 6;
        else if (otherCtr == 'Private Partners') this.otherType = 7;
        if (otherCtr) {
          this.formCreateOther.controls['budgetType'].setValue(otherCtr);
        }
        this.getOther(this.otherType);
      }
    });
  }
  getOther(otherType: any) {
    this.spinner.show();
    this.fund.getOthersById(this.budgetId, this.ocb).subscribe((resp) => {
      this.spinner.hide();

      if (resp) {
        this.otherData = resp.data;
        this.formCreateOther.controls['amount'].setValue(
          this.otherData?.amount
        );
        this.formCreateOther.controls['installment'].setValue(
          this.otherData?.installment
        );
        this.formCreateOther.controls['tranche'].setValue(
          this.otherData?.tranche
        );
        this.formCreateOther.controls['otherAmountReleasedSoFar'].setValue(
          this.otherData?.otherAmountReleasedSoFar
        );
        this.formCreateOther.controls['comments'].setValue(
          this.otherData?.comments
        );
        this.formCreateOther.controls['releaseDate'].setValue(
          this.otherData?.releaseDate
        );
      }
    });
  }
  onSubmit(value: any) {
    this.submitted = true;
    if (this.formCreateOther.invalid) {
      return;
    } else {
      let otherFund = {
        installment: value['installment'],
        releaseDate: (<HTMLInputElement>document.getElementById('releaseDate'))
          .value,
        budgetId: this.budgetId,
        tranche: value['tranche'],
        amount: value['amount'],
        releaseFromAap: value['releaseFromAap'],
        otherAmountReleasedSoFar: value['otherAmountReleasedSoFar'],
        budgetType: this.ocb,
        comments: value['comments'],
      };
      this.saveOther(otherFund);
    }
  }
  saveOther(data: any) {
    let formData: any = new FormData();
    if (this.fileData) {
      formData.append('sanctionLetterCopy', this.fileData, this.fileData.name);
    } else {
      formData.append('sanctionLetterCopy', '');
    }
    formData.append(
      'otherFund',
      new Blob([JSON.stringify(data, null, 2)], {
        type: 'application/json',
      })
    );
    this.spinner.show();
    this.fund.addOthers(formData).subscribe((resp) => {
      this.spinner.hide();
      if (resp.message == 'SUCCESS') {
        this.common.showSuccess('Successfully Added Central Share');
        this.route.navigateByUrl('/dashboard/fund/central');
      } else {
        this.common.showError(resp.message);
      }
    });
  }
  onReset(): void {
    this.submitted = false;
    this.formCreateOther.reset();
  }
  onChange(event: any) {
    if (event.target.files[0].size >= this.common.maxFileSize) {
      this.common.showError(
        'File size too large. Please upload a file less than 50mb'
      );
      this.formCreateOther.get('fileList')?.setValue(null);
      return;
    } else if (
      event.target.files[0].name.length >= this.common.maxFileNameLength
    ) {
      this.common.showError(
        'File name length restricted to maximum ' +
          this.common.maxFileNameLength +
          ' characters'
      );
      this.formCreateOther.get('fileList')?.setValue(null);
      return;
    } else {
      this.fileData = event.target.files[0];
    }
  }
  getInstallment() {
    this.common.getInstallmentData().subscribe((resp) => {
      if (resp) {
        this.installment = resp.data;
      }
    });
  }
  getTranche() {
    this.common.getTrancheData().subscribe((resp) => {
      if (resp) {
        this.trancheData = resp.data;
      }
    });
  }
  splitAmount(amt: number) {
    if (amt != 0) {
      return amt.toFixed(2);
    } else {
      return 0;
    }
  }
  download(url: string, docname:string) {
    this.common.dwnldFile(url, docname);
  }
  access() {
    this.role = this.common.screenRole("fund_release");
  }
}
