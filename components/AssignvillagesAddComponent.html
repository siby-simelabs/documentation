import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { ManagerService } from 'src/app/providers/manager/manager.service';
import { BudgetService } from 'src/app/providers/project/budget/budget.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-assignvillages-add',
  templateUrl: './assignvillages-add.component.html',
  styleUrls: ['./assignvillages-add.component.scss'],
})
export class AssignvillagesAddComponent implements OnInit {
  crpId: any;
  locId: any;
  dist: any;
  blk: any;
  crpInfo: any;
  proName: any;
  datas: any = [];
  blkId: any | undefined;
  filterArray: any;
  panchayat: any;
  submitted: boolean = false;
  gram: any;
  villageData: any;
  role:any;

  constructor(
    public manager: ManagerService,
    public common: CommonService,
    public route: Router,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public project: ProjectService,
    public budget: BudgetService,
    public spinner: NgxSpinnerService
  ) { }
  filterForm = this.formBuilder.group(
    {
      gramapanchayat: [null]
    },
  );
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }

  ngOnInit(): void {
    this.access();
    this.crpId = this.common.getCrpId();
    this.villageData = history.state.data;
    console.log(this.villageData)
    this.locId = history.state.locationId;
    this.blkId = this.common.getLocal("blockId")
    this.dist = history.state.districtName;
    this.blk = history.state.blockName;
    
    if (this.blkId == undefined || null) {
      this.route.navigate(['/dashboard/crp/assignvillages']);
      this.common.showError("Select a Village")
    }
    this.project.getProjectByID(this.crpId).subscribe((resp) => {
      if (resp) {
        
        this.crpInfo = resp.data;
        this.proName = resp.data?.name;
      }
    });
    this.getVillList();
  }
  access() {
    this.role = this.common.screenRole("assignvillage");
  }
  getVillList() {
    this.spinner.show();
    this.manager.getAllVillById(this.crpId, this.blkId).subscribe(resp => {
      this.spinner.hide();
      
      if (resp) {
        this.datas = resp.data?.villageModels;
        this.filterArray = resp.data?.villageModels;
        let countData:number = 0;
        let truData = [];
        for(let i=0; i<this.filterArray?.length; i++){
          for(let j=0; j<this.filterArray[i]?.villageListModelList.length; j++){
            if(this.filterArray[i].villageListModelList[j].selected == true){
              truData.push(this.filterArray[i].villageListModelList[j])
            }
          }
        }
        countData = truData.length;
        this.villageData.noOfVillagesAlreadyIdentified = countData;

        this.common.trigger('updateHeaderDetails',[{
          name : 'Project ID' ,
          value : this.crpInfo.id 
        },{
          name : "Project Name" ,
          value : this.crpInfo.name
        },{
          name : 'State' ,
          value : this.villageData.stateName 
        },{
          name : 'District' ,
          value : this.villageData.districtName 
        },{
          name : 'Block' ,
          value : this.villageData.blockName 
        }])
        
      }
    })
  }
  getVillageLength(villages: any) {
    let counter = 0;
    for (let i = 0; i < villages.length; i++) {
      if (villages[i].selected == true) {
        counter++
      }
    }
    return counter;
  }
  addVillage() {
    this.route.navigateByUrl('/dashboard/crp/add-villages', { state: { data:this.villageData, blockid: this.blkId, dist: this.dist, blk: this.blk } });
  }
  edit(villages: any) {
    this.route.navigateByUrl('/dashboard/crp/add-villages', { state: { data:this.villageData, blockid: this.blkId, villages: villages, flag: "edit", dist: this.dist, blk: this.blk } });
  }
  view(villages: any) {
    this.route.navigateByUrl('/dashboard/crp/add-villages', { state: { data:this.villageData, blockid: this.blkId, villages: villages, flag: "view", dist: this.dist, blk: this.blk } });
  }
  delete(villages: any) {
    let deleteVill = [];
    for (let i = 0; i < villages.length; i++) {
      if (villages[i].selected == true) {
        deleteVill.push(villages[i]);
        
      }
    }
    let delData = {
      "villageListModelList": deleteVill,
      "projectId": this.crpId,
    }
    if (confirm('Are you sure you want to delete this record?')) {
      this.manager.deletVillageList(delData).subscribe(resp => {
        
        if (resp) {
          this.common.showSuccess("Successfully Deleted");
          this.getVillList();
        }
      })
    }
  }
  onSubmit(value: any): void {
    this.submitted = true;
    if (this.filterForm.invalid) {
      return;
    }
    else {
      const filterByGram = this.filterArray.filter((val: { gramapanchayat: any; }) => val.gramapanchayat == this.gram);
      this.datas = filterByGram;
      
    }
    this.common.closeActiveFilter();
  }
  clearFilter() {
    this.getVillList();
    this.datas = this.filterArray;
    this.filterForm.reset();
    this.panchayat = '';
    this.common.closeActiveFilter();
  }
}
