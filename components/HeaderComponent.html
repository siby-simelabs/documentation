import { Component, OnInit } from '@angular/core';
import { UserService } from 'src/app/providers/user.service';
import { TranslateService } from '@ngx-translate/core';
import { updateDefaultLanguage } from 'src/app/_helpers/language';
@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss'],
})
export class HeaderComponent implements OnInit {
  isShowDiv = false;
  username: any;
  constructor(public user: UserService, public translate: TranslateService) {
  }

  ngOnInit(): void {
    if (history.state.flag) {
      location.reload();
    }
    if (this.user.hasLoggedIn) {
      this.username = this.user.profile.userName;
    }
  }
  handleLanguageChange(value: string) {
    if (value && value.trim()) {
      this.translate.setDefaultLang(value);
      updateDefaultLanguage(value);
    }
  }

  showUser() {
    this.isShowDiv = !this.isShowDiv;
  }

  logout() {
    this.user.logout();
  }
}
