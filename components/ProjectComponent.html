import { Component, OnInit } from '@angular/core';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Subject } from 'rxjs';
import { CommonService } from 'src/app/providers/common.service';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import {
  AbstractControl,
  FormBuilder,
  FormGroup,
  Validator,
  Validators,
} from '@angular/forms';

import { UserService } from 'src/app/providers/user.service';
import { TranslateService } from '@ngx-translate/core';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
@Component({
  selector: 'app-project',
  templateUrl: './project.component.html',
  styleUrls: ['./project.component.scss'],
})
export class ProjectComponent implements OnInit {
  projects: any = [];
  currentPage = 1;
  pageNo: number = 1;
  pageSize: number = 10;
  prCount: number = 0;
  minDate: any;
  submitted: boolean = false;
  sector: any;
  stateList: any;
  blockList: any;
  districtList: any;
  stateId: number = 0;
  projectStatus: any;
  searchData: any = null;
  role: any;
  viewPr: boolean = true;
  editPr: boolean = true;
  deletePr: boolean = true;
  addPr: boolean = true;
  placement: TypeDatePickerPlacement = 'bottom';
  constructor(
    public project: ProjectService,
    public common: CommonService,
    public route: Router,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder,
    private activatedRoute: ActivatedRoute,
    public user: UserService,
    public translate: TranslateService,
    public dp: DatePickerPlacement
  ) {}
  filterForm = this.formBuilder.group({
    projectId: [null],
    nameOfProject: [null],
    status: [null],
    durationFrom: [null],
    durationTo: [null],
    interventionName: [null],
    sector: [null],
    state: [null],
    district: [null],
    village: [null],
  });
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }

  ngOnInit(): void {
    this.access();

    this.common.clearProjectId();
    this.common.clearProjectViewId();

    this.getProjects();

    this.common.getMasterData().subscribe((resp) => {
      if (resp) {
        this.projectStatus =
          resp.data.projectStatus.ProjectStatus.toString().split(',');
      }
    });
    this.common.getState().subscribe((resp) => {
      if (resp) {
        this.stateList = resp.data;
      }
    });
    this.common.getAllSectors().subscribe((resp) => {
      if (resp) {
        this.sector = resp.data;
      }
    });
  }
  access() {
    this.role = this.common.screenRole('project_profile');
  }
  getProjects() {
    this.projects = [];
    this.spinner.show();
    this.project.getProject(this.pageNo, this.pageSize).subscribe((resp) => {
      if (resp) {
        this.spinner.hide();
        this.projects = resp.data?.projectList;
        this.prCount = resp.data?.totalCount;
        if (resp.data == null) {
          this.prCount = 0;
          this.projects = [];
        }
      }
    });
  }
  pageChanged(event: any) {
    if (event.page != this.pageNo) {
      this.pageNo = event.page;
      this.pageSize = event.itemsPerPage;
      if (this.searchData) {
        this.searchData.pageNumber = event.page;
        this.searchFile(this.searchData);
      } else {
        this.getProjects();
      }
    }
  }
  _stateUserStatus(status: any) {
    if (status == 'Submission by state') {
      return false;
    } else if (
      status == 'L1 scrutiny' ||
      status == 'Appraisal Completed' ||
      status == 'PC Approved' ||
      status == 'EC Approved' ||
      status == 'Final Approved and locked' ||
      status == 'Rejected' ||
      status == 'Project Closed' ||
      status == 'Reject to merge proposal' ||
      status == 'Unlocked for centre for edit'
    ) {
      return false;
    } else {
      return true;
    }
  }
  _centralUser(status: any, type: any) {
    if (type == 'edit') {
      if (status != 'Unlocked for centre for edit') {
        return false;
      } else {
        return true;
      }
    } else if (type == 'delete') {
      return false;
    } else {
      return false;
    }
  }
  _commonUser(status: any, type: any) {
    if (type == 'edit') {
      if (status != 'Final Approved and locked') {
        return true;
      } else {
        return false;
      }
    } else {
      return false;
    }
  }
  editStatus(status: any, type: any) {
    if (
      this.user.profile.roleName == 'pm_state_user_level1' ||
      this.user.profile.roleName == 'pm_state_user_level2'
    ) {
      return this._stateUserStatus(status);
    } else if (this.user.profile.roleName == 'pm_central_user_level2') {
      return this._centralUser(status, type);
    } else if (
      this.user.profile.roleName == 'pm_central_user_level1_final_approval'
    ) {
      return this._commonUser(status, type);
    } else {
      return true;
    }
  }

  editProject(projId: any) {
    this.common.storeProjectId(projId);
    this.route.navigate(['/dashboard/project/edit-profile']);
  }
  deleteProject(projId: any) {
    this.spinner.show();
    if (confirm('Are you sure you want to delete this record?')) {
      this.project.deleteProject(projId).subscribe((resp) => {
        this.spinner.hide();
        if (resp.message == 'SUCCESS') {
          this.common.showSuccess('Successfully Deleted');
          this.getProjects();
        } else {
          this.common.showError(resp.message);
        }
      });
    } else {
      this.spinner.hide();
    }
  }
  onSubmit(value: any): void {
    this.pageNo = 1;
    this.pageSize = 10;
    this.currentPage = 1;
    this.submitted = true;
    if (this.filterForm.invalid) {
      return;
    } else {
      this.searchData = {
        pageNumber: this.pageNo,
        pageSize: this.pageSize,
        searchModel: {
          projectId: value['projectId'],
          nameOfProject: value['nameOfProject'],
          status: value['status'],
          durationFrom:
            (<HTMLInputElement>document.getElementById('durationFrom')).value ||
            null,
          durationTo:
            (<HTMLInputElement>document.getElementById('durationTo')).value ||
            null,
          interventionName: value['interventionName'],
          sector: value['sector'],
          state: value['state'],
          district: value['district'],
          village: value['village'],
        },
      };
      this.searchFile(this.searchData);
    }
  }
  searchFile(data: any) {
    this.spinner.show();
    this.project.filterProjects(data).subscribe((resp) => {
      this.project.trigger('onFilterApply');
      this.spinner.hide();
      if (resp) {
        this.projects = resp.data?.projectList;
        this.prCount = resp.data?.totalCount;
      }
    });
  }
  viewProject(id: any) {
    this.common.storeProjectViewId(id);
    this.common.storeProjectId(id);
    this.route.navigate(['/dashboard/project/edit-profile']);
  }
  sectorComa(sector: any) {
    if (sector != null) {
      return sector.toString().replace(/,/g, ', ');
    }
  }
  trimString(string: string, value: number) {
    let str = '';
    str = string;
    return str.substring(0, value);
  }
  onValueChange(event: any) {
    this.filterForm.controls['durationTo'].clearValidators();
    this.filterForm.get('durationTo')?.setValue('');
    this.minDate = event;
    if (event != undefined && event != '') {
      this.filterForm.controls['durationTo'].setValidators([
        Validators.required,
      ]);
      this.filterForm.controls['durationTo'].updateValueAndValidity({
        onlySelf: true,
        emitEvent: false,
      });
    }
  }

  stateChange(event: any) {
    this.blockList = [];
    this.districtList = [];
    this.filterForm.get('district')?.setValue('');
    this.filterForm.get('village')?.setValue('');
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.common.getDistrict(SelectVal).subscribe((resp) => {
      this.districtList = resp.data;
      this.stateId = SelectVal;
    });
  }
  distChange(event: any) {
    this.blockList = [];
    const SelectVal =
      event.target.options[event.target.selectedIndex].getAttribute('data-id');
    this.common.getBlock(SelectVal, this.stateId).subscribe((resp) => {
      this.blockList = resp.data;
    });
  }
  clearFilter() {
    this.pageNo = 1;
    this.pageSize = 10;
    this.currentPage = 1;
    this.getProjects();
    this.filterForm.reset();
    this.searchData = null;
    this.districtList = [];
    this.blockList = [];
    this.project.trigger('onFilterApply');
  }
}
