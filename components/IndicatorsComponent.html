import { Component, OnInit, Renderer2, ViewChild } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { Subject } from 'rxjs';
import { CommonService } from 'src/app/providers/common.service';

import { IndicatorsService } from 'src/app/providers/project/indicators/indicators.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-indicators',
  templateUrl: './indicators.component.html',
  styleUrls: ['./indicators.component.scss'],
})
export class IndicatorsComponent implements OnInit {
  @ViewChild('toggleButton') toggleButton: any;
  @ViewChild('filterContainer') filterContainer: any;
  projectId: any;
  pageNo: number = 1;
  pageSize: number = 100;
  datas: any;
  projectInfo: any;
  vewId: any = '';
  prCount: any;
  keyword: any = 'name';
  searchResult: any;
  selecteData: any;
  indId: any;
  filterForm: FormGroup;
  role: any;
  submitted: boolean = false;
 
  constructor(
    private renderer: Renderer2,
    public ind: IndicatorsService,
    public common: CommonService,
    public route: Router,
    public project: ProjectService,
    public spinner: NgxSpinnerService,
    public formBuilder: FormBuilder
  ) {
    this.filterForm = this.formBuilder.group({
      intervention: [''],
    });
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId();
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError('Please Select/Create a Project');
    }
  }
  ngOnInit(): void {
    this.access();
   
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    });
    this.getInd();
  }

  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }

  onSubmit(value: any): void {
    this.common.trigger('closeFilter');
    this.submitted = true;
    if (this.filterForm.invalid) {
      this.spinner.hide();
      return;
    } else {
      let data: any = {
        name: this.filterForm.getRawValue().intervention,
        projectId: this.projectId,
      };
      data = JSON.stringify(data, null, 2);
      this.ind.indiByName(data).subscribe((resp) => {
        this.project.closeActiveFilter();
        if (resp) {
          this.datas = resp.data;
          if (this.datas == null) this.datas = [];
        }
      });
    }
  }
  access() {
    this.role = this.common.screenRole('indicators_measures');
  }

  getInd() {
    this.spinner.show();
    this.ind
      .getIndicators(this.pageNo, this.pageSize, this.projectId)
      .subscribe((resp) => {
        this.spinner.hide();
        if (resp) {
          this.datas = resp.data.indicatorsModelList;
          this.prCount = resp.data.totalCount;
         }
      })
  }
 
  selectEvent(event: any) {
    this.selecteData = event;
  }

  onCleared(event: any) {
    this.getInd();
    this.searchResult = [];
  }
  add(indId: any, outId: any) {
    this.route.navigateByUrl('/dashboard/project/edit-indicators', {
      state: { id: indId, outId: outId },
    });
  }

  edit(indId: any, outId: any) {
    this.route.navigateByUrl('/dashboard/project/edit-indicators', {
      state: { id: indId, outId: outId },
    });
    return ;
  }
  view(indId: any, outId: any) {
    this.route.navigateByUrl('/dashboard/project/edit-indicators', {
      state: { id: indId, outId: outId, flag: 'view' },
    });
  }
  delete(indId: any) {
    this.spinner.show();
    if (
      confirm('Are you sure you want to delete this record?')
    ) {
      this.ind.delete(indId).subscribe((resp) => {
        this.spinner.hide();
        if (resp) {
          this.common.showSuccess('Successfully Deleted');
          this.getInd();
        }
      });
    } else {
      this.spinner.hide();
    }
  }

  clearFilter() {
    this.common.trigger('closeFilter');
    this.datas = [];
    this.getInd();
    this.filterForm.reset();
    this.project.closeActiveFilter();
  }
}
