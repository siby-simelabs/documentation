import { Component, OnInit } from '@angular/core';
import { AdminService } from '../admin.service';
import { BsModalService } from 'ngx-bootstrap/modal';
import { AssignUserRoleComponent } from 'src/app/shared/common/assign-user-role/assign-user-role.component';
@Component({
  selector: 'app-users',
  templateUrl: './users.component.html',
  styleUrls: ['./users.component.scss'],
})
export class UsersComponent implements OnInit {
  users: any = [];
  bsModalRef: any;
   pageNo: number = 1;
  currentPage = 1;
  pageSize: number = 10;
  total: any;
  constructor(private admin: AdminService, private modal: BsModalService) {}

  ngOnInit(): void {
    this.fetchUsers();
  }
  fetchUsers() {
    this.admin.getUsers(this.pageNo, this.pageSize).subscribe((res) => {
      this.users = res.data.userList;
      this.total = res.data.totalCount;
    });
  }
  changeRole($event: any, id: number) {
    let user = this.users.find((item: any) => {
      return item.id == id;
    });
    let initialState: any = {
      user,
      isEditing: true,
    };
    this.showPopup(initialState);
  }
  addUser() {
    this.showPopup({
      isEditing : false 
    });
  }
  showPopup(_initialState: any = {}) {
    let initialState: any = {
      ..._initialState
    };
    this.bsModalRef = this.modal.show(AssignUserRoleComponent, {
      initialState,
    });
    this.bsModalRef.content.event.subscribe((res: any) => {
      if (res.close) {
        this.modal.hide();
      }
      if (res.created) {
        this.fetchUsers();
      }
    });
  }
  pageChanged(event: any) {
    if (event.page != this.pageNo) {
      this.pageNo = event.page;
      this.pageSize = event.itemsPerPage;
      this.fetchUsers();
    }
    

  }
}
