import { HttpClient } from '@angular/common/http';
import { Component, Input, OnInit, OnChanges } from '@angular/core';
import { Router } from '@angular/router';
import * as $ from 'jquery';
import { Api } from 'src/app/providers/api.service';
import { CommonService } from 'src/app/providers/common.service';
import { UserService } from 'src/app/providers/user.service';

@Component({
  selector: 'app-nav',
  templateUrl: './nav.component.html',
  styleUrls: ['./nav.component.scss']
})
export class NavComponent implements OnInit {
  project: boolean = false;
  admin: boolean = false;
  kpiconfig: boolean = false;
  adminvalue: boolean = false;
  crp: boolean = false;
  assign: boolean = false;
  fund: boolean = false;
  utilization: boolean = false;
  track: boolean = false;
  report: boolean = false;
  key: boolean = false;
  role: boolean = false;
  userpage: boolean = false;
  screensPrmsn: any;
  load: boolean = false;
  permissons: any = [];
  roleName:string | undefined;
  @Input() pageData: any;
  constructor(public route: Router, public common: CommonService, public user: UserService, public http: HttpClient, private api: Api) {

  }

  ngOnInit(): void {
    this.getScreen();
    this.RoutingUrl();
    this.common.subscribe( this.updateSidebar , "updateSidebar" );
  }
  ngOnDestroy(){
    this.common.unsubscribe(this.updateSidebar , "updateSidebar" );
  }
  updateSidebar = ()=>{
    this.RoutingUrl();
  }
  getScreen() {
    let endpoint = 'auth/user/getRoleByToken';
    this.api.get(endpoint).subscribe((data) => {
      this.screensPrmsn = data?.data?.screen;
      this.roleName = data?.data?.name;
      this.load = true;
     });
  }
  getScreenPermis(screen: string) {
    for (let i = 0; i < this.screensPrmsn?.length; i++) {
      if (this.screensPrmsn[i].name === screen) {
        return this.screensPrmsn[i].auth;
      }
    }
  }
  onClickHandler() {
    this.common.trigger("reset_role_page");
    this.RoutingUrl();
  }
  RoutingUrl() {
    setTimeout(() => {
      this.project = false;
      this.crp = false;
      this.admin = false;
      this.kpiconfig = false;
      this.adminvalue = false;
      this.assign = false;
      this.fund = false;
      this.utilization = false;
      this.track = false;
      this.report = false;
      this.key = false;
      this.role = false;
      this.userpage = false;
      if (window.location.href.indexOf("project") > -1) {
        this.project = true;
      }
      if (window.location.href.indexOf("crp/") > -1) {
        this.crp = true;
      }
      if (window.location.href.indexOf("assigntarget") > -1) {
        this.assign = true;
      }
      if (window.location.href.indexOf("fund") > -1) {
        this.fund = true;
      }
      if (window.location.href.indexOf("utilization") > -1) {
        this.utilization = true;
      }
      if (window.location.href.indexOf("track-kpi") > -1) {
        this.track = true;
      }
      if ((window.location.href.indexOf("kpi-config") > -1) || (window.location.href.indexOf("outcome") > -1) || (window.location.href.indexOf("event") > -1)) {
        this.kpiconfig = true;
      }
      if ((window.location.href.indexOf("master-values") > -1) || (window.location.href.indexOf("details") > -1) || (window.location.href.indexOf("ddl") > -1)) {
        this.adminvalue = true;
      }
      if ((window.location.href.indexOf("reports") > -1)) {
        this.report = true;
      }
      if ((window.location.href.indexOf("key-indicator") > -1)) {
        this.key = true;
      }
      if ((window.location.href.indexOf("roles") > -1)) {
        this.role = true;
      }
      if ((window.location.href.indexOf("users") > -1)) {
        this.userpage = true;
      }
    }, 200);


  }
  ngAfterContentInit() {
    this.RoutingUrl()
  }

}
