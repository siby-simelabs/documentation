import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';

@Component({
  selector: 'app-target-tab',
  templateUrl: './target-tab.component.html',
  styleUrls: [ ]
})
export class TargetTabComponent implements OnInit {
 
  kpi: boolean = false;
  blktarget: boolean=false;
  crptarget: boolean=false;

  constructor(
    public route: Router
  ) { }

  ngOnInit(): void {
    this.RoutingUrl();
    
  }
  RoutingUrl(){
    setTimeout(()=>{
     
      this.blktarget=false;
      this.crptarget=false;
      
      if(
        this.route.url == '/dashboard/assigntarget/blocktarget' 
      ){
        this.blktarget=true;
      }
      if(
        this.route.url == '/dashboard/assigntarget/crptarget' 
      ){
        this.crptarget=true;
      }

    },100);
  }
  ngAfterContentInit(){
    this.RoutingUrl()
  }

}
