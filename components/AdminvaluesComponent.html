import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { CommonService } from 'src/app/providers/common.service';
import { ProjectService } from 'src/app/providers/project/project.service';

@Component({
  selector: 'app-adminvalues',
  templateUrl: './adminvalues.component.html',
  styleUrls: ['./adminvalues.component.scss']
})
export class AdminvaluesComponent implements OnInit {
  data:any;
  temp: any;
  master:any;
  role: any;
  constructor( public route: Router, public common:CommonService, public project:ProjectService) { }
  ngOnInit(): void {
    this.getAllMasters();
    this.access();
  }
   
  access() {
    this.role = this.common.screenRole("adminvalues");
    
  }
  edit(grp:any){
    this.route.navigateByUrl('/dashboard/admin/ddl-add/'+grp.id+'', { state: { data: grp, flag: "edit" } });
  }
  view(grp:any){
    this.route.navigateByUrl('/dashboard/admin/ddl/'+grp.id+'?view=true', { state: { data: grp, flag: "edit" } });
  }
  add(grp:any){
    this.route.navigateByUrl('/dashboard/admin/ddl/'+grp.id+'', { state: { data: grp, flag: "edit" } });
  }
  delete(data: any) {
    if (confirm("Are you sure you want to delete this record?") == true) {
      let deleteId = {"id":data.id}
      this.common.deleteMaster(deleteId, "deleteMaster").subscribe((resp) => {
        if (resp) {
          this.getAllMasters();
        }
      });
    }
  }
  getAllMasters(){
    this.common.getAllMaster().subscribe((resp) => {
      if (resp) {
        
        this.data = resp.data.sort((a: { id: number; },b: { id: number; }) => b.id - a.id);
        this.temp = this.data;
      }
    });
  }
  filter(){
    if(this.master){
      this.data = this.find(this.temp, this.master?.toLowerCase());
      this.project.trigger('onFilterApply');
    }
  }
  clearFilter(){
    this.master = "";
    this.data = this.temp;
    this.project.trigger('onFilterApply');
  }
  find(items: any, text: any) {
    text = text.split(' ');
    return items.filter((item: { masterValue: any | any[]; }) => {
      return text.every((el: any) => {
        let result = item.masterValue;
        let caps = result.toLowerCase()
        return caps?.includes(el);
      });
    });
  }
  getStatus(status:boolean){    
    if(status == true){
      return "Active";
    }
    else{
      return "Inactive";
    }
  }

}
