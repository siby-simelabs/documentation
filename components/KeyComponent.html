import { Component, OnInit } from '@angular/core';
import { AbstractControl, FormBuilder, Validators } from '@angular/forms';
import { TypeaheadMatch } from 'ngx-bootstrap/typeahead';
import { NgxSpinnerService } from 'ngx-spinner';
import { Api } from 'src/app/providers/api.service';
import { CommonService } from 'src/app/providers/common.service';
import { IndicatorsService } from 'src/app/providers/project/indicators/indicators.service';
import { InterventionService } from 'src/app/providers/project/interventions/intervention.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { UtilizationService } from 'src/app/providers/utilization/utilization.service';
import { environment as env } from '../../../../../../environments/environment.dev';

@Component({
  selector: 'app-key',
  templateUrl: './key.component.html',
  styleUrls: ['./key.component.scss']
})
export class KeyComponent implements OnInit {

  submitted: boolean = false;
  nameOfProject: any;
  projects: any;
  interList: any;
  role: any;
  outcomeData: any;
  interVentionId: any;
  activityList: any;
  projectStatus: any = [];
  status: string = "Final Approved and locked";


  constructor(
    public formBuilder: FormBuilder,
    public project: ProjectService,
    public api: Api,
    public inter: InterventionService,
    public common: CommonService,
    public ind: IndicatorsService,
    public spinner: NgxSpinnerService,
    public utilization: UtilizationService
  ) { }
  filterForm = this.formBuilder.group(
    {
      projectId: [null, Validators.required],
      nameOfProject: [null, Validators.required],
      interventionName: [null, Validators.required],
      interId: [null, Validators.required],
      activity: [null, Validators.required],
      activityid: [null, Validators.required],
      status: ["Final Approved and locked", Validators.required]
    },
  );
  get f(): { [key: string]: AbstractControl } {
    return this.filterForm.controls;
  }

  ngOnInit(): void {
    this.access();
  }
  access() {
    this.role = this.common.screenRole("key_report");

  }
  onSubmit(value: any): void {
    this.submitted = true;
    if (this.filterForm.invalid) {
      return;
    } else {

      this.getReport(value["projectId"], value["interId"], value["activityid"])
    }
  }
  getReport(prid: string, interId: string, actId: number) {
    this.spinner.show();
    let url = "project/kpiReport?projectId=" + prid + "&interventionId=" + interId + "&activityId=" + actId + "";
    let filename = "KPI_Report.xlsx";
    this.download(url, filename, prid);
  }
  download(url: string, docname:string, value?:any) {
    this.common.dwnldFile(url, docname);
    this.spinner.hide();
  }
  onSelect(event: any): void {

    this.filterForm.get('projectId')?.setValue(event.item.id);
    this.getAllIntervention(event.item.id)
  }
  nameChnage(event: any) {
    this.filterForm.get('projectId')?.setValue(null);

    let name = event.target.value;
    if (event.target.value == "") { name = null };
    this.getProjectByStatus(name);
  }

  onReset(): void {
    this.submitted = false;
    this.filterForm.reset();
  }
  getAllIntervention(prId: string) {
    this.ind.getIndicators(1, 1000, prId).subscribe((resp) => {
      if (resp) {
        this.interList = resp.data?.indicatorsModelList

      }
    });
  }
  interSelect(event: any) {

    this.filterForm.get('interId')?.setValue(event.item.interventionId);
    this.interVentionId = event.item.interventionId;
    this.getActivity(event.item.kpiOutcomeModel.id, event.item.interventionId)
  }

  getActivity(outId: any, interId: any) {
    let data =
    {
      "pageNumber": 1,
      "pageSize": 100,
      "searchModel": {
        "outcome": null,
        "category": null,
        "subCategory": null,
        "interventionId": this.interVentionId,
        "outcomeId": outId
      }
    }
    this.ind.filterIndicator(data).subscribe((resp) => {
      if (resp) {

        this.activityList = resp?.data;
      }
    });
  }

  activtySelect(event: any) {

    this.filterForm.get('activityid')?.setValue(event.item.activityId);

  }

  statusChange(value: string) {
    this.status = value;
    this.filterForm.patchValue({ 'nameOfProject': null, 'projectId': null, 'interventionName': null, 'interId': null, 'activity': null, 'activityid': null });
    this.projects = [];
  }
  getProjectByStatus(value: string) {
    let searchData =
    {
      "searchModel": {
        "projectId": null,
        "nameOfProject": value,
        "status": this.status,
        "sector": null
      },
      "pageNumber": 1,
      "pageSize": 100
    }
    this.utilization.filterProject(searchData).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.projects = resp.data?.fundProjectModelsList;
      }
    })
  }

}
