import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormControl,
  FormGroup,
  Validators,
  FormBuilder,
} from '@angular/forms';
import { Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonService } from 'src/app/providers/common.service';
import { DocumentService } from 'src/app/providers/project/document/document.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { DatePickerPlacement } from 'src/app/_helpers/datePickerPlacement';
@Component({
  selector: 'app-documents-add',
  templateUrl: './documents-add.component.html',
  styleUrls: ['./documents-add.component.scss'],
})
export class DocumentsAddComponent implements OnInit {
  projectId: any;
  projectInfo: any;
  vewId: any;
  docAddForm: FormGroup;
  pageview: any;
  submitted: boolean = false;
  minDate: any;
  maxDate: any;
  fileData: any;
  parms: any;
  docs: any;
  documentFile: any;
  qtrData: any;
  code='en';
  doctype: any;
  placement : TypeDatePickerPlacement= 'bottom' ;
  constructor(
    public doc: DocumentService,
    public common: CommonService,
    public formBuilder: FormBuilder,
    public route: Router,
    public project: ProjectService,
    public spinner: NgxSpinnerService ,
    public dp :DatePickerPlacement
  ) {
    this.projectId = this.common.getProjectId();
    this.vewId = this.common.getProjecViewtId();
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError('Please Select/Create a Project');
    }
    this.docAddForm = this.formBuilder.group({
      name: ['', Validators.required],
      type: ['', Validators.required],
      qtr: [''],
      docDate: ['', Validators.required],
      description: [''],
      documentFile: ['', Validators.required],
    });
  }

  ngOnInit(): void {
    this.parms = history.state.id;
    this.pageview = history.state.flag;
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.minDate = this.dateForMindate(this.projectInfo.proposedStartDate);
        this.maxDate = this.dateForMindate(this.projectInfo.proposedEndDate);
        this.common.trigger('updateHeaderDetails', [
          { name: 'Project ID', value: resp.data.id },
          { name: 'Project Name', value: resp.data.name },
        ]);
      }
    });

    if (this.parms) {
      this.spinner.show();
      this.doc.edit(this.parms).subscribe((resp) => {
        if (resp) {
          this.spinner.hide();
          this.docs = resp.data;
          this.docAddForm.get('name')?.setValue(this.docs.name);
          this.docAddForm.get('type')?.setValue(this.docs.type);
          this.docAddForm.get('qtr')?.setValue(this.docs.qtr);
          this.docAddForm.get('docDate')?.setValue(this.docs.docDate);
          this.docAddForm.get('description')?.setValue(this.docs.description);
          this.docAddForm.get('documentFile')?.setValue(this.docs.documentFile);
        }
      });
    }
    this.doc.getDocType(this.code).subscribe((resp) => {
      if(resp){
        this.doctype = resp.data;
      }
    })
    this.getQuarter();
  }

  dateForMindate(date: any) {
    if (!date || typeof date !== 'string') return;
    let datearray = date.split('/');
    let newdate = datearray[1] + '/' + datearray[0] + '/' + datearray[2];

    return new Date(newdate);
  }
  get f(): { [key: string]: AbstractControl } {
    return this.docAddForm.controls;
  }

  onSubmit(value: any) {
    this.submitted = true;
    if (this.parms) {
      this.docAddForm.controls['documentFile']?.clearValidators();
      this.docAddForm.controls['documentFile']?.updateValueAndValidity();
    }
    if (this.docAddForm.invalid) {
      return;
    } else {
      let docdata = {
        name: value['name'],
        type: value['type'],
        qtr: value['qtr'],
        description: value['description'],
        docDate: (<HTMLInputElement>document.getElementById('docDate')).value,
        projectId: this.projectId,
        id: this.parms || null,
      };
      this.saveDocs(docdata);
    }
  }
  saveDocs(data: any) {
    this.spinner.show();
    let formData: any = new FormData();
    if (this.fileData) {
      formData.append('documentFile', this.fileData, this.fileData.name);
    } else {
      formData.append('documentFile', null);
    }
    formData.append(
      'document',
      new Blob([JSON.stringify(data, null, 2)], {
        type: 'application/json',
      })
    );
    this.doc.saveDoc(formData).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.common.showSuccess('Successfully Created');
        this.route.navigate(['/dashboard/project/documents']);
      }
    });
  }
  onReset(): void {
    this.submitted = false;
    this.docAddForm.reset();
  }
  onChange(event: any) {
    if (event.target.files[0].size >= this.common.maxFileSize) {
      this.common.showError(
        'File size too large. Please upload a file less than 50mb'
      );
      this.docAddForm.get('documentFile')?.setValue(null);
      return;
    } else if (
      event.target.files[0].name.length >= this.common.maxFileNameLength
    ) {
      this.common.showError(
        'File name length restricted to maximum ' +
          this.common.maxFileNameLength +
          ' characters'
      );
      this.docAddForm.get('documentFile')?.setValue(null);
      return;
    } else {
      this.fileData = event.target.files[0];
    }
  }
  getQuarter() {
    this.common.getQuarterData().subscribe((resp) => {
      if (resp) {
        this.qtrData = resp.data;
      }
    });
  }
  download(url: string, docname: string) {
    this.common.dwnldFile(url, docname);
  }
}
