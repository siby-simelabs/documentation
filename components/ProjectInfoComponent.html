import { Component, OnInit, Input } from '@angular/core';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Router } from '@angular/router';
import { CommonService } from 'src/app/providers/common.service';
@Component({
  selector: 'app-project-info',
  templateUrl: './project-info.component.html',
  styleUrls: ['./project-info.component.scss'],
})
export class ProjectInfoComponent {
  @Input() isEditing: boolean = false;
  id: any = '';
  name: any = null;
  state: string = '';
  tabs: any = this.common.getLocalProject();
  constructor(
    private project: ProjectService,
    private route: Router,
    private common: CommonService
  ) {
    this.common.subscribe(this.onTabChange, 'updateHeaderDetails');
  }
  onTabChange = (tabs: any) => {
    this.tabs = tabs
      .filter((item: any) => {
        if (typeof item.value === 'string') {
          return item.value && item.value.trim();
        } else {
          if (Array.isArray(item.value) && item.value[0]) {
            return true;
          }
          return false;
        }
      })
      .map((item: any) => {
        if (Array.isArray(item.value)) {
          return {
            ...item,
            value: item.value[0],
          };
        }
        return item;
      });
  };
  onStateUpdate = (state: any) => {
    this.state = state;
  };
  ngOnDestroy() {
    this.common.unsubscribe(this.onTabChange, 'onStateUpdate');
  }

  showIdAndName() {
    return this.route.url !== '/dashboard/project/profile';
  }
  showDistrict() {
    return this.route.url === '/dashboard/project/geo-coverage';
  }
  isIndicatorsAndMeasures(): boolean {
    return false;
   
  }
  trimItem(item: string) {
    if (item.length > 15) {
      return item.substring(0, 10) + '...';
    } else {
      return item;
    }
  }
}
