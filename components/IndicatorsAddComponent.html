import { Component, OnInit, ChangeDetectorRef } from '@angular/core';
import {
  AbstractControl,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { NgxSpinnerService } from 'ngx-spinner';
import { combineLatest, Subject, Subscription } from 'rxjs';
import { CommonService } from 'src/app/providers/common.service';
import { IndicatorsService } from 'src/app/providers/project/indicators/indicators.service';
import { InterventionService } from 'src/app/providers/project/interventions/intervention.service';
import { ProjectService } from 'src/app/providers/project/project.service';
import { Validator } from 'src/app/_helpers/validators';
import { BsModalService, BsModalRef, ModalOptions } from 'ngx-bootstrap/modal';
import { EventAddComponent } from '../../../admin/kpiconfig/event/event-add/event-add.component';
import { OutcomeService } from 'src/app/providers/admin/kpiconfig/outcome/outcome.service';
import { AddtargetComponent } from '../addtarget/addtarget.component';

@Component({
  selector: 'app-indicators-add',
  templateUrl: './indicators-add.component.html',
  styleUrls: ['./indicators-add.component.scss'],
})
export class IndicatorsAddComponent implements OnInit {
  activityList: any = [];
  submitted = false;
  formCreateIndicators: FormGroup;
  projectId: any;
  vewId: any;
  projectInfo: any;
  name: any;
  indview: any;
  interventonData: any;
  parms: any;
  showkpibtn: any;
  OutcomeId: any;
  bsModalRef?: BsModalRef;
  subscriptions: Subscription[] = [];
  messages: string[] = [];
  ifOutcome: boolean = false;
  outcomeDetails: any;
  prCount: number = 0;
  outcomeList: any;
  activitycato: any;
  selectedOutcomeData: any;
  selectedOutId: number | undefined;

  constructor(
    public ind: IndicatorsService,
    public common: CommonService,
    public route: Router,
    public project: ProjectService,
    public spinner: NgxSpinnerService,
    public inter: InterventionService,
    public formBuilder: FormBuilder,
    public validator: Validator,
    public outcomeData: OutcomeService,
    private modalService: BsModalService,
    private changeDetection: ChangeDetectorRef
  ) {
    this.projectId = this.common.getProjectId();
    this.formCreateIndicators = this.formBuilder.group({
      outcome: [null, Validators.required],
      category: [null],
      subCategory: [null],
    });
    this.vewId = this.common.getProjecViewtId();
    if (!this.projectId) {
      this.route.navigate(['/dashboard/project/list']);
      this.common.showError('Please Select/Create a Project');
    }
  }

  ngOnInit(): void {
    this.showkpibtn = history.state.flag;
    this.OutcomeId = history.state.outId;
    this.parms = history.state.id;
    if (!this.parms) {
      this.common.showError('Choose an intervention from the list!');
      this.route.navigateByUrl('/dashboard/project/indicators', {
        state: { flag: history.state.flag },
      });
    }
    this.getProjectData();
    this.getInterventionData();
    if (history.state.flag != 'edit') {
      this.indview = history.state.flag;
    }
    if (this.OutcomeId != null) {
      this.outComeById(this.OutcomeId);
      this.gettargetList();

    }
  }
  getProjectData() {
    this.project.getProjectByID(this.projectId).subscribe((resp) => {
      if (resp) {
        this.projectInfo = resp.data;
        this.updateHeader();
      }
    });
  }
  getTabs() {
    let brName = 'Assign Outcome/KPI target to intervention';
    if (this.OutcomeId)
      brName = (this.indview === 'view' ? 'View' : 'Edit') + '/KPI target';
    let items: any = [
      {
        name: 'List of interventions & Assigned outcome',
        link: '/dashboard/project/indicators',
      },
      {
        name: brName,
        link: '/dashboard/project/edit-indicators',
      },
    ];
    return items;
  }
  getInterventionData() {
    this.inter.getInterById(this.parms).subscribe((resp) => {
      this.interventonData = resp.data;
      this.updateHeader();
      if (this.OutcomeId == null) {
        this.getOutcome();
      }
    });
  }
  updateHeader() {
    if (this.projectInfo && this.interventonData) {
      this.common.trigger('updateHeaderDetails', [
        { name: 'Project ID', value: this.projectInfo.id },
        { name: 'Project Name', value: this.projectInfo.name },
        { name: 'Int. Name', value: this.interventonData.name },
        { name: 'Sector', value: this.interventonData.sector },
        { name: 'Int. Category', value: this.interventonData.category },
      ]);
    }
  }
  outComeById(outcomeId: number) {
    this.outcomeData.getOutById(outcomeId).subscribe((resp) => {
      this.outcomeDetails = resp.data;
      this.formCreateIndicators
        .get('outcome')
        ?.setValue(this.outcomeDetails.outcome);
      if (this.outcomeDetails.outComeLevel == 'Project level') {
        this.getActivityCato(this.outcomeDetails.interventionCategoryId, 'pl');
      } else {
        this.getActivityCato(this.outcomeDetails.interventionCategoryId, 'pa');
      }
    });
  }
  gettargetList() {
    if (this.parms) {
      this.spinner.show();
      let outID;
      if (this.OutcomeId == null) {
        outID = this.selectedOutId;
      } else {
        outID = this.OutcomeId;
      }
      this.ind.getActivityByOutcomeId(this.parms, outID).subscribe((resp) => {
        this.spinner.hide();
        this.activityList = resp?.data;
        if (this.activityList == null) this.activityList = [];
        this.prCount = resp.data?.length;
        this.ifOutcome = true;
      });
    }
  }
  get f(): { [key: string]: AbstractControl } {
    return this.formCreateIndicators.controls;
  }

  onSubmit() {
    this.submitted = true;
    if (this.formCreateIndicators.invalid) {
      return;
    }
    let outID;
    if (this.OutcomeId == null) {
      outID = this.selectedOutId;
    } else {
      outID = this.OutcomeId;
    }
    let data = {
      pageNumber: 1,
      pageSize: 100,
      searchModel: {
        category: this.formCreateIndicators.get('category')?.value,
        subCategory: '',
        interventionId: this.parms,
        outcomeId: outID,
      },
    };
    this.filterdata(data);
  }
  filterdata(data: any) {
    this.spinner.show();
    this.activityList = [];
    this.ind.filterIndicator(data).subscribe(
      (resp) => {
        this.spinner.hide();
        if (resp) {
          this.activityList = resp?.data;
          if (this.activityList == null) this.activityList = [];
          this.ifOutcome = true;
          this.prCount = this.activityList?.length;
        }
      },
      (error: any) => {
        this.spinner.hide();
        this.common.showError('Network error! Please try again later.');
      }
    );
  }

  getOutcome() {
    this.spinner.show();
    let outcome = {
      pageNumber: 1,
      pageSize: 100,
      kpiOutcomeByCategoryModel: {
        interventionCategory: this.interventonData?.category,
        interventionSubCategory: this.interventonData?.subCategory,
      },
    };
    this.ind.getOutcomeByInter(outcome).subscribe((resp) => {
      this.spinner.hide();
      if (resp) {
        this.outcomeList = resp.data;
      }
    });
  }

  outChange(event: any) {
    let outId: number = parseInt(event.target.value);
    this.selectedOutId = outId;
    this.outcomeData.getOutById(outId).subscribe((resp) => {
      this.outcomeDetails = resp.data;
      if (this.outcomeDetails.outComeLevel == 'Project level') {
        this.getActivityCato(this.outcomeDetails.interventionCategoryId, 'pl');
      } else {
        this.getActivityCato(this.outcomeDetails.interventionCategoryId, 'pa');
      }
    });
  }
  getActivityCato(id: number, pl: string) {
    this.common.getActivityCato(id, pl).subscribe((resp) => {
      if (resp) {
        this.activitycato = resp.data;
      }
    });
  }

  saveTarget(id: any, outcomeId: any, name: any) {
    let flag = 'edit';
    if (this.indview) {
      flag = 'view';
    }
    const initialState: ModalOptions = {
      initialState: {
        outcomename: this.outcomeDetails.outcome,
        interId: this.parms,
        kpi: name,
        flag: flag,
        id: id,
        outcomeId: outcomeId,
      },
      class: 'modal-lg',
    };
    this.bsModalRef = this.modalService.show(AddtargetComponent, initialState);
    let _combine;
    if (this.bsModalRef?.onHide && this.bsModalRef?.onHidden) {
      _combine = combineLatest(
        this.bsModalRef.onHide,
        this.bsModalRef.onHidden
      ).subscribe(() => this.changeDetection.markForCheck());
    }
    if (this.bsModalRef?.onHide) {
      this.subscriptions.push(
        this.bsModalRef.onHide.subscribe((reason: string | any) => {
          console.log(reason)
          if(reason == "updated"){
            this.OutcomeId = outcomeId;
            this.formCreateIndicators.get('outcome')?.setValue(this.outcomeDetails.outcome);
          }
          this.gettargetList();
          
        })
      );
    }
    if (_combine) {
      this.subscriptions.push(_combine);
    }
  }

  clearFilter() {
    this.formCreateIndicators.reset();
    this.activityList = [];
  }
  openModalWithComponent() {
    let outID;
    if (this.OutcomeId == null) {
      outID = this.selectedOutId;
    } else {
      outID = this.OutcomeId;
    }
    const initialState: ModalOptions = {
      initialState: {
        flag: 'project',
        outcome: outID,
      },
      class: 'modal-lg',
    };
    this.bsModalRef = this.modalService.show(EventAddComponent, initialState);
    let _combine;
    if (this.bsModalRef?.onHide && this.bsModalRef?.onHidden) {
      _combine = combineLatest(
        this.bsModalRef.onHide,
        this.bsModalRef.onHidden
      ).subscribe(() => this.changeDetection.markForCheck());
    }
    if (this.bsModalRef?.onHide) {
      this.subscriptions.push(
        this.bsModalRef.onHide.subscribe((reason: string | any) => {
          this.onSubmit();
        })
      );
    }
    if (_combine) {
      this.subscriptions.push(_combine);
    }
  }
  unsubscribe() {
    this.subscriptions.forEach((subscription: Subscription) => {
      subscription.unsubscribe();
    });
    this.subscriptions = [];
  }
}
