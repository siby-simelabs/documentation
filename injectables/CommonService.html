  subscribe(callback: any, event: string) {
    let callbacks = this.callbacks[event] || [];
    if (
      callback &&
      typeof callback === 'function' &&
      callbacks.indexOf(callback) === -1
    ) {
      callbacks.push(callback);
      this.callbacks[event] = callbacks;
    }
  }
  get headerEvent() {
    return 'updateHeaderDetails';
  }
  trigger(event: string, param?: any) {
    if (param) {
      if (event === 'updateHeaderDetails') {
        this.project = param.filter((item: any) => {
          return ['Project Name', 'Project ID'].indexOf(item.name) > -1;
        });
      }
    }
    Array.isArray(this.callbacks[event]) &&
      this.callbacks[event].forEach((callback: any) => {
        callback(param);
      });
  }
  unsubscribe(callback: any, event: string) {
    let callbacks = this.callbacks[event] || [];
    if (callbacks.indexOf(callback) > -1) {
      let index = callbacks.indexOf(callback);
      this.callbacks[event] = callbacks.filter(
        (item: any, _index: number) => index !== _index
      );
    }
  }

  dwnldFile(url:string, name:string){
    this.downloadPdf(url).subscribe(resp => {
      let blob = new Blob([resp], {
        type: 'application/pdf'
      });
      let docext  = name.split('.').pop();
      let link = document.createElement('a');
      link.href = window.URL.createObjectURL(blob);
      let doctrimName = name.replace(/\.[^/.]+$/, '');
      doctrimName = doctrimName.replace(/[^a-zA-Z0-9]/g, "");
      link.download = ''+doctrimName+'.'+docext+'';
      link.click();
      window.URL.revokeObjectURL(link.href);
    })
