import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { setTime } from 'ngx-bootstrap/chronos/utils/date-setters';
import { Api } from '../api.service';

@Injectable({
  providedIn: 'root',
})
export class ProjectService {
  project: any = [];
  callbacks: any = {};
  state: any = '';
  constructor(public api: Api) {}

  getProject(pageNo: number, pageSize: number) {
    let endpoint = 'project/getAllProjectProfile?';
    endpoint += 'pageNo=' + pageNo + '&' + 'pageSize=' + pageSize;
    return this.api.get(endpoint);
  }
  updateState(name: any = '') {
    this.state = name;
    this.trigger('onStateUpdate', name);
  }
  addProject(data: any) {
    let endpoint = 'project/addProjectProfile';
    return this.api.postAsJson(endpoint, data);
  }
  searchProjectByName(data: any) {
    let endpoint = 'project/getProjectByName?projectName=' + data + '';
    return this.api.get(endpoint);
  }
  getProjectByID(id: any) {
    let endpoint = 'project/getProjectById?id=' + id + '';
    return this.api.get(endpoint);
  }
  getLocalProject() {
    let project = localStorage.getItem("projectId") ;
    console.log( project , this.project );
    if( project && this.project.id === project ){
      return this.project;
    }
    else{
      this.project = null ;
    }
    return null ;
    
  }
  deleteProject(id: string) {
    let endpoint = 'project/deleteProjectProfile?id=' + id + '';
    return this.api.get(endpoint);
  }
  filterProjects(data: any) {
    let endpoint = 'project/filterProjectProfile';
    return this.api.post(endpoint, data);
  }
  clearProject() {
    this.project = [];
  }
  fundReport(id: string) {
    let endpoint = 'project/fundReport?projectId=' + id + '';
    return this.api.get(endpoint);
  }
  subscribe(callback: any, event: string) {
    let callbacks = this.callbacks[event] || [];
    if (
      callback &&
      typeof callback === 'function' &&
      callbacks.indexOf(callback) === -1
    ) {
      callbacks.push(callback);
      this.callbacks[event] = callbacks;
    }
  }
  trigger(event: string, param?: any) {
    if (param) {
      if (event === 'updateHeaderDetails') {
        this.project = param.filter((item: any) => {
          return ['Project Name', 'Project ID'].indexOf(item.name) > -1;
        });
      }
    }
    Array.isArray(this.callbacks[event]) &&
      this.callbacks[event].forEach((callback: any) => {
        callback(param);
      });
  }
  unsubscribe(callback: any, event: string) {
    let callbacks = this.callbacks[event] || [];
    if (callbacks.indexOf(callback) > -1) {
      let index = callbacks.indexOf(callback);
      this.callbacks[event] = callbacks.filter(
        (item: any, _index: number) => index !== _index
      );
    }
  }
  closeActiveFilter() {
    this.trigger('closeFilter');
  }
  getMode(code: any) {
    let endpoint = 'admin/master/getModeOfProject?code='+code;
    return this.api.get(endpoint);
  }
  getFunding(code: any) {
    let endpoint = 'admin/master/getFundingSource?code='+code;
    return this.api.get(endpoint);
  }
}
