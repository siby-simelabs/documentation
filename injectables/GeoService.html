import { Injectable } from '@angular/core';
import { Api } from '../../api.service';

@Injectable({
  providedIn: 'root'
})
export class GeoService {

  constructor(public api: Api) { }
  getGeoCoverage(pageNo: number, pageSize: number, projectId: any) {
    let endpoint = 'project/geoCoverage/getAllGeoCoverage?';
    endpoint +=
      'pageNo=' +
      pageNo +
      '&' +
      'pageSize=' +
      pageSize +
      '&projectId=' +
      projectId;
    return this.api.get(endpoint);
  }
  saveGeo(data:any){
    let endpoint = 'project/geoCoverage/addGeoCoverage';
    return this.api.postAsJson(endpoint, data);
  }
  getGeoById(id:any){
    let endpoint = 'project/geoCoverage/getGeoCoverageById?id='+id;
    return this.api.get(endpoint);
  }
  delete(id:any){
    let endpoint = 'project/geoCoverage/delete?id='+id+'';
    return this.api.get(endpoint);
  }
}
