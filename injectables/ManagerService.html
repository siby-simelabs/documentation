import { Injectable } from '@angular/core';
import { Api } from '../api.service';

@Injectable({
  providedIn: 'root'
})
export class ManagerService {

  constructor(public api: Api) { }

  getDistrictList(crpId: any) {
    let endpoint = 'project/manger/getAllDistrictLevelManager?projectId=' + crpId + '';
    return this.api.get(endpoint);
  }
  filterDistrict(data: any) {
    let endpoint = 'project/manger/filterDistrictLevel';
    return this.api.post(endpoint, data);
  }
  getAllUsers(crpid: any, ep:string) {
    let endpoint = 'project/manger/'+ep+'?projectId=' + crpid + '';
    return this.api.get(endpoint);
  }
  addDistrict(data: any) {
    let endpoint = 'project/manger/addDistrictLevelManager';
    return this.api.postAsJson(endpoint, data);
  }
  getDistById(id: any) {
    let endpoint = 'project/manger/getDistrictOfficerById?id=' + id + '';
    return this.api.get(endpoint);
  }


  getBlkList(crpId: any) {
    let endpoint = 'project/manger/getAllBlockLevelManager?projectId=' + crpId + '';
    return this.api.get(endpoint);
  }
  filterBlock(data: any) {
    let endpoint = 'project/manger/filterBlockLevel';
    return this.api.post(endpoint, data);
  }
  addBlk(data: any) {
    let endpoint = 'project/manger/addBlockLevelManager';
    return this.api.postAsJson(endpoint, data);
  }
  getBlockById(id: any) {
    let endpoint = 'project/manger/getDistrictOfficerById?id=' + id + '';
    return this.api.get(endpoint);
  }

  getVillList(id: any) {
    let endpoint = 'project/manger/listingOfVillages?approvalFlag=true&projectId=' + id + '';
    return this.api.get(endpoint);
  }
  getAllVillById(id: any, blkid: any) {
    let endpoint = 'project/manger/getVillageByBlock?blockId=' + blkid + '&projectId=' + id + '';
    return this.api.get(endpoint);
  }
  deletVillageList(data: any) {
    let endpoint = 'project/manger/deleteVillages';
    return this.api.post(endpoint, data);
  }
  listAllVillage(blkid: any) {
    let endpoint = 'project/manger/getVillage?blockId=' + blkid + '';
    return this.api.get(endpoint);
  }
  saveVillages(data: any, endpoints: any) {
    let endpoint = 'project/manger/' + endpoints + '';
    return this.api.post(endpoint, data);
  }

  getCrpList(crpId: any) {
    let endpoint = 'project/manger/getAllAssignCrp?projectId=' + crpId + '';
    return this.api.get(endpoint);
  }
  saveCrp(data: any) {
    let endpoint = 'project/manger/addAssignCrp';
    return this.api.post(endpoint, data);
  }

  deleteAssignVillage(personId: any, crpId: any) {
    let endpoint = 'project/manger/deleteCrpAssigned?projectId=' + crpId + '&personId=' + personId + '';
    return this.api.get(endpoint);
  }
  assignVillages(data: any) {
    let endpoint = 'project/manger/addAssignCrp';
    return this.api.post(endpoint, data);
  }
  listAllVillagebyAssign(cpid: any, flag: any, personId: any) {
    if ((flag == "edit") || (flag == "view")) {
      let endpoint = 'project/manger/getAllCrpAssignVillage?projectId=' + cpid + '&personId=' + personId + '';
      return this.api.get(endpoint);
    }
    else {
      let endpoint = 'project/manger/getAllAssignVillage?projectId=' + cpid + '';
      return this.api.get(endpoint);
    }

  }
  getCrpUserKerala(cpid: any) {
    let endpoint = 'project/manger/getCrpUser?projectId=' + cpid + '';
    return this.api.get(endpoint);
  }

  getAllShg(crpId: any) {
    let endpoint = 'project/manger/listShg?projectId=' + crpId + '';
    return this.api.get(endpoint);
  }
  saveshg(data: any) {
    let endpoint = 'project/manger/addShg';
    return this.api.post(endpoint, data);
  }
  deletShg(personId: any, crpId: any) {
    let endpoint = 'project/manger/deleteShgAssigned?projectId=' + crpId + '&personId=' + personId + '';
    return this.api.get(endpoint);
  }
  history(crpId: any, locId: any, ep: any, data?:any) {
    if (ep == "getCrpLevelHistory") {
      let endpoint = 'project/manger/getCrpLevelHistory';
      return this.api.post(endpoint, data);
    }
    else {
      let endpoint = 'project/manger/' + ep + '?projectId=' + crpId + '&locationId=' + locId + '';
      return this.api.get(endpoint);
    }

  }

  approveVillage(blockid: any, crpId: string) {
    let endpoint = 'project/manger/listingOfVillages?projectId=' + crpId + '&approvalFlag=true&blockCode=' + blockid + '';
    return this.api.get(endpoint);
  }


}
