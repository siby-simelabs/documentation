import { Injectable } from '@angular/core';
import { Api } from 'src/app/providers/api.service';
import { CommonService } from 'src/app/providers/common.service';

@Injectable({
  providedIn: 'root'
})
export class OutcomeService {

  constructor(public api: Api, public common:CommonService) { }
   
  getKpi(pageNo: number, pageSize: number, projectId: any) {
    let endpoint = 'admin/kpi/getAllKpiOutcome?';
    endpoint +=
      'pageNo=' +
      pageNo +
      '&' +
      'pageSize=' +
      pageSize;
    return this.api.get(endpoint);
  }
  saveOutcome(data: any) {
    let endpoint = 'admin/kpi/addKpiOutcome';
    return this.api.postAsJson(endpoint, data);
  }
  getOutById(id: any) {
    let endpoint = 'admin/kpi/getKpiOutcomeById?id=' + id;
    return this.api.get(endpoint);
  }
  delete(id: any) {
    let endpoint = 'admin/kpi/deleteKpiOutcome?id=' + id;
    return this.api.get(endpoint);
  }
  saveEvent(data:any){
    let endpoint = 'admin/kpi/addKpiActivity';
    return this.api.postAsFormData(endpoint, data);
  }
  filterKpiOutcome(data:any){
    let endpoint = 'admin/kpi/filterKpiOutcome';
    return this.api.postAsJson(endpoint, data);
  }

}
