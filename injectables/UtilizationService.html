import { Injectable } from '@angular/core';
import { Api } from '../api.service';

@Injectable({
  providedIn: 'root'
})
export class UtilizationService {

  constructor(public api:Api) { }
  getApproved(pageNo: number, pageSize: number){
    let endpoint = 'project/fund/getAllApprovedProject?';
    endpoint += 'pageNo=' + pageNo + '&' + 'pageSize=' + pageSize;
    return this.api.get(endpoint);
  }
  getUtilization(id:any){
    let endpoint = 'project/fund/getAllFundProjectComponent?projectId='+id+'';
    return this.api.get(endpoint);
  }
  saveUtilization(data:any){
    let endpoint = 'project/fund/addFund';
    return this.api.postAsFormData(endpoint, data);
  }
  getFundList(expid:any){
    let endpoint = 'project/fund/getAllUtilization?expenseProvisionId='+expid+'';
    return this.api.get(endpoint);
  }
  genrUc(prid:any){
    let endpoint = 'project/fund/generateCertificate?projectId='+prid+'';
    return this.api.get(endpoint);
  }
  getBudget(prid:any){
    let endpoint = 'project/fund/getBudgetFund?projectId='+prid+'';
    return this.api.get(endpoint);
  }
  getFundById(id:any){
    let endpoint = 'project/fund/getFundById?id='+id+'';
    return this.api.get(endpoint);
  }
  getDelete(id:any){
    let endpoint = 'project/fund/deleteFund?id='+id+'';
    return this.api.get(endpoint);
  }
  filterProject(data:any){
    let endpoint = 'project/fund/filterFundProject';
    return this.api.post(endpoint, data);
  }
  filterComponent(data:any){
    let endpoint = 'project/fund/filterFund';
    return this.api.post(endpoint, data);
  }

}
